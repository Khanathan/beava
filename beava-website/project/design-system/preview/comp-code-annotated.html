# a per-customer analytics dashboard
import beava as bv

@bv.event
class PageView:
    user_id:  str
    path:     str
    category: str

@bv.table(key="user_id")
def Dashboard(e: PageView):
    return e.agg(
        views_total     = bv.count(),                   ← lifetime views
        views_24h       = bv.count(window="24h"),      ← today's activity bar
        distinct_cats_7d= bv.distinct(e.category, window="7d"), ← category breakdown
        last_seen       = bv.latest(e.ts),              ← "last seen" timestamp
    )

bv.App("0.0.0.0:6400").register(PageView, Dashboard).serve()  ← replaces old table, serves new one
