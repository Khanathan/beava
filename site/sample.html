# features.py
import beava as bv

@bv.stream
class Click:
    user_id: str
    item_id: str
    clicked: bool

@bv.table
def rolling_ctr(clicks: Stream[Click]) -> Table[str, float]:
    return clicks.group_by("user_id").window("1h").mean("clicked")

# test_features.py
from beava.testing import test_pipeline
from features import rolling_ctr, Click

def test_rolling_ctr():
    with test_pipeline([rolling_ctr]) as p:
        p.push(Click(user_id="u1", item_id="i1", clicked=True))
        p.push(Click(user_id="u1", item_id="i2", clicked=False))

        assert p.query(rolling_ctr, "u1") == 0.5
