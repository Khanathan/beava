fn collect_rs_files(dir: &Path) -> Vec<PathBuf> {
    let mut out = Vec::new();
    fn visit(dir: &Path, out: &mut Vec<PathBuf>) {
        if let Ok(rd) = std::fs::read_dir(dir) {
            for entry in rd.flatten() {
                let p = entry.path();
                if p.is_dir() { visit(&p, out); }
                else if p.extension().and_then(|s| s.to_str()) == Some("rs") {
                    out.push(p);
                }
            }
        }
    }
    visit(dir, &mut out);
    out
}

    //! coverage invariant.
    //!
    //! **Locked architectural commitment from Phase 12.8 (D-03 hard reject at
    //! register-time):** every op-string registered in
    //! `crates/beava-core/src/agg_compile.rs::op_str_to_agg_kind` MUST have a
    //! corresponding match arm in
    //! `crates/beava-core/src/register_validate.rs::lifetime_bound_for_op_str`
    //! that returns a non-`Unbounded` variant. v0 commits to "users size their
    //! box up front": every operator declares its lifetime memory ceiling at
    //! register-time.
    //!
    //! This is a coverage tripwire — if a future PR adds a new op (extends
    //! `AggKind` and `op_str_to_agg_kind`) without also extending
    //! `lifetime_bound_for_op_str`, this test fails with a list of missing
    //! op-strings. Forces the bound table to stay in lockstep with the op

    //! **Sister phase pattern:** Phase 12.6 Plan 10

    //! `cargo test -p beava-server --test phase12_8_lifetime_ops_have_bounds`.

    use std::path::{Path, PathBuf};

    fn workspace_root() -> PathBuf {

            .parent()

    /// Drop lines whose first non-whitespace characters are `//` (line comments)
    /// or `///` / `//!` (doc comments). This prevents a doc-comment "// "count" =>"
    /// inside register_validate from matching the grep.
    fn strip_line_comments(src: &str) -> String {
        src.lines()
            .filter(|l| {
                let t = l.trim_start();
                !(t.starts_with("//") || t.starts_with("///") || t.starts_with("//!"))
            })
            .collect::<Vec<_>>()
            .join("\n")
    }

    /// Extract op-strings from `op_str_to_agg_kind` (the canonical agg_compile
    /// op-string list). Looks for `"<name>" =>` patterns inside the function
    /// body. Returns a sorted, deduped Vec.
    fn extract_agg_compile_op_strings() -> Vec<String> {
        let path = workspace_root().join("crates/beava-core/src/agg_compile.rs");
        let src = std::fs::read_to_string(&path).expect("agg_compile.rs exists");
        let stripped = strip_line_comments(&src);

        // Locate the function body — find `fn op_str_to_agg_kind` and read until
        // the matching closing brace at the function's depth.
        let fn_start = stripped
            .find("fn op_str_to_agg_kind")
            .expect("op_str_to_agg_kind function exists in agg_compile.rs");
        let body_start = stripped[fn_start..]
            .find('{')
            .expect("function body brace exists")
            + fn_start;
        // Find balanced closing brace.
        let mut depth = 0_usize;
        let mut body_end = body_start;
        for (i, ch) in stripped[body_start..].char_indices() {
            match ch {
                '{' => depth += 1,
                '}' => {
                    depth -= 1;
                    if depth == 0 {
                        body_end = body_start + i;
                        break;
                    }
                }
                _ => {}
            }
        }
        let body = &stripped[body_start..=body_end];

        // Match `"<op_str>" =>` patterns (any non-quote sequence between the
        // quotes; the => follows after optional whitespace).
        let mut ops: Vec<String> = Vec::new();
        let mut chars = body.char_indices().peekable();
        while let Some((i, ch)) = chars.next() {
            if ch == '"' {
                // Find closing quote.
                let start = i + 1;
                let mut end = start;
                while let Some(&(j, c)) = chars.peek() {
                    if c == '"' {
                        end = j;
                        chars.next();
                        break;
                    }
                    chars.next();
                }
                if end > start {
                    let candidate = &body[start..end];
                    // Look ahead: skip whitespace + must be `=>` next.
                    let after = &body[end + 1..];
                    let trimmed = after.trim_start();
                    if trimmed.starts_with("=>") {
                        ops.push(candidate.to_string());
                    }
                }
            }
        }
        ops.sort();
        ops.dedup();
        ops
    }

    fn extract_register_validate_op_strings() -> Vec<String> {
        let path = workspace_root().join("crates/beava-core/src/register_validate.rs");
        let src = std::fs::read_to_string(&path).expect("register_validate.rs exists");
        let stripped = strip_line_comments(&src);

        // Locate the function body.
        let fn_start = stripped
            .find("fn lifetime_bound_for_op_str")
            .expect("lifetime_bound_for_op_str function exists in register_validate.rs");
        let body_start = stripped[fn_start..]
            .find('{')
            .expect("function body brace exists")
            + fn_start;
        let mut depth = 0_usize;
        let mut body_end = body_start;
        for (i, ch) in stripped[body_start..].char_indices() {
            match ch {
                '{' => depth += 1,
                '}' => {
                    depth -= 1;
                    if depth == 0 {
                        body_end = body_start + i;
                        break;
                    }
                }
                _ => {}
            }
        }
        let body = &stripped[body_start..=body_end];

        // Match `"<op_str>" |` and `"<op_str>" =>` patterns. The match arms
        // chain quoted strings via `|`.
        let mut ops: Vec<String> = Vec::new();
        let mut chars = body.char_indices().peekable();
        while let Some((i, ch)) = chars.next() {
            if ch == '"' {
                let start = i + 1;
                let mut end = start;
                while let Some(&(j, c)) = chars.peek() {
                    if c == '"' {
                        end = j;
                        chars.next();
                        break;
                    }
                    chars.next();
                }
                if end > start {
                    ops.push(body[start..end].to_string());
                }
            }
        }
        // Filter to op-strings that look like kebab/snake op names — drop any
        // string that's clearly NOT an op (e.g. error reasons containing
        // spaces, "n", "k", "samples" kwarg names which appear inside
        // BoundedByRequiredKwarg("n")).
        // Strategy: keep strings matching the rough op-name shape:
        // [a-z][a-z0-9_]+ AND length >= 3 (excludes "n", "k").
        ops.retain(|s| {
            s.len() >= 3
                && s.chars()
                    .all(|c| c.is_ascii_lowercase() || c.is_ascii_digit() || c == '_')
                && s.chars().next().map(|c| c.is_ascii_lowercase()).unwrap_or(false)
        });
        ops.sort();
        ops.dedup();
        ops
    }

    /// `agg_compile::op_str_to_agg_kind` MUST appear in
    /// `register_validate::lifetime_bound_for_op_str`.
    #[test]
    fn every_op_str_in_agg_compile_has_bound_in_register_validate() {
        let agg_compile_ops = extract_agg_compile_op_strings();
        let register_validate_ops = extract_register_validate_op_strings();

        );

            .collect();

             registers these op-strings but register_validate.rs::\
             lifetime_bound_for_op_str does NOT classify them: {missing:?}.\n\n\
             Fix: add a match arm for each missing op-string in \
             lifetime_bound_for_op_str returning the appropriate \
             OpLifetimeBound variant. See \
             .planning/phases/12.8-memory-governance/12.8-CONTEXT.md \
             memory-bound classes for the canonical mapping.\n\n\
             agg_compile_ops: {agg_compile_ops:?}\n\
             register_validate_ops: {register_validate_ops:?}"
        );
    }

    /// Sanity test: confirm the source-grep extracted at least 30 op-strings
    /// from agg_compile (catches a refactor that moves the function or

    fn sanity_agg_compile_has_at_least_30_op_strings() {
