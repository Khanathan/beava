// crates/beava-core/src/wire.rs — additions on top of Plan 01's module

use bytes::{Buf, BufMut, Bytes, BytesMut};
use thiserror::Error;

/// A decoded frame. `payload` is a cheap-to-clone `Bytes` slice.
///
/// Wire layout (big-endian lengths, content_type is a single byte):
/// ```text
/// [u32 length][u16 op][u8 content_type][payload: length - 3 bytes]
/// ```
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct Frame {
    pub op: u16,
    pub content_type: u8,
    pub payload: Bytes,
}

impl Frame {
    /// Build a Frame owning its payload. Convenience for tests and handlers.
    pub fn new(op: u16, content_type: u8, payload: impl Into<Bytes>) -> Self {
        Self {
            op,
            content_type,
            payload: payload.into(),
        }
    }
}

/// Size of the length prefix in bytes.
const LEN_PREFIX_BYTES: usize = 4;
/// Size of the header that follows the length prefix (op + content_type).
const HEADER_AFTER_LEN_BYTES: usize = 3;

/// Codec-level errors. Handler-level errors (unknown_op / op_not_implemented /
/// unsupported_content_type) live in Plan 04's tcp.rs — those are policy choices
/// made AFTER the frame has been parsed successfully.
#[derive(Debug, Error, PartialEq, Eq)]
pub enum FrameError {
    #[error("frame length {declared_len} exceeds limit {limit} (max_frame_bytes + {} for op+ct)", HEADER_AFTER_LEN_BYTES)]
    TooLarge { declared_len: u32, limit: u32 },
    #[error("frame length {declared_len} < {HEADER_AFTER_LEN_BYTES}: cannot cover op + content_type")]
    LengthUnderflow { declared_len: u32 },
}

/// Append a frame to `out`. Never fails in v0 — the payload size is bounded by
/// the caller (the handler; ultimately by `max_frame_bytes`). Writing a frame
/// with payload.len() > u32::MAX - 3 would wrap; handlers check this before
/// calling encode_frame. For pragmatism, encode_frame does a debug_assert on
/// the size.
pub fn encode_frame(frame: &Frame, out: &mut BytesMut) {
    let payload_len = frame.payload.len();
    debug_assert!(
        payload_len <= (u32::MAX as usize) - HEADER_AFTER_LEN_BYTES,
        "payload too large for u32 length prefix"
    );
    let total_len = (payload_len + HEADER_AFTER_LEN_BYTES) as u32;
    out.reserve(LEN_PREFIX_BYTES + total_len as usize);
    out.put_u32(total_len);                 // big-endian by default in `bytes`
    out.put_u16(frame.op);
    out.put_u8(frame.content_type);
    out.extend_from_slice(&frame.payload);
}

/// Attempt to decode one frame from `buf`. Returns:
///   Ok(Some(Frame))  — a complete frame was consumed; buf advanced past it.
///   Ok(None)         — not enough bytes yet; buf unchanged. Caller reads more.
///   Err(FrameError)  — protocol violation; caller writes an error response and
///                      (for TooLarge) closes the connection.
///
/// `max_frame_bytes` is the configured maximum payload size (CONTEXT.md D-01).
/// The frame's declared length includes op + content_type; a payload of exactly
/// `max_frame_bytes` bytes is accepted (declared_len = max_frame_bytes + 3).
pub fn decode_frame(buf: &mut BytesMut, max_frame_bytes: u32) -> Result<Option<Frame>, FrameError> {
    if buf.len() < LEN_PREFIX_BYTES {
        return Ok(None);
    }

    // Peek the length without advancing
    let declared_len = u32::from_be_bytes([buf[0], buf[1], buf[2], buf[3]]);

    if declared_len < HEADER_AFTER_LEN_BYTES as u32 {
        // Length can't even cover op + content_type — kill the frame.
        // We don't advance the cursor: Plan 04's handler decides to write an
        // error and (depending on the variant) close the connection.
        return Err(FrameError::LengthUnderflow { declared_len });
    }

    let limit = max_frame_bytes.saturating_add(HEADER_AFTER_LEN_BYTES as u32);
    if declared_len > limit {
        return Err(FrameError::TooLarge { declared_len, limit });
    }

    let total_needed = LEN_PREFIX_BYTES + declared_len as usize;
    if buf.len() < total_needed {
        return Ok(None);                    // wait for more bytes
    }

    // Consume the frame atomically.
    buf.advance(LEN_PREFIX_BYTES);           // drop length prefix
    let op = buf.get_u16();                  // advances 2
    let content_type = buf.get_u8();         // advances 1
    let payload_len = declared_len as usize - HEADER_AFTER_LEN_BYTES;
    let payload = buf.split_to(payload_len).freeze();    // zero-copy-ish

    Ok(Some(Frame { op, content_type, payload }))
}
```

### Proptest strategy

```rust
use proptest::prelude::*;

fn arb_frame() -> impl Strategy<Value = Frame> {
    (any::<u16>(), any::<u8>(), proptest::collection::vec(any::<u8>(), 0..10_000))
        .prop_map(|(op, ct, payload)| Frame {
            op,
            content_type: ct,
            payload: Bytes::from(payload),
        })
}

proptest! {
    #[test]
    fn round_trip_is_byte_identical(frame in arb_frame()) {
        let mut buf = BytesMut::new();
        encode_frame(&frame, &mut buf);
        let out = decode_frame(&mut buf, /*max_frame_bytes=*/ 16 * 1024 * 1024).unwrap();
        prop_assert_eq!(out, Some(frame));
        prop_assert_eq!(buf.len(), 0, "buf fully drained");
    }

    #[test]
    fn two_frames_concatenated_decode_in_order(a in arb_frame(), b in arb_frame()) {
        let mut buf = BytesMut::new();
        encode_frame(&a, &mut buf);
        encode_frame(&b, &mut buf);
        let fa = decode_frame(&mut buf, 16 * 1024 * 1024).unwrap();
        let fb = decode_frame(&mut buf, 16 * 1024 * 1024).unwrap();
        prop_assert_eq!(fa, Some(a));
        prop_assert_eq!(fb, Some(b));
        prop_assert_eq!(buf.len(), 0);
    }

    #[test]
    fn streaming_partial_reads_reassemble_correctly(frame in arb_frame()) {
        let mut full = BytesMut::new();
        encode_frame(&frame, &mut full);
        let bytes: Vec<u8> = full.to_vec();

        // Feed one byte at a time into a streaming buffer; exactly one decode
        // should succeed at the end, and every earlier call should return None.
        let mut streaming = BytesMut::new();
        for (i, b) in bytes.iter().enumerate() {
            streaming.extend_from_slice(&[*b]);
            let result = decode_frame(&mut streaming, 16 * 1024 * 1024).unwrap();
            if i + 1 < bytes.len() {
                prop_assert!(result.is_none(), "premature decode at byte {}", i);
            } else {
                prop_assert_eq!(result, Some(frame.clone()));
                prop_assert_eq!(streaming.len(), 0);
            }
        }
    }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| `&mut BytesMut` from Plan 04 → decode_frame | Hostile bytes from remote clients land in this buffer; decode_frame must not panic or allocate unboundedly |
| Codec → FrameError variants | Error enum is part of the stability contract — Plan 04 pattern-matches on it |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02.5-02-01 | DoS | Declared_len near u32::MAX triggers allocator OOM | mitigate | `declared_len > max_frame_bytes + HEADER_AFTER_LEN_BYTES` check uses `saturating_add`; any declared_len above the limit short-circuits BEFORE `buf.len() < total_needed` is evaluated, so we never attempt to allocate / reserve based on attacker-chosen lengths. `decode_saturating_limit_no_overflow` test proves no panic at u32::MAX. |
| T-02.5-02-02 | DoS | Slowloris-style byte-drip attack | accept | decode_frame returns Ok(None) on incomplete frames — the buffer grows up to `max_frame_bytes + 4` bytes per connection, bounded. Plan 04 decides when a connection has been idle too long (tokio read timeout); not a codec concern. |
| T-02.5-02-03 | Tampering | Crafted frame with declared_len < 3 | mitigate | `LengthUnderflow` returned before any cursor advance; Plan 04 writes the `malformed_frame` error and closes the connection. Test `decode_length_prefix_of_0_returns_underflow` covers. |
| T-02.5-02-04 | Tampering | Endian confusion (little-endian client) | mitigate | Codec is strictly big-endian per D-01; `encode_frame_big_endian_byte_layout` test asserts exact bytes. Clients using LE will produce declared_len values that are almost always either 0..3 (underflow) or huge (too_large) → caught. Not a silent-corruption risk. |
| T-02.5-02-05 | Information disclosure | FrameError includes declared_len in the error message | accept | declared_len IS a client-supplied value; echoing it back is fine (no server state leaked). Plan 04 does NOT pass the error message verbatim over the wire — it builds an error response with code "frame_too_large" / "malformed_frame" only, not the internal Display text. |
| T-02.5-02-06 | DoS | Allocation of `total_needed` before `buf.len() < total_needed` check | mitigate | We only call `buf.split_to(payload_len)` AFTER confirming `buf.len() >= total_needed`; no speculative allocation. `BytesMut::split_to` is O(1) in `bytes` v1 (doesn't copy — splits ref-counted slice). |
</threat_model>

<stability_notes>
## Locked for future phases

- **Wire layout**: `[u32 length BE][u16 op BE][u8 content_type][payload]` — LOCKED. Changing this breaks every deployed client.
- **length semantics**: `length` is the count of bytes AFTER the 4-byte length prefix (i.e., `op + content_type + payload = length`). Minimum valid length is 3. Documented in the module comment.
- **`Frame` struct public fields**: `op: u16`, `content_type: u8`, `payload: Bytes`. Using `Bytes` (not `Vec<u8>`) is load-bearing — zero-copy handoff to handlers is assumed by Plan 04. Changing this would require rewriting `handle_register` / `handle_ping`.
- **`FrameError` variants**: `TooLarge { declared_len, limit }`, `LengthUnderflow { declared_len }`. Plan 04 pattern-matches on these. Adding new variants is forward-compatible; renaming breaks Plan 04.
- **`decode_frame` streaming contract**: `Ok(None)` means "need more bytes, buf unchanged"; `Ok(Some(f))` means "consumed one full frame, buf advanced"; `Err(...)` means "protocol violation, buf unchanged (cursor not advanced)". Plan 04 relies on the no-advance-on-error behavior to compose error handling: write error response, then close.

## Not locked

- Whether to adopt `tokio_util::codec::LengthDelimitedCodec` later (D-08 Claude's Discretion): we chose hand-rolled for simplicity + zero-new-tokio-util-dep. If Phase 13 perf requires it we can swap in LDC behind the same `encode_frame` / `decode_frame` signature.
- Proptest iteration count (256 default; Phase 13 may bump to 4096 for soak testing).
</stability_notes>
