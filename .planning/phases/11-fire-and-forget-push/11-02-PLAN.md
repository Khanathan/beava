       ```python
       def encode_push_binary(stream_name: str, event: dict) -> bytes:
           """Encode PUSH payload in the Phase 11 binary format.

           Format: [u16 name_len][name][u16 field_count][field...]
               field := [u16 key_len][key][u8 type_tag][value]
               type tags: NULL=0, BOOL=1 (1B), I64=2 (8B BE), F64=3 (8B BE), STR=4 ([u16 len][utf-8])

           Raises ProtocolError for unsupported value types.
           """
           buf = bytearray()
           name_bytes = stream_name.encode("utf-8")
           buf += _U16.pack(len(name_bytes))
           buf += name_bytes
           buf += _U16.pack(len(event))
           for key, value in event.items():
               key_bytes = key.encode("utf-8")
               buf += _U16.pack(len(key_bytes))
               buf += key_bytes
               if value is None:
                   buf.append(TYPE_NULL)
               elif isinstance(value, bool):  # MUST come before int check
                   buf.append(TYPE_BOOL)
                   buf.append(0x01 if value else 0x00)
               elif isinstance(value, int):
                   # Range-check: i64 is [-2^63, 2^63-1]
                   if value < -(1 << 63) or value > (1 << 63) - 1:
                       raise ProtocolError(
                           f"integer field {key!r} out of i64 range: {value}"
                       )
                   buf.append(TYPE_I64)
                   buf += _I64.pack(value)
               elif isinstance(value, float):
                   if value != value or value == float("inf") or value == float("-inf"):
                       raise ProtocolError(
                           f"float field {key!r} is not finite: {value}"
                       )
                   buf.append(TYPE_F64)
                   buf += _F64.pack(value)
               elif isinstance(value, str):
                   v_bytes = value.encode("utf-8")
                   buf.append(TYPE_STR)
                   buf += _U16.pack(len(v_bytes))
                   buf += v_bytes
               else:
                   raise ProtocolError(
                       f"unsupported event field type for key {key!r}: {type(value).__name__}"
                   )
           return bytes(buf)
       ```

       ```python
       def drain_errors_nonblock(self) -> None:
           """Non-blocking readability probe. Surfaces any pending server error frame.

           Called by App before every user-facing operation (push, push_sync, flush, get).
           Reads at most one frame per call. A stored pending error is raised first.
           """
           if self._pending_error is not None:
               err, self._pending_error = self._pending_error, None
               raise err

           if self._sock is None:
               return

           try:
               ready, _, _ = select.select([self._sock], [], [], 0)
           except (OSError, ValueError):
               # Socket may be in a bad state; let the next real op surface it.
               return

           if not ready:
               return

           try:
               status, payload = self._recv_frame()
           except ConnectionError:
               # The connection is dead; next real op will reconnect and error.
               self._sock = None
               return

           if status == STATUS_ERROR:
               raise ProtocolError(payload.decode("utf-8", errors="replace"))
           # status == STATUS_OK: discard; it's a stray ACK from a prior path.
       ```
