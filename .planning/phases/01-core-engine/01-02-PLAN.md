```rust
use std::time::SystemTime;
use serde::{Serialize, Deserialize};
use crate::types::FeatureValue;
use crate::error::TallyError;
use super::window::RingBuffer;

/// Trait implemented by all streaming operators.
/// - `push` processes an incoming event. Called once per event per operator.
/// - `read` returns the current aggregate value. Called to collect features.
pub trait Operator: std::fmt::Debug {
    fn push(&mut self, event: &serde_json::Value, now: SystemTime) -> Result<(), TallyError>;
    fn read(&self, now: SystemTime) -> FeatureValue;
}
```

Implement CountOp:
```rust
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct CountOp {
    buffer: RingBuffer<u64>,
}

impl CountOp {
    pub fn new(window_duration: std::time::Duration, bucket_duration: std::time::Duration) -> Self {
        Self {
            buffer: RingBuffer::new(window_duration, bucket_duration),
        }
    }
}

impl Operator for CountOp {
    fn push(&mut self, _event: &serde_json::Value, now: SystemTime) -> Result<(), TallyError> {
        // count needs no field -- always succeeds regardless of event shape (CONTEXT.md)
        self.buffer.add_to_current(1u64, now);
        Ok(())
    }

    fn read(&self, now: SystemTime) -> FeatureValue {
        // Must call advance_to to expire old buckets before summing.
        // But read takes &self, not &mut self. Two approaches:
        // (a) Clone buffer, advance, sum -- but that's wasteful.
        // (b) Store a "last_read" view or make read take &mut self.
        // Decision: read() should use the buffer's current state. The caller
        // (pipeline engine) is responsible for calling advance_to via push
        // before reading. For read-only queries (GET without PUSH), the
        // pipeline engine calls a separate advance method.
        // For now: sum all buckets. The advance_to is called in push.
        let total = self.buffer.sum_all();
        if total == 0 {
            FeatureValue::Missing  // Zero events in window -> Missing (CONTEXT.md)
        } else {
            FeatureValue::Int(total as i64)
        }
    }
}
```

IMPORTANT DESIGN NOTE: The `read` method needs to handle time-based expiration. Since `read` takes `&self` but `advance_to` takes `&mut self`, we need to design this carefully. The recommended approach: add an `advance(&mut self, now: SystemTime)` method to the Operator trait that the pipeline engine calls before read. Alternatively, make `read` take `&mut self`. Choose whichever approach is cleaner -- the key constraint is that `read` must return a value reflecting only events within the current window, not stale data.

Write unit tests in `#[cfg(test)] mod tests` at the bottom of operators.rs. Use `serde_json::json!({})` for events since count ignores content.

SumOp:
```rust
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SumOp {
    field: String,
    buffer: RingBuffer<f64>,
    optional: bool,
}

impl SumOp {
    pub fn new(field: impl Into<String>, window_duration: Duration, bucket_duration: Duration, optional: bool) -> Self {
        Self {
            field: field.into(),
            buffer: RingBuffer::new(window_duration, bucket_duration),
            optional,
        }
    }
}

impl Operator for SumOp {
    fn push(&mut self, event: &serde_json::Value, now: SystemTime) -> Result<(), TallyError> {
        match event.get(&self.field) {
            None => {
                if self.optional {
                    Ok(())  // optional=true: absent field -> skip silently
                } else {
                    Err(TallyError::Type {
                        field: self.field.clone(),
                        expected: "numeric".into(),
                        got: "absent".into(),
                    })
                }
            }
            Some(val) => {
                // Extract numeric value. Int or Float accepted, String -> type error.
                if let Some(f) = val.as_f64() {
                    self.buffer.add_to_current(f, now);
                    Ok(())
                } else {
                    Err(TallyError::Type {
                        field: self.field.clone(),
                        expected: "numeric".into(),
                        got: format!("{}", val),
                    })
                }
            }
        }
    }

    fn read(&self, _now: SystemTime) -> FeatureValue {
        // Check if any events contributed to the sum.
        // Use count_nonzero to detect if any bucket has data.
        if self.buffer.count_nonzero() == 0 {
            FeatureValue::Missing  // Zero events -> Missing
        } else {
            FeatureValue::Float(self.buffer.sum_all())
        }
    }
}
```

AvgOp -- uses paired ring buffers (count + sum) per RESEARCH.md Pattern and ENG-05:
```rust
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct AvgOp {
    field: String,
    count_buffer: RingBuffer<u64>,
    sum_buffer: RingBuffer<f64>,
    optional: bool,
}

impl AvgOp {
    pub fn new(field: impl Into<String>, window_duration: Duration, bucket_duration: Duration, optional: bool) -> Self {
        Self {
            field: field.into(),
            count_buffer: RingBuffer::new(window_duration, bucket_duration),
            sum_buffer: RingBuffer::new(window_duration, bucket_duration),
            optional,
        }
    }
}

impl Operator for AvgOp {
    fn push(&mut self, event: &serde_json::Value, now: SystemTime) -> Result<(), TallyError> {
        match event.get(&self.field) {
            None => {
                if self.optional {
                    Ok(())
                } else {
                    Err(TallyError::Type {
                        field: self.field.clone(),
                        expected: "numeric".into(),
                        got: "absent".into(),
                    })
                }
            }
            Some(val) => {
                if let Some(f) = val.as_f64() {
                    self.count_buffer.add_to_current(1u64, now);
                    self.sum_buffer.add_to_current(f, now);
                    Ok(())
                } else {
                    Err(TallyError::Type {
                        field: self.field.clone(),
                        expected: "numeric".into(),
                        got: format!("{}", val),
                    })
                }
            }
        }
    }

    fn read(&self, _now: SystemTime) -> FeatureValue {
        let count = self.count_buffer.sum_all();
        if count == 0 {
            FeatureValue::Missing  // Zero events -> Missing, not NaN (CONTEXT.md)
        } else {
            let sum = self.sum_buffer.sum_all();
            FeatureValue::Float(sum / count as f64)
        }
    }
}
```
