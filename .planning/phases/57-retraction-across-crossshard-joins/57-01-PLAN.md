## ShardOp::RetractDownstream dispatch arm logic (D-B1 + D-B3 + D-B4)

```rust
ShardOp::RetractDownstream { stream_name, row_key, reason, depth, target_shard } => {
    if depth >= MAX_RETRACTION_DEPTH {
        metrics::counter!(RETRACTION_DEPTH_EXCEEDED_TOTAL).increment(1);
        let _ = response_tx.send(ShardResult::RetractOk(RetractOutcome::DepthExceeded));
        return;
    }
    let outcome = shard.apply_retraction(&stream_name, &row_key, &reason, depth);
    match outcome {
        RetractOutcome::Retracted => metrics::counter!(RETRACTIONS_APPLIED_TOTAL,
            "operator" => stream_name.clone()).increment(1),
        RetractOutcome::NoOp => metrics::counter!(RETRACTIONS_NOOPED_TOTAL,
            "operator" => stream_name.clone()).increment(1),
        RetractOutcome::BeyondHistory => metrics::counter!(RETRACTION_BEYOND_HISTORY_TOTAL,
            "operator" => stream_name.clone()).increment(1),
        RetractOutcome::DepthExceeded => metrics::counter!(RETRACTION_DEPTH_EXCEEDED_TOTAL).increment(1),
    }
    let _ = response_tx.send(ShardResult::RetractOk(outcome));
}
```

## Shard::apply_retraction body (D-B4 idempotency + D-C1 history_ttl)

```rust
pub fn apply_retraction(
    &mut self,
    stream_name: &str,
    row_key: &str,
    reason: &RetractReason,
    depth: u8,
) -> RetractOutcome {
    if depth >= MAX_RETRACTION_DEPTH {
        return RetractOutcome::DepthExceeded;
    }
    // 1. Idempotency probe: if row missing OR already-tombstoned, NoOp
    let Some(entity) = self.read_entity(row_key) else {
        return RetractOutcome::NoOp;
    };
    let Some(stream) = entity.streams.get(stream_name) else {
        return RetractOutcome::NoOp;
    };
    if stream.is_tombstoned() /* or a dedicated tombstoned field */ {
        return RetractOutcome::NoOp;
    }
    // 2. history_ttl probe: extract event_time from stream.last_event_at OR
    //    contributing_inputs.primary_event_id -> decode embedded epoch_ms
    //    (D-A3: (shard_id: u16, epoch_ms: u48) packed into u64)
    let history_ttl = self.resolve_history_ttl(stream_name);
    let event_ts = self.resolve_event_timestamp(&entity, stream_name);
    let wm = self.watermark.global_max_event_time();
    if event_ts + history_ttl < wm {
        return RetractOutcome::BeyondHistory;
    }
    // 3. Happy path: tombstone the row and record the retraction reason.
    self.tombstone_stream_row(row_key, stream_name, reason);
    RetractOutcome::Retracted
}
```

/// shard that owns the affected downstream row. Same-shard fast path invokes

///   - Ok(RetractOutcome::NoOp) — already-retracted / never-existed (idempotent)
///   - Ok(RetractOutcome::BeyondHistory) — event_time < watermark - history_ttl; skipped
///   - Ok(RetractOutcome::DepthExceeded) — depth >= 16 (D-B5)

///

/// Same 3-point proof — source never try_sends to own inbox; try_send non-blocking;
/// target drains on its own pinned thread.
