**Constants:**
```rust
const HLL_P: usize = 14;              // Precision: 14 bits (locked decision)
const HLL_M: usize = 1 << HLL_P;     // 16384 registers
const HLL_ALPHA: f64 = 0.7213 / (1.0 + 1.079 / HLL_M as f64);
```

**Hll struct:**
```rust
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Hll {
    registers: Vec<u8>,  // Vec<u8> for Clone, not [u8; 16384] which is Copy but 16KB memcpy
}

impl Default for Hll {
    fn default() -> Self { Self { registers: vec![0u8; HLL_M] } }
}

impl Hll {
    pub fn new() -> Self { Self::default() }

    pub fn insert(&mut self, value: &str) {
        let hash = hash_value(value);
        let index = (hash >> (64 - HLL_P)) as usize;   // Top 14 bits -> register index
        let remaining = (hash << HLL_P) | (1 << (HLL_P - 1));  // Set guard bit
        let leading_zeros = remaining.leading_zeros() as u8 + 1;
        self.registers[index] = self.registers[index].max(leading_zeros);
    }

    pub fn count(&self) -> f64 {
        let sum: f64 = self.registers.iter()
            .map(|&r| 2.0_f64.powi(-(r as i32)))
            .sum();
        let raw = HLL_ALPHA * (HLL_M as f64) * (HLL_M as f64) / sum;
        // Small range correction (linear counting)
        if raw <= 2.5 * HLL_M as f64 {
            let zeros = self.registers.iter().filter(|&&r| r == 0).count();
            if zeros > 0 {
                return (HLL_M as f64) * (HLL_M as f64 / zeros as f64).ln();
            }
        }
        raw
    }

    pub fn merge(&mut self, other: &Hll) {
        for (a, &b) in self.registers.iter_mut().zip(other.registers.iter()) {
            *a = (*a).max(b);
        }
    }

    pub fn is_empty(&self) -> bool {
        self.registers.iter().all(|&r| r == 0)
    }
}
```

**DistinctCountOp** uses a manual ring buffer of Hll sketches (NOT RingBuffer<Hll>) because:
- RingBuffer<Hll> would Clone 16KB sketches on bucket rotation
- Manual ring buffer can use `Hll::default()` (which allocates a new Vec) for clearing

```rust
use crate::engine::operators::Operator;
use crate::types::FeatureValue;
use crate::error::TallyError;
use std::time::{Duration, SystemTime};

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct DistinctCountOp {
    field: String,
    buckets: Vec<Hll>,
    head: usize,
    bucket_duration: Duration,
    window_duration: Duration,
    current_bucket_start: Option<SystemTime>,
    optional: bool,
}

impl DistinctCountOp {
    pub fn new(field: impl Into<String>, window_duration: Duration, bucket_duration: Duration, optional: bool) -> Self {
        let num_buckets = (window_duration.as_secs_f64() / bucket_duration.as_secs_f64()).ceil() as usize;
        Self {
            field: field.into(),
            buckets: (0..num_buckets).map(|_| Hll::new()).collect(),
            head: 0,
            bucket_duration,
            window_duration,
            current_bucket_start: None,
            optional,
        }
    }

    /// Advance to `now`, clearing skipped buckets. Same logic as RingBuffer::advance_to.
    fn advance_to(&mut self, now: SystemTime) {
        let start = match self.current_bucket_start {
            Some(s) => s,
            None => {
                let aligned = self.bucket_start_for(now);
                self.current_bucket_start = Some(aligned);
                self.head = 0;
                return;
            }
        };
        let elapsed = now.duration_since(start).unwrap_or(Duration::ZERO);
        let steps = (elapsed.as_secs_f64() / self.bucket_duration.as_secs_f64()) as usize;
        if steps == 0 { return; }
        let n = self.buckets.len();
        if steps >= n {
            for b in self.buckets.iter_mut() { *b = Hll::new(); }
            self.head = 0;
        } else {
            for i in 1..=steps {
                let idx = (self.head + i) % n;
                self.buckets[idx] = Hll::new();
            }
            self.head = (self.head + steps) % n;
        }
        self.current_bucket_start = Some(self.bucket_start_for(now));
    }

    fn bucket_start_for(&self, time: SystemTime) -> SystemTime {
        let since_epoch = time.duration_since(SystemTime::UNIX_EPOCH).unwrap_or(Duration::ZERO);
        let bucket_secs = self.bucket_duration.as_secs();
        let aligned_secs = (since_epoch.as_secs() / bucket_secs) * bucket_secs;
        SystemTime::UNIX_EPOCH + Duration::from_secs(aligned_secs)
    }
}

impl Operator for DistinctCountOp {
    fn push(&mut self, event: &serde_json::Value, now: SystemTime) -> Result<(), TallyError> {
        match event.get(&self.field) {
            None => {
                if self.optional { Ok(()) }
                else { Err(TallyError::Type { field: self.field.clone(), expected: "string or numeric".into(), got: "absent".into() }) }
            }
            Some(val) => {
                let str_val = match val {
                    serde_json::Value::String(s) => s.clone(),
                    serde_json::Value::Number(n) => n.to_string(),
                    serde_json::Value::Bool(b) => b.to_string(),
                    _ => return Err(TallyError::Type { field: self.field.clone(), expected: "string or numeric".into(), got: format!("{}", val) }),
                };
                self.advance_to(now);
                self.buckets[self.head].insert(&str_val);
                Ok(())
            }
        }
    }

    fn read(&mut self, now: SystemTime) -> FeatureValue {
        self.advance_to(now);
        // Check if all buckets are empty
        if self.buckets.iter().all(|b| b.is_empty()) {
            return FeatureValue::Missing;
        }
        // Merge all non-empty buckets into a single HLL, then count
        // Single-pass merge: iterate register positions, take max across all buckets
        let mut merged = Hll::new();
        for bucket in &self.buckets {
            if !bucket.is_empty() {
                merged.merge(bucket);
            }
        }
        let count = merged.count();
        FeatureValue::Float(count)
    }
}
```
