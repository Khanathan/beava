    ```rust
    use std::sync::OnceLock;
    use std::sync::atomic::{AtomicU64, Ordering};

    /// Plan 19.2-07 (D-07): process-static snapshot of the latest TRACE_AGG
    /// per_kind output. Updated by `apply_event_to_aggregations` whenever
    /// `BEAVA_TRACE_AGG_TIMING=1` is set; read by the optional /debug/op-cost
    /// HTTP route. Empty when tracing has never been enabled in this process.
    pub struct PerKindSnapshot {
        pub captured_at_ms: AtomicU64,
        pub data: parking_lot::Mutex<Vec<(crate::agg_op::AggKind, std::time::Duration, u32)>>,
    }

    static PER_KIND_LATEST: OnceLock<PerKindSnapshot> = OnceLock::new();

    pub fn per_kind_latest() -> &'static PerKindSnapshot {
        PER_KIND_LATEST.get_or_init(|| PerKindSnapshot {
            captured_at_ms: AtomicU64::new(0),
            data: parking_lot::Mutex::new(Vec::new()),
        })
    }
    ```

    ```rust
        // Plan 19.2-07: also snapshot per_kind data for /debug/op-cost endpoint.
        let snap = per_kind_latest();
    ```

    ```rust
    #[derive(Serialize)]
    struct DebugOpCostEntry {
        kind: String,                  // AggKind::Count → "Count"
        tier: u8,                      // 1 / 2 / 3 — derived from a static map
        last_traced_ns: u128,          // duration in nanoseconds
        last_traced_count: u32,
    }

    #[derive(Serialize)]
    struct DebugOpCostResponse {
        ops: Vec<DebugOpCostEntry>,
        captured_at_ms: u64,
    }

    async fn handle_debug_op_cost() -> impl IntoResponse {
        let snap = beava_core::agg_apply::per_kind_latest();
        let captured_at_ms = snap.captured_at_ms.load(std::sync::atomic::Ordering::Relaxed);
        let data = snap.data.lock();
        let ops: Vec<DebugOpCostEntry> = data.iter().map(|(kind, dur, cnt)| DebugOpCostEntry {
            kind: format!("{:?}", kind),
            tier: tier_for(*kind),       // see helper below
            last_traced_ns: dur.as_nanos(),
            last_traced_count: *cnt,
        }).collect();
        (StatusCode::OK, Json(DebugOpCostResponse { ops, captured_at_ms }))
    }

    fn tier_for(kind: beava_core::agg_op::AggKind) -> u8 {
        use beava_core::agg_op::AggKind::*;
        match kind {
            // Tier 1 (38 ops post Plan 19.2-06)
            Count | Sum | Avg | Min | Max | Variance | StdDev | Ratio
            | First | Last | FirstN | LastN | Lag
            | FirstSeen | LastSeen | Age | HasSeen | TimeSince | TimeSinceLastN
            | Streak | MaxStreak | NegativeStreak | FirstSeenInWindow
            | Ewma | EwVar | EwZScore | DecayedSum | DecayedCount | Twa
            | RateOfChange | InterArrivalStats | BurstCount | DeltaFromPrev
            | Trend | TrendResidual | ValueChangeCount | ZScore
            | Histogram | HourOfDayHistogram | DowHourHistogram | SeasonalDeviation
            | MostRecentN | ReservoirSample
            | GeoSpread | DistanceFromHome => 1,
            // Tier 2 (6 ops post Plan 19.2-06 — UniqueCells + GeoEntropy removed)
            OutlierCount | BloomMember | GeoVelocity | GeoDistance | CountDistinct => 2,
            // Tier 3 (9 ops post Plan 19.2-06)
            Percentile | TopK | Entropy | EventTypeMix => 3,
            // Internal: Windowed dispatches to inner kind; map by inner kind in production.
            // For /debug/op-cost we expose Windowed as its own row; report Tier 1 (delegated).
            Windowed => 1,
        }
    }
    ```
