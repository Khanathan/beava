    ```rust
    /// Plan 19.2-03 (D-03): EntityKey hybrid — three storage shapes selected
    /// by group_keys cardinality + value type. Avoids SmallVec build +
    /// CompactString canonicalization on the single-key numeric path.
    #[derive(Debug, Clone)]
    pub enum EntityKeyShape {
        /// Numeric or bool or datetime single-key. The u64 encodes the variant
        /// tag in the high 4 bits so I64(42) and F64(42.0) (different bits)
        /// don't collide. Bit-cast is bijective for non-NaN floats; NaN keys
        /// are rejected upstream (push-time None-return).
        SingleU64(u64),
        /// String single-key. Stores both the FxHash and the original string
        /// so equality compares the strings (not just the hashes — birthday
        /// paradox at ~10⁻¹⁹ for 10M keys, but we don't rely on it).
        SingleStr(u64, compact_str::CompactString),
        /// Compound key (≥2 group_keys columns). Existing SmallVec shape.
        Multi(smallvec::SmallVec<[(compact_str::CompactString, crate::row::Value); 2]>),
    }

    impl EntityKeyShape {
        pub fn from_row(group_keys: &[String], row: &crate::row::Row) -> Option<Self> {
            if group_keys.len() == 1 {
                let key = &group_keys[0];
                match row.get(key)? {
                    crate::row::Value::Null => None,
                    crate::row::Value::Str(s) => Some(Self::SingleStr(
                        Self::hash_str(s.as_str()),
                        s.clone(),
                    )),
                    crate::row::Value::I64(n) => Some(Self::SingleU64(
                        Self::tag_u64(VariantTag::I64, *n as u64),
                    )),
                    crate::row::Value::F64(f) => {
                        if f.is_nan() { return None; }   // NaN → drop event for this agg
                        Some(Self::SingleU64(Self::tag_u64(VariantTag::F64, f.to_bits())))
                    }
                    crate::row::Value::Bool(b) => Some(Self::SingleU64(
                        Self::tag_u64(VariantTag::Bool, *b as u64),
                    )),
                    crate::row::Value::Datetime(ms) => Some(Self::SingleU64(
                        Self::tag_u64(VariantTag::Datetime, *ms as u64),
                    )),
                    _ => None,                          // Bytes/Json/List/Map → drop
                }
            } else {
                // Existing Multi path: same logic as today's EntityKey::from_row,
                // but inserts into Multi(...) instead of EntityKey(...).
                let mut pairs: smallvec::SmallVec<[(compact_str::CompactString, crate::row::Value); 2]>
                    = smallvec::SmallVec::with_capacity(group_keys.len());
                for key in group_keys {
                    let canonical: compact_str::CompactString = match row.get(key) {
                        None => return None,
                        Some(crate::row::Value::Null | crate::row::Value::Bytes(_) |
                             crate::row::Value::Json(_) | crate::row::Value::List(_) |
                             crate::row::Value::Map(_)) => return None,
                        Some(crate::row::Value::Str(s)) => s.clone(),
                        Some(crate::row::Value::I64(n)) => n.to_string().into(),
                        Some(crate::row::Value::F64(f)) if f.is_nan() => return None,
                        Some(crate::row::Value::F64(f)) => format!("{:?}", f).into(),
                        Some(crate::row::Value::Bool(b)) => b.to_string().into(),
                        Some(crate::row::Value::Datetime(ms)) => ms.to_string().into(),
                    };
                    pairs.push((compact_str::CompactString::from(key.as_str()),
                                crate::row::Value::Str(canonical)));
                }
                Some(Self::Multi(pairs))
            }
        }

        #[inline]
        fn hash_str(s: &str) -> u64 {
            use std::hash::{Hash, Hasher};
            let mut h = fxhash::FxHasher::default();
            s.hash(&mut h);
            h.finish()
        }

        #[inline]
        fn tag_u64(tag: VariantTag, payload: u64) -> u64 {
            // Place tag in high 4 bits; payload in low 60.
            // Bool & Datetime tagging keeps zero-collision because tag bits differ.
            ((tag as u64) << 60) | (payload & 0x0FFF_FFFF_FFFF_FFFF)
        }
    }

    #[derive(Debug, Clone, Copy)]
    enum VariantTag { I64 = 1, F64 = 2, Bool = 3, Datetime = 4 }

    impl PartialEq for EntityKeyShape { /* match same-variant compare */ }
    impl Eq for EntityKeyShape {}
    impl std::hash::Hash for EntityKeyShape { /* hash variant tag + payload */ }
    ```

    Note on tag_u64 collision avoidance: with high 4 bits used for tag, payloads up to 60 bits are preserved. Datetime ms-since-epoch ≤ 2^41 (year 9999) → fits. I64 negative payloads keep their sign bits truncated by the mask — collision for two-complement negatives mapping to same low 60 bits is theoretical but limited to keys with absolute value > 2^59. Fraud workloads never hit this. Document the tradeoff.

        // Plan 19.2-03 (Warning 4): DECLARATION ORDER, not sorted. The existing
        // EntityKey::from_row produces order-sensitive keys (column_name+value
