    pub enum Mode { Bootstrap, Catchup, Done }

    pub struct ModeMachine { current: Mode }

    impl ModeMachine {
        pub fn new() -> Self { Self { current: Mode::Bootstrap } }
        pub fn current(&self) -> Mode { self.current }
        pub fn transition_to(&mut self, next: Mode, context: &str) -> Result<(), ModeError> {
            // Valid: Bootstrap->Catchup, Catchup->Done. Everything else = Err.
            let from = self.current;
            let valid = matches!((from, next),
                (Mode::Bootstrap, Mode::Catchup) | (Mode::Catchup, Mode::Done));
            if !valid { return Err(ModeError::InvalidTransition { from, to: next }); }
            tracing::info!("mode: {:?} -> {:?} ({})", from, next, context);
            self.current = next;
            Ok(())
        }
    }

    pub fn check_scope(scope: &Scope, stream: &str, key: &str) -> Result<(), OutOfScopeError> {
        if !scope.streams.iter().any(|s| s == stream) {
            return Err(OutOfScopeError { stream: stream.into(), key: key.into(), reason: "stream not in scope" });
        }
        if let Some(keys) = &scope.keys {
            if !keys.iter().any(|k| k == key) {
                return Err(OutOfScopeError { stream: stream.into(), key: key.into(), reason: "key not in scope.keys" });
            }
        }
        if let Some(prefix) = &scope.key_prefix {
            if !key.starts_with(prefix) {
                return Err(OutOfScopeError { stream: stream.into(), key: key.into(), reason: "key does not match scope.key_prefix" });
            }
        }
        Ok(())
    }

    pub struct FrozenClient { store: ClientStateStore, scope: Scope }
    impl FrozenClient {
        pub fn get(&self, stream: &str, key: &str) -> Result<Option<Event>, OutOfScopeError> {
            check_scope(&self.scope, stream, key)?;
            Ok(self.store.get(stream, key))
        }
    }
