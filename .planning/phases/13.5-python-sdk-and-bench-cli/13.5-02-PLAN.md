Step 1 — Rewrite `python/beava/_app.py` to implement the bv.App class. Required structure:

```python
"""bv.App — Phase 13.5 v0 client core.

Public client class implementing the 7 wire-mapped lifecycle methods documented
in docs/sdk-api/python.md § App class. Transport selection is via URL scheme:
http(s):// → HTTP/JSON, tcp:// → custom-framed TCP, None → embed mode.
"""
from __future__ import annotations

import os
import warnings
from typing import Any
from urllib.parse import urlparse

from beava._errors import BinaryNotFoundError, RegistrationError, ValidationError
from beava._transport import (
    EmbedTransport, HttpTransport, TcpTransport, Transport, make_transport,
)


class App:
    def __init__(
        self,
        url: str | None = None,
        *,
        timeout: float = 30.0,
        test_mode: bool = False,
    ) -> None:
        self._url = url
        self._timeout = timeout
        self._test_mode = test_mode
        self._transport: Transport | None = None
        self._closed = False
        self._entered = False
        # Transport kind classification — used by tests + close() routing.
        if url is None:
            self._transport_kind = "embed"
        else:
            scheme = urlparse(url).scheme
            if scheme in ("http", "https"):
                self._transport_kind = "http"
                if test_mode:
                    warnings.warn(
                        "test_mode kwarg ignored for network mode (url is set); "
                        "server controls test mode in network mode per Phase 13.4 D-03.",
                        UserWarning,
                        stacklevel=2,
                    )
            elif scheme == "tcp":
                self._transport_kind = "tcp"
                if test_mode:
                    warnings.warn(
                        "test_mode kwarg ignored for network mode (url is set); "
                        "server controls test mode in network mode per Phase 13.4 D-03.",
                        UserWarning,
                        stacklevel=2,
                    )
            else:
                raise ValueError(f"unsupported URL scheme: {scheme!r}")

    def __enter__(self) -> "App":
        self._entered = True
        self._transport = make_transport(
            url=self._url, timeout=self._timeout, test_mode=self._test_mode
        )
        return self

    def __exit__(self, *_exc: object) -> None:
        self.close()

    def __del__(self) -> None:
        try:
            self.close()
        except Exception:
            pass

    def close(self) -> None:
        if self._closed:
            return
        if self._transport is not None:
            self._transport.close()
        self._closed = True

    def _require_transport(self) -> Transport:
        if self._closed:
            raise RuntimeError("App is closed")
        if self._transport_kind == "embed" and not self._entered:
            raise RuntimeError(
                "Embed mode requires use as a context manager: `with bv.App() as app:`"
            )
        if self._transport is None:
            self._transport = make_transport(
                url=self._url, timeout=self._timeout, test_mode=self._test_mode
            )
        return self._transport

    # 7 wire-mapped methods follow.
    def register(self, *descriptors: Any, force: bool = False, dry_run: bool = False) -> dict[str, Any]:
        t = self._require_transport()
        return t.send_register(descriptors=descriptors, force=force, dry_run=dry_run)

    def push(self, event_name: str, fields: dict[str, Any]) -> dict[str, Any]:
        t = self._require_transport()
        return t.send_push(event_name=event_name, fields=fields)

    def get(self, table: str, key: str | list[str | int | bool] | None = None) -> dict[str, Any]:
        # Per ADR-003: 1-arg call (no key) routes to global table via empty-string entity_id.
        t = self._require_transport()
        if key is None:
            return t.send_get(table=table, key="")
        return t.send_get(table=table, key=key)

    def batch_get(
        self,
        requests: list[tuple[str, str | list[str | int | bool]]],
    ) -> list[dict[str, Any]]:
        t = self._require_transport()
        return t.send_batch_get(requests=list(requests))

    def reset(self) -> None:
        t = self._require_transport()
        return t.send_reset()

    def ping(self) -> dict[str, Any]:
        t = self._require_transport()
        return t.send_ping()
```
