fn test_workload_exercises_diverse_op_families() {
    let f = workloads::load_by_name("fraud").unwrap();
    let op_names: std::collections::HashSet<&str> =
        f.derivations.iter().flat_map(|d| d.op_kinds()).collect();
    let families = ["geo", "sketch", "recency", "decay", "core"];
    for fam in families {
        let has_family_op = op_names.iter().any(|op| op.contains(fam) || matches_family(op, fam));
        assert!(has_family_op, "fraud workload missing {} family op", fam);
    }
}

fn matches_family(op: &str, fam: &str) -> bool {
    let mapping: &[(&str, &[&str])] = &[
        ("core", &["count", "sum", "mean", "min", "max", "var", "std"]),
        ("sketch", &["n_unique", "quantile", "top_k", "entropy", "bloom_member"]),
        ("recency", &["first_seen", "last_seen", "age", "streak", "time_since"]),
        ("decay", &["ewma", "ewvar", "ew_zscore", "decayed_sum", "decayed_count", "twa"]),
        ("geo", &["geo_velocity", "geo_distance", "geo_spread", "distance_from_home"]),
    ];
    mapping.iter()
        .find(|(f, _)| *f == fam)
        .map(|(_, ops)| ops.iter().any(|o| op.contains(o)))
        .unwrap_or(false)
}

//! Workload registry — Phase 13.5 Plan 09.

use anyhow::{anyhow, Result};
use serde_json::Value;

pub mod adtech;
pub mod fraud;
pub mod ecommerce;

/// A workload bundles a register payload + event generator.
pub struct Workload {
    pub name: String,
    pub register_payload: Vec<Value>, // JSON descriptors per docs/wire-spec.md OP_REGISTER
    pub derivations: Vec<DerivationInfo>,
    pub event_generator: Box<dyn Fn(u64) -> Box<dyn Iterator<Item = GeneratedEvent> + Send> + Send + Sync>,
}

#[derive(Debug, Clone)]
pub struct DerivationInfo {
    pub name: String,
    pub op_chain: Vec<String>, // op kinds for family-coverage testing
}

impl DerivationInfo {
    pub fn op_kinds(&self) -> impl Iterator<Item = &str> {
        self.op_chain.iter().map(String::as_str)
    }
}

#[derive(Debug, Clone)]
pub struct GeneratedEvent {
    pub event_name: String,
    pub fields: serde_json::Map<String, Value>,
}

pub fn load_by_name(name: &str) -> Result<Workload> {
    match name {

}

use rand::{rngs::StdRng, Rng, SeedableRng};
use rand::distributions::WeightedIndex;
use rand::prelude::*;
use super::{Workload, DerivationInfo, GeneratedEvent};

pub fn build_adtech_workload() -> Workload {
    let register_payload = vec![
        // 3 events
        json!({"kind": "event", "name": "Impression", "fields": [
            {"name": "user_id", "type": "str"},
            {"name": "campaign_id", "type": "str"},
            {"name": "page", "type": "str"},
        ]}),
        json!({"kind": "event", "name": "Click", "fields": [
            {"name": "user_id", "type": "str"},
            {"name": "campaign_id", "type": "str"},
        ]}),
        json!({"kind": "event", "name": "Conversion", "fields": [
            {"name": "user_id", "type": "str"},
            {"name": "campaign_id", "type": "str"},
            {"name": "value", "type": "float"},
        ]}),
        // ~12 derivations using diverse families.
        // Per-user features:
        json!({"kind": "derivation", "name": "UserImpressionsLastHour", "output_kind": "table",

fn derivation_info_from_register(payload: &[Value]) -> Vec<DerivationInfo> {
    payload.iter()
        .filter(|d| d["kind"] == "derivation")
        .map(|d| {
            let name = d["name"].as_str().unwrap_or_default().to_string();
            let mut ops: Vec<String> = vec![];
            if let Some(chain) = d["spec"]["chain"].as_array() {
                for step in chain {
                    if let Some(aggs) = step["aggs"].as_object() {
                        for (_, v) in aggs {
                            if let Some(op) = v["op"].as_str() {
                                ops.push(op.to_string());
                            }
                        }
                    }
                }
            }
            DerivationInfo { name, op_chain: ops }
        })
        .collect()
}

struct AdtechIter {
    rng: StdRng,
    remaining: u64,
    n_users: usize,
    n_campaigns: usize,
    pages: Vec<&'static str>,
    type_dist: WeightedIndex<usize>,
    types: [&'static str; 3],
}

impl AdtechIter {
    fn new(remaining: u64) -> Self {
        Self {
            rng: StdRng::seed_from_u64(42),
            remaining,
            n_users: 5_000,
            n_campaigns: 50,
            pages: vec!["/home","/pricing","/blog","/docs","/checkout","/landing/a","/landing/b","/promo"],
            type_dist: WeightedIndex::new([60, 30, 10]).unwrap(),
            types: ["Impression", "Click", "Conversion"],
        }
    }
}

impl Iterator for AdtechIter {
    type Item = GeneratedEvent;
    fn next(&mut self) -> Option<Self::Item> {
        if self.remaining == 0 { return None; }
        self.remaining -= 1;
        let kind = self.types[self.type_dist.sample(&mut self.rng)];
        let mut fields = serde_json::Map::new();
        let user_id = format!("u_{}", self.rng.gen_range(0..self.n_users));
        let campaign_id = format!("c_{}", self.rng.gen_range(0..self.n_campaigns));
        fields.insert("user_id".into(), Value::String(user_id));
        fields.insert("campaign_id".into(), Value::String(campaign_id));
        match kind {
            "Impression" => {
                fields.insert("page".into(), Value::String((*self.pages.choose(&mut self.rng).unwrap()).into()));
            }
            "Conversion" => {
                let v = (-self.rng.gen::<f64>().ln() * 50.0).abs(); // exp-shaped
                fields.insert("value".into(), serde_json::Number::from_f64(v).map(Value::Number).unwrap_or(Value::Null));
            }
            _ => {} // Click has no extra fields
        }
        Some(GeneratedEvent { event_name: kind.into(), fields })
    }
}
