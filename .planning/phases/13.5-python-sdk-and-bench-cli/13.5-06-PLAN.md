**Generate the 3 bundled demo datasets** (~3MB total per D-02). Each dataset is a small but realistic stream of events that exercises the operator catalogue — not just `bv.count`. `bv.demo("adtech")` etc. become functional after this plan.

Dataset shapes:
- **adtech** (~10K events): `Impression(user_id, campaign_id, page, ts)` + `Click(user_id, campaign_id, ts)` + `Conversion(user_id, campaign_id, value, ts)`. Exercises `count`, `n_unique` (campaign breadth), `time_since` (last click latency), `quantile` (conversion value distribution), `decayed_count` (recent activity).
- **fraud** (~10K events): `Txn(user_id, amount, merchant, ip, lat, lon, ts)` + `Login(user_id, ip, device_id, ts)`. Exercises `sum`, `n_unique`, `geo_velocity`, `geo_distance`, `quantile`, `streak`, `histogram`.
- **ecommerce** (~10K events): `PageView(user_id, page, ts)` + `AddToCart(user_id, sku, price, ts)` + `Purchase(user_id, total, n_items, ts)`. Exercises `count`, `top_k` (popular pages), `last_n`, `histogram`, `inter_arrival_stats`, `event_type_mix`.

```python
    assert "PageView" in names
    assert "AddToCart" in names
    assert "Purchase" in names


def test_event_lines_are_valid_json() -> None:
    """Every events.jsonl line must parse as JSON."""
    import importlib.resources
    for name in ("adtech", "fraud", "ecommerce"):
        ref = importlib.resources.files("beava.demos") / name / "events.jsonl"
        with ref.open("r") as f:
            for i, line in enumerate(f):
                line = line.strip()
                if not line:
                    continue
                try:
                    json.loads(line)
                except json.JSONDecodeError as e:
                    pytest.fail(f"{name}/events.jsonl line {i}: invalid JSON: {e}")
```

```python
"""Deterministic dataset generator for bv.demo() — Phase 13.5 Plan 06.

Run via: python -m beava.demos._generate

Generates 3 datasets (adtech / fraud / ecommerce). Total install impact ~3 MB.
"""
from __future__ import annotations

import json
import random
from pathlib import Path

SEED = 42
DATASET_SIZE = 10_000  # events per dataset

HERE = Path(__file__).parent


def _write(path: Path, schema: list[dict], events: list[dict]) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    with (path / "schema.json").open("w") as f:
        json.dump(schema, f, indent=2)
    with (path / "events.jsonl").open("w") as f:
        for ev in events:
            f.write(json.dumps(ev) + "\n")


def generate_adtech() -> None:
    random.seed(SEED)
    base_ts = 1_700_000_000_000  # epoch ms; deterministic
    n_users = 500
    n_campaigns = 20
    pages = ["/home", "/pricing", "/blog", "/docs", "/checkout", "/landing/a", "/landing/b", "/promo"]
    schema = [
        {"kind": "event", "name": "Impression", "fields": [
            {"name": "user_id", "type": "str"},
            {"name": "campaign_id", "type": "str"},
            {"name": "page", "type": "str"},
            {"name": "ts", "type": "int"},
        ]},
        {"kind": "event", "name": "Click", "fields": [
            {"name": "user_id", "type": "str"},
            {"name": "campaign_id", "type": "str"},
            {"name": "ts", "type": "int"},
        ]},
        {"kind": "event", "name": "Conversion", "fields": [
            {"name": "user_id", "type": "str"},
            {"name": "campaign_id", "type": "str"},
            {"name": "value", "type": "float"},
            {"name": "ts", "type": "int"},
        ]},
    ]
    events: list[dict] = []
    n_imp, n_click, n_conv = 6_000, 3_000, 1_000
    for i in range(n_imp):
        events.append({
            "_event": "Impression",
            "user_id": f"u_{random.randrange(n_users)}",
            "campaign_id": f"c_{random.randrange(n_campaigns)}",
            "page": random.choice(pages),
            "ts": base_ts + i * 100 + random.randrange(0, 80),
        })
    for i in range(n_click):
        events.append({
            "_event": "Click",
            "user_id": f"u_{random.randrange(n_users)}",
            "campaign_id": f"c_{random.randrange(n_campaigns)}",
            "ts": base_ts + i * 250 + random.randrange(0, 200),
        })
    for i in range(n_conv):
        events.append({
            "_event": "Conversion",
            "user_id": f"u_{random.randrange(n_users)}",
            "campaign_id": f"c_{random.randrange(n_campaigns)}",
            "value": round(random.lognormvariate(3.5, 0.6), 2),
            "ts": base_ts + i * 1_000 + random.randrange(0, 800),
        })
    events.sort(key=lambda e: e["ts"])
    _write(HERE / "adtech", schema, events)


def generate_fraud() -> None:
    random.seed(SEED + 1)
    base_ts = 1_700_000_000_000
    n_users = 300
    merchants = [f"m_{i}" for i in range(50)]
    schema = [
        {"kind": "event", "name": "Txn", "fields": [
            {"name": "user_id", "type": "str"},
            {"name": "amount", "type": "float"},
            {"name": "merchant", "type": "str"},
            {"name": "ip", "type": "str"},
            {"name": "lat", "type": "float"},
            {"name": "lon", "type": "float"},
            {"name": "ts", "type": "int"},
        ]},
        {"kind": "event", "name": "Login", "fields": [
            {"name": "user_id", "type": "str"},
            {"name": "ip", "type": "str"},
            {"name": "device_id", "type": "str"},
            {"name": "ts", "type": "int"},
        ]},
    ]
    events: list[dict] = []
    n_txn, n_login = 7_500, 2_500
    for i in range(n_txn):
        events.append({
            "_event": "Txn",
            "user_id": f"u_{random.randrange(n_users)}",
            "amount": round(random.lognormvariate(3.0, 1.0), 2),
            "merchant": random.choice(merchants),
            "ip": f"10.{random.randrange(0,256)}.{random.randrange(0,256)}.{random.randrange(0,256)}",
            "lat": round(random.uniform(25.0, 49.0), 4),
            "lon": round(random.uniform(-125.0, -67.0), 4),
            "ts": base_ts + i * 200 + random.randrange(0, 180),
        })
    for i in range(n_login):
        events.append({
            "_event": "Login",
            "user_id": f"u_{random.randrange(n_users)}",
            "ip": f"10.{random.randrange(0,256)}.{random.randrange(0,256)}.{random.randrange(0,256)}",
            "device_id": f"d_{random.randrange(1000)}",
            "ts": base_ts + i * 600 + random.randrange(0, 500),
        })
    events.sort(key=lambda e: e["ts"])
    _write(HERE / "fraud", schema, events)


def generate_ecommerce() -> None:
    random.seed(SEED + 2)
    base_ts = 1_700_000_000_000
    n_users = 800
    pages = [f"/p/{i}" for i in range(200)]
    skus = [f"sku_{i:04d}" for i in range(500)]
    schema = [
        {"kind": "event", "name": "PageView", "fields": [
            {"name": "user_id", "type": "str"},
            {"name": "page", "type": "str"},
            {"name": "ts", "type": "int"},
        ]},
        {"kind": "event", "name": "AddToCart", "fields": [
            {"name": "user_id", "type": "str"},
            {"name": "sku", "type": "str"},
            {"name": "price", "type": "float"},
            {"name": "ts", "type": "int"},
        ]},
        {"kind": "event", "name": "Purchase", "fields": [
            {"name": "user_id", "type": "str"},
            {"name": "total", "type": "float"},
            {"name": "n_items", "type": "int"},
            {"name": "ts", "type": "int"},
        ]},
    ]
    events: list[dict] = []
    n_pv, n_atc, n_pur = 7_000, 2_000, 1_000
    for i in range(n_pv):
        events.append({
            "_event": "PageView",
            "user_id": f"u_{random.randrange(n_users)}",
            "page": random.choice(pages),
            "ts": base_ts + i * 100 + random.randrange(0, 80),
        })
    for i in range(n_atc):
        events.append({
            "_event": "AddToCart",
            "user_id": f"u_{random.randrange(n_users)}",
            "sku": random.choice(skus),
            "price": round(random.uniform(5.0, 250.0), 2),
            "ts": base_ts + i * 400 + random.randrange(0, 300),
        })
    for i in range(n_pur):
        events.append({
            "_event": "Purchase",
            "user_id": f"u_{random.randrange(n_users)}",
            "total": round(random.uniform(10.0, 800.0), 2),
            "n_items": random.randrange(1, 8),
            "ts": base_ts + i * 1_200 + random.randrange(0, 1000),
        })
    events.sort(key=lambda e: e["ts"])
    _write(HERE / "ecommerce", schema, events)


def main() -> None:
    generate_adtech()
    generate_fraud()
    generate_ecommerce()


if __name__ == "__main__":
    main()
```
