```ts
import { HttpTransport } from "./transport.js";
import { TcpTransport } from "./transport-tcp.js";
import {
  spawnEmbeddedServer, teardownProcess, type EmbedHandle,
} from "./embed.js";
import type { ChildProcess } from "node:child_process";
import type {
  AppOptions, Descriptor, RegisterOptions, RegisterResult,
  PushResult, PingResult, GetRequest, FeatureRow,
} from "./types.js";

interface Transport {
  request<T>(method: string, path: string, body?: unknown): Promise<T>;
  close(): Promise<void>;
}

export class BeavaApp {
  readonly url: string;
  private transport: Transport | null = null;
  private embedProc: ChildProcess | null = null;
  private readyPromise: Promise<void> | null = null;
  private readonly opts: AppOptions;

  constructor(url?: string, opts: AppOptions = {}) {
    this.opts = opts;
    this.url = url ?? "";
    if (url && url.startsWith("http")) {
      this.transport = new HttpAdapter(new HttpTransport(url, opts.timeout ?? 30_000));
    } else if (url && url.startsWith("tcp://")) {
      const u = new URL(url);
      this.transport = new TcpAdapter(new TcpTransport(u.hostname, parseInt(u.port, 10), opts.timeout ?? 30_000));
    } else if (!url) {
      // Embed mode — defer spawn until first call
      this.transport = null;
    } else {
      throw new Error(`unsupported URL scheme: ${url}`);
    }
  }

  private async ensureReady(): Promise<Transport> {
    if (this.transport) return this.transport;
    if (this.readyPromise) {
      await this.readyPromise;
      return this.transport!;
    }
    this.readyPromise = (async () => {
      const handle: EmbedHandle = await spawnEmbeddedServer({ testMode: this.opts.test_mode });
      this.embedProc = handle.proc;
      this.transport = new HttpAdapter(new HttpTransport(handle.httpUrl, this.opts.timeout ?? 30_000));
    })();
    await this.readyPromise;
    return this.transport!;
  }

  async ping(): Promise<PingResult> {
    const t = await this.ensureReady();
    return t.request<PingResult>("GET", "/health");
  }

  async register(
    descriptors: Descriptor[],
    opts: RegisterOptions = {},
  ): Promise<RegisterResult> {
    const t = await this.ensureReady();
    return t.request<RegisterResult>("POST", "/register", {
      nodes: descriptors,
      force: opts.force ?? false,
      dry_run: opts.dry_run ?? false,
    });
  }

  // push / pushSync / get / batchGet / reset / close land in subsequent tasks.
  async close(): Promise<void> {
    if (this.transport) {
      await this.transport.close();
      this.transport = null;
    }
    if (this.embedProc) {
      await teardownProcess(this.embedProc);
      this.embedProc = null;
    }
  }
}

// HttpAdapter / TcpAdapter wrap the transports with a uniform request() signature.
// HttpAdapter passes (method, path, body) → HttpTransport.request directly.
// TcpAdapter maps (method, path, body) → opcode + body, ignores method/path.
// (Implementation details inline; ~30 LOC each.)
```

const maybeIt = haveBinary ? it : it.skip;
