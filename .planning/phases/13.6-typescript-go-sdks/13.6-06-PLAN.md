func (a *App) Get(ctx context.Context, table string, key any) (FeatureResult, error) {
    return a.getInternal(ctx, table, key)
}

func (a *App) GetGlobal(ctx context.Context, table string) (FeatureResult, error) {
    return a.getInternal(ctx, table, "")
}

func (a *App) getInternal(ctx context.Context, table string, key any) (FeatureResult, error) {
    body := map[string]any{"table": table, "key": key}
    raw, err := a.transport.Request(ctx, http.MethodPost, "/get", body)
    if err != nil {
        return nil, err
    }
    var out FeatureResult
    if len(raw) == 0 || string(raw) == "{}" {
        return FeatureResult{}, nil
    }
    if err := json.Unmarshal(raw, &out); err != nil {
        return nil, err
    }
    if out == nil {
        out = FeatureResult{}
    }
    return out, nil
}
```

package beava_test

import (
    "context"
    "os"
    "os/exec"
    "testing"

    beava "github.com/beava-dev/beava/sdk/go"
)

func haveBeavaBinary() bool {
    if os.Getenv("BEAVA_BINARY") != "" { return true }
    if _, err := exec.LookPath("beava"); err == nil { return true }
    if _, err := os.Stat("../../target/debug/beava"); err == nil { return true }
    return false
}

func TestEmbedMode_Smoke(t *testing.T) {
    if !haveBeavaBinary() {
        t.Skip("beava binary not available; set BEAVA_BINARY to run")
    }
    ctx := context.Background()
    app, err := beava.NewApp(ctx, "", beava.WithTestMode())
    if err != nil { t.Fatalf("NewApp: %v", err) }
    defer app.Close(ctx)

    ping, err := app.Ping(ctx)
    if err != nil { t.Fatalf("Ping: %v", err) }
    if ping.ServerVersion == "" {
        t.Errorf("expected non-empty ServerVersion")
    }
}
```
