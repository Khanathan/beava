```python
"""Cross-SDK conformance: drive Python + TS + Go SDKs against the same scenario.

Per Phase 13.6 D-03, a single Python orchestrator is the source of truth for
cross-SDK wire agreement.
"""
from __future__ import annotations

import json
import os
import shutil
import subprocess
from pathlib import Path

import pytest

HERE = Path(__file__).parent
SCENARIO = HERE / "scenario.json"


def _have_beava_binary() -> bool:
    if os.environ.get("BEAVA_BINARY"):
        return True
    if shutil.which("beava"):
        return True
    # Walk upward from HERE looking for target/debug/beava
    for parent in [HERE, *HERE.parents]:
        if (parent / "target" / "debug" / "beava").is_file():
            return True
    return False


def _have_node() -> bool:
    return shutil.which("node") is not None


def _have_go() -> bool:
    return shutil.which("go") is not None


def _run_python(scenario: dict) -> list[dict]:
    import beava as bv

    app = bv.App(test_mode=True)
    try:
        app.register_json(scenario["register_payload"])  # forward-ref helper; if absent, fall back to per-node loop
        for ev in scenario["events"]:
            app.push(ev["event_name"], ev["fields"])
        results = []
        for g in scenario["gets"]:
            if g["key"] == "":
                results.append(app.get(g["table"]))
            else:
                results.append(app.get(g["table"], g["key"]))
        return results
    finally:
        app.close()


def _run_ts(scenario_path: Path) -> list[dict]:
    proc = subprocess.run(
        ["node", "--import", "tsx", str(HERE / "run_ts.ts"), str(scenario_path)],
        capture_output=True, text=True, timeout=60,
        env={**os.environ, "TSX_TSCONFIG_PATH": str(HERE.parent.parent / "sdk" / "typescript" / "tsconfig.json")},
    )
    if proc.returncode != 0:
        raise RuntimeError(f"TS adapter failed: {proc.stderr}")
    return json.loads(proc.stdout.strip())["results"]


def _run_go(scenario_path: Path) -> list[dict]:
    proc = subprocess.run(
        ["go", "run", "run_go.go", str(scenario_path)],
        capture_output=True, text=True, timeout=60, cwd=str(HERE),
    )
    if proc.returncode != 0:
        raise RuntimeError(f"Go adapter failed: {proc.stderr}")
    return json.loads(proc.stdout.strip())["results"]


@pytest.mark.skipif(not _have_beava_binary(), reason="beava binary not available")
def test_cross_sdk_agreement():
    scenario = json.loads(SCENARIO.read_text())
    expected = [g["expected"] for g in scenario["gets"]]

    py_results = _run_python(scenario)
    assert py_results == expected, f"Python diverged from scenario expected: {py_results} != {expected}"

    if _have_node():
        ts_results = _run_ts(SCENARIO)
        assert ts_results == py_results, f"TS diverged from Python: {ts_results} != {py_results}"
    else:
        pytest.skip("node not available; skipping TS comparison")

    if _have_go():
        go_results = _run_go(SCENARIO)
        assert go_results == py_results, f"Go diverged from Python: {go_results} != {py_results}"
    else:
        pytest.skip("go not available; skipping Go comparison")
```

```go
package main

import (
    "context"
    "encoding/json"
    "os"

    beava "github.com/beava-dev/beava/sdk/go"
)

type Scenario struct {
    RegisterPayload map[string]any   `json:"register_payload"`
    Events          []map[string]any `json:"events"`
    Gets            []map[string]any `json:"gets"`
}

func main() {
    raw, err := os.ReadFile(os.Args[1])
    if err != nil { panic(err) }
    var s Scenario
    if err := json.Unmarshal(raw, &s); err != nil { panic(err) }

    ctx := context.Background()
    app, err := beava.NewApp(ctx, "", beava.WithTestMode())
    if err != nil { panic(err) }
    defer app.Close(ctx)

    nodes, _ := s.RegisterPayload["nodes"].([]any)
    descs := make([]beava.Descriptor, len(nodes))
    for i, n := range nodes {
        descs[i] = n.(map[string]any)
    }
    if _, err := app.Register(ctx, descs); err != nil { panic(err) }

    for _, ev := range s.Events {
        if _, err := app.Push(ctx, ev["event_name"].(string), ev["fields"].(map[string]any)); err != nil { panic(err) }
    }

    results := make([]beava.FeatureResult, 0, len(s.Gets))
    for _, g := range s.Gets {
        table := g["table"].(string)
        if g["key"] == "" {
            r, err := app.GetGlobal(ctx, table)
            if err != nil { panic(err) }
            results = append(results, r)
        } else {
            r, err := app.Get(ctx, table, g["key"])
            if err != nil { panic(err) }
            results = append(results, r)
        }
    }

    out, _ := json.Marshal(map[string]any{"sdk": "go", "results": results})
    os.Stdout.Write(out)
}
```
