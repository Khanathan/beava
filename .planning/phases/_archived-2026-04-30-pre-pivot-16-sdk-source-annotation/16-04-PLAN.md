<objective>
Ship opt-in client-side lingering for high-throughput Python producers
(D-10). Constructor kwargs `linger_ms=0` (default) and `max_batch=256`
enable per-`Event`-descriptor buffering in memory; flushes fire on
timer, on max_batch, on explicit `app.flush()`, on `app.close()`, or on
context-manager `__exit__`.

Purpose: Python SDK users producing 10k-100k events/sec today pay one
request per event. Lingering collapses that to one request per
max_batch or per linger_ms window, without changing call-site ergonomics
(same `app.push(Event, row)` signature). Opt-in (default off) → zero
behavior change for existing callers.

Output: new `_linger.py` module; constructor kwargs; `app.push` routed
via buffer when enabled; explicit flush + close + context-manager
drains; 2 REQ-IDs closed.

**Scope note:** Python SDK only; zero server changes. Depends on a
`push_many`-shaped wire for maximum win — see GA-6 fallback.
</objective>

      ```python
      import threading, time
      class LingerBuffer:
          def __init__(self, *, linger_ms: int, max_batch: int, flush_cb):
              # flush_cb(event_name: str, rows: list[dict]) -> None
              self._linger_ms = linger_ms
              self._max_batch = max_batch
              self._flush_cb = flush_cb
              self._buffers: dict[str, list[dict]] = {}
              self._first_enq: dict[str, float] = {}
              self._lock = threading.Lock()
              self._timer: threading.Timer | None = None
              self._closed = False

          def enqueue(self, event_name: str, row: dict) -> None:
              with self._lock:
                  if self._closed:
                      raise RuntimeError("LingerBuffer closed")
                  buf = self._buffers.setdefault(event_name, [])
                  if not buf:
                      self._first_enq[event_name] = time.monotonic()
                  buf.append(row)
                  if len(buf) >= self._max_batch:
                      rows = self._buffers.pop(event_name)
                      self._first_enq.pop(event_name, None)
                      # flush outside the lock
                      to_flush = (event_name, rows)
                  else:
                      to_flush = None
                  self._ensure_timer_locked()
              if to_flush is not None:
                  self._flush_cb(*to_flush)

          def _ensure_timer_locked(self):
              # Arm a timer for the earliest first_enq + linger_ms, if any buffer non-empty.
              # Cancel + rearm on each enqueue for simplicity (O(1) per push).
              if self._timer is not None:
                  self._timer.cancel()
              if not self._buffers:
                  self._timer = None
                  return
              earliest = min(self._first_enq.values())
              delay = max(0.0, earliest + self._linger_ms/1000.0 - time.monotonic())
              self._timer = threading.Timer(delay, self._on_timer)
              self._timer.daemon = True
              self._timer.start()

          def _on_timer(self):
              drained: list[tuple[str, list[dict]]] = []
              now = time.monotonic()
              with self._lock:
                  for name in list(self._buffers.keys()):
                      if now - self._first_enq[name] >= self._linger_ms/1000.0:
                          drained.append((name, self._buffers.pop(name)))
                          self._first_enq.pop(name, None)
                  self._ensure_timer_locked()
              for name, rows in drained:
                  self._flush_cb(name, rows)

          def flush(self) -> None:
              drained: list[tuple[str, list[dict]]] = []
              with self._lock:
                  for name, rows in self._buffers.items():
                      drained.append((name, rows))
                  self._buffers.clear()
                  self._first_enq.clear()
                  if self._timer is not None:
                      self._timer.cancel()
                      self._timer = None
              for name, rows in drained:
                  self._flush_cb(name, rows)

          def close(self) -> None:
              self.flush()
              with self._lock:
                  self._closed = True
      ```
