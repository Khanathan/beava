           let features = engine.push_with_cascade(&stream_name, &payload, store, now)?;
           let result = feature_map_to_json(&features);

           // Append raw event to primary stream's event log
           if let Some(ref mut log) = event_log {
               let event_bytes = serde_json::to_vec(&payload).unwrap_or_default();
               let _ = log.append(&stream_name, &event_bytes, now);
           }

           // Append event to downstream streams' event logs (for backfill in Phase 8)
           if let Some(ref mut log) = event_log {
               let event_bytes = serde_json::to_vec(&payload).unwrap_or_default();
               let downstream = engine.get_cascade_targets(&stream_name);
               for ds_name in &downstream {
                   // Only log if downstream actually processed (key_field present in event)
                   let ds_def = engine.get_stream(ds_name);
                   let should_log = match ds_def {
                       Some(d) => match &d.key_field {
                           Some(kf) => matches!(payload.get(kf), Some(serde_json::Value::String(s)) if !s.is_empty()),
                           None => true, // keyless downstream always logs
                       },
                       None => false,
                   };
                   if should_log {
                       let _ = log.append(ds_name, &event_bytes, now);
                   }
               }
           }

           // Fan-out: ONLY for the primary push target, NOT for cascade-triggered pushes.
           // Fan-out handles implicit cross-key updates (different key_field).
           // Cascade handles explicit dependency chains (depends_on).
           // Streams in the cascade DAG are NOT eligible for fan-out (avoid double-processing).
           let primary_key_field = engine.get_stream(&stream_name)
               .and_then(|s| s.key_field.as_deref());
           let cascade_streams = engine.get_cascade_targets(&stream_name);
           let targets = engine.fan_out_targets();
           for (target_name, target_key_field) in &targets {
               if target_name == &stream_name { continue; }
               if primary_key_field == Some(target_key_field.as_str()) { continue; }
               // Skip streams that are cascade targets (avoid double-processing)
               if cascade_streams.contains(target_name) { continue; }
               if let Some(serde_json::Value::String(key_val)) = payload.get(target_key_field.as_str()) {
                   if !key_val.is_empty() {

                           let _ = log.append(target_name, &event_bytes, now);
                       }
                   }
               }

           app.metrics.push_latency_seconds = push_elapsed.as_secs_f64();

           Ok(result)

       ```rust
       /// Return all streams downstream of the given stream (for event log and fan-out isolation).
       pub fn get_cascade_targets(&self, stream_name: &str) -> Vec<String> {
           let mut targets = Vec::new();
           let mut visited = AHashSet::new();
           let mut queue = Vec::new();
           if let Some(direct) = self.downstream_map.get(stream_name) {
               queue.extend(direct.iter().cloned());
           }
           while let Some(current) = queue.pop() {
               if visited.insert(current.clone()) {
                   targets.push(current.clone());
                   if let Some(next) = self.downstream_map.get(&current) {
                       queue.extend(next.iter().cloned());
                   }
               }
           }
           targets
       }
       ```
