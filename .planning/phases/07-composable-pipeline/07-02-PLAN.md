    class TestDependsOn:
        def test_depends_on_sets_attribute(self):
            """depends_on stores class references."""
            @st.stream()
            class Upstream:
                pass

            @st.stream(key="user_id", depends_on=[Upstream])
            class Downstream:

        def test_depends_on_register_json(self):

            @st.stream()
            class Upstream:
                pass

            class Downstream:
                c = st.count(window="1h")
            j = Downstream._to_register_json()
            assert j["depends_on"] == ["Upstream"]

        def test_depends_on_multiple(self):
            """Multiple depends_on classes resolved."""
            @st.stream()
            class A:
                pass

            class B:

            class C:
                c = st.count(window="1h")

            assert j["depends_on"] == ["A", "B"]
