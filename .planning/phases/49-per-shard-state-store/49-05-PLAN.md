In `handle_push_core_ex` (after the existing push completes via the StateStore/PipelineEngine), add a shard-path mirror write for entity state:

```rust
// Wave 1 shard integration: mirror entity state into Shard-0 after the
// existing StateStore write. At N=1, shard index is always 0.
// This validates the shard data path is live without disrupting existing behavior.
// Wave 2 (Phase 50) will make the shard path PRIMARY and remove the StateStore write.
{
    let sharded = state.sharded_store.lock().expect("sharded_store mutex poisoned");
    let stream_def = state.engine.read().streams.get(stream_name).cloned();
    let key_field = stream_def.as_ref().and_then(|s| s.key_field.as_deref());
    for event in events {
        let mut shard = sharded.shard_for_event(event, key_field);
        // Mirror: populate Shard.state with entity key from this event.
        // EntityState is reconstructed from the existing store read.
        if let Some(key_field) = key_field {
            if let Some(serde_json::Value::String(key)) = event.get(key_field) {
                // Ensure entity exists in shard state (Wave 1: identity copy from StateStore).
                shard.state.entry(key.clone()).or_insert_with(crate::state::store::EntityState::default);
                // Mark dirty in shard dirty_set.
                shard.dirty_set.insert(key.clone());
                // Mirror watermark observe to shard (Wave 1: N=1, shard-0's WatermarkState IS the watermark).
                // event_time is SystemTime::now() at push time.
                shard.watermark.observe(stream_name, now);
            }
        }
    }
}
```
