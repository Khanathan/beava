/// Plan 13.4-04 ‚Äî extract `event` from JSON body for verb-style
/// POST /push, then synthesize the corresponding WireRequest with the
/// inner `data` payload as the existing dispatch expects.
fn parse_verb_push(body: &[u8], sync: bool) -> WireRequest {
    let parsed: serde_json::Value = match serde_json::from_slice(body) {
        Ok(v) => v,
        Err(_) => return WireRequest::ParseError {
            reason: "invalid_json_body".into(),
        },
    };
    let event_name = match parsed.get("event").and_then(|v| v.as_str()) {
        Some(s) => s.to_string(),
        None => return WireRequest::ParseError {
            reason: "missing_event_name_in_body".into(),
        },
    };
    let data = parsed.get("data").cloned().unwrap_or(serde_json::Value::Null);
    let body_bytes = bytes::Bytes::from(serde_json::to_vec(&data).unwrap_or_default());
    if sync {
        WireRequest::HttpPushSync { event_name, body: body_bytes, body_format: beava_core::wire::CT_JSON }
    } else {
        WireRequest::HttpPush { event_name, body: body_bytes, body_format: beava_core::wire::CT_JSON }
    }
}
