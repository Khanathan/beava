```rust
pub fn register_check_force_required(
    diff: &RegisterDiff,
    force: bool,
) -> Result<(), ForceRequiredError> {
    if !diff.destructive.is_empty() && !force {
        return Err(ForceRequiredError {
            code: "force_required",
            reason: "Destructive registry change requires force=true. See diff for details.".into(),
            diff: diff.clone(),
        });
    }
    Ok(())
}
```
