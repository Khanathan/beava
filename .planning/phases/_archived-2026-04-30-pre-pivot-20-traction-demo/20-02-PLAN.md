    pub async fn require_loopback_or_token(
        ConnectInfo(addr): ConnectInfo<SocketAddr>,
        State(state): State<SharedState>,
        headers: HeaderMap,
        req: Request,
        next: Next,
    ) -> Result<Response, (StatusCode, Json<Value>)> {
        if addr.ip().is_loopback() { return Ok(next.run(req).await); }
        if let Some(tok) = &state.admin_token {
            if let Some(h) = headers.get(AUTHORIZATION).and_then(|v| v.to_str().ok()) {
                if h == format!("Bearer {}", tok) { return Ok(next.run(req).await); }
            }
        }
        Err((StatusCode::FORBIDDEN, Json(json!({"error":"admin route: loopback or bearer token required"}))))
    }

    async fn public_features(Path(key): Path<String>, State(s): State<SharedState>) -> impl IntoResponse {
        // Use existing debug_key handler logic but project out only the {feature_name: value} map.
        // Strip anything not matching the whitelist: value | number | string | bool | null.
        match s.store.get_entity(&key) {
            Some(entity) => {
                let features: serde_json::Map<String, Value> = entity.features_read_only();
                (StatusCode::OK, cors(), Json(json!({"key": key, "features": features}))).into_response()
            }
            None => (StatusCode::NOT_FOUND, cors(), Json(json!({"error":"key not found"}))).into_response(),
        }
    }

    async fn public_recent_events(Query(q): Query<RecentQuery>, State(s): State<SharedState>) -> impl IntoResponse {
        let limit = q.limit.unwrap_or(20).min(100);
        let events = s.recent_events.snapshot(limit);  // new ring in AppState
        (StatusCode::OK, cors(), Json(json!({"events": events}))).into_response()
    }

    async fn public_stats(State(s): State<SharedState>) -> Json<Value> {
        let hist = &s.latency_tracker.command_histograms[CMD_PUSH];
        Json(json!({
            "events_total":    s.events_total.load(Ordering::Relaxed),
            "current_eps":     s.throughput.eps_5s(),
            "p99_push_us":     hist.percentile(99.0),
            "p50_push_us":     hist.percentile(50.0),
            "uptime_seconds":  s.started_at.elapsed().as_secs(),
            "keys_total":      s.store.entity_count(),
        }))
    }

## Risks & Mitigations

- **ConnectInfo not propagated through test harness.** ‚Üí Use `oneshot` with explicit `extensions_mut().insert(ConnectInfo(addr))` as documented in Axum 0.7 test examples. If harness is too awkward, fall back to live `TcpListener` bound to 127.0.0.1 with real clients; tests still pass.
- **Event log not yet exposing tail cheaply.** ‚Üí Added `RecentEventsRing` in AppState with fixed 100-capacity VecDeque ‚Äî O(1) write from the PUSH handler, O(limit) read. Independent of Phase 6 event log shape.
- **CORS mis-set breaks blog widget.** ‚Üí `Access-Control-Allow-Origin: *` on `/public/*` in this plan; Caddyfile in Plan 03 adds a stricter origin override if needed for production.
- **Exposing feature values leaks user data.** ‚Üí The demo instance is seeded with synthetic data only (Plan 01 generator); no real user data ever lands on the public instance. Documented in Plan 03 deployment README.
- **Router split breaks existing route ordering.** ‚Üí All existing integration tests (`cargo test`) must continue to pass ‚Äî add that as part of verify.
