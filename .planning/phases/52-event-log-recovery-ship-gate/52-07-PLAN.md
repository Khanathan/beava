use proptest::collection::vec;

// Event strategy
fn event_strategy() -> impl Strategy<Value = TestEvent> {
    (any::<u32>(), "[a-z]{3,8}", any::<i64>()).prop_map(|(stream_ord, key, value)| {
        TestEvent { stream_ord, key, value }
    })
}

// Batch strategy: 1..=500 events, deterministic seed
fn batch_strategy() -> impl Strategy<Value = Vec<TestEvent>> {
    vec(event_strategy(), 1..=500)
}
