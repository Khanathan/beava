<objective>
Add an automated architectural test that enforces, going forward, the two locked architectural commitments from Phase 12.6:

1. **mio is the only hot-path data-plane entry.** All push/get/upsert/delete/retract dispatches go through `apply_shard.rs::dispatch_*_sync`. The legitimate callers of `apply_event_to_aggregations` are exactly two: `apply_shard.rs::dispatch_one` (data plane) + `recovery.rs::replay` (cold path on boot). Any third caller is an architectural regression.

2. **axum is restricted to the admin sidecar.** axum::Router / axum::Json / axum::Extension imports may only appear within the ServerV18 admin-sidecar block in `server.rs`. Any other appearance (especially in a new file or a re-introduced data-plane handler) is a regression.

The architectural-test mechanism (per CONTEXT discretion clause): **runtime grep** (file-walking the workspace at test execution). Rationale:
- Build-time `#[deny]` lints don't catch import-by-string-name (e.g., grep-defeat patterns).
- Runtime assertion at server boot would only catch the running config, not future code added by an unwary executor.
- A grep-based test runs in `cargo test`, lives with the rest of the workspace tests, fails clearly with a file:line citation, and is easy to read.

    use std::path::Path;
    use std::fs;

    fn workspace_server_src() -> &'static Path {
        // env!("CARGO_MANIFEST_DIR") is crates/beava-server when running tests in this crate.
        Path::new(env!("CARGO_MANIFEST_DIR")).join("src").leak()
    }

    fn collect_rs_files(dir: &Path) -> Vec<std::path::PathBuf> {
        let mut out = Vec::new();
        fn visit(dir: &Path, out: &mut Vec<std::path::PathBuf>) {
            if let Ok(rd) = fs::read_dir(dir) {
                for entry in rd.flatten() {
                    let p = entry.path();
                    if p.is_dir() { visit(&p, out); }
                    else if p.extension().map_or(false, |e| e == "rs") { out.push(p); }
                }
            }
        }
        visit(dir, &mut out);
        out
    }

    fn strip_comments(src: &str) -> String {
        // Naive: drop lines starting with // (after whitespace). Block comments not stripped (rare in this codebase).
        src.lines().filter(|l| !l.trim_start().starts_with("//")).collect::<Vec<_>>().join("\n")
    }

    #[test]
    fn only_apply_shard_and_recovery_call_apply_event_to_aggregations() {
        let src = workspace_server_src();
        let files = collect_rs_files(src);
        let mut violations = Vec::new();
        for f in files {
            let basename = f.file_name().unwrap().to_string_lossy().to_string();
            // Allowlist: apply_shard.rs (dispatch path), recovery.rs (cold-path replay).
            if basename == "apply_shard.rs" || basename == "recovery.rs" { continue; }
            let content = fs::read_to_string(&f).unwrap_or_default();
            let stripped = strip_comments(&content);
            if stripped.contains("apply_event_to_aggregations(") {
                violations.push(format!("{}: contains apply_event_to_aggregations(", f.display()));
            }
        }
        assert!(violations.is_empty(),
            "Architectural regression: only apply_shard.rs and recovery.rs may call apply_event_to_aggregations. \
             Violations:\n{}", violations.join("\n"));
    }

    #[test]
    fn axum_imports_only_in_server_rs_admin_sidecar() {
        let src = workspace_server_src();
        let files = collect_rs_files(src);
        let mut violations = Vec::new();
        for f in files {
            let basename = f.file_name().unwrap().to_string_lossy().to_string();
            // Allowlist: server.rs (ServerV18 admin sidecar).
            if basename == "server.rs" { continue; }
            let content = fs::read_to_string(&f).unwrap_or_default();
            let stripped = strip_comments(&content);
            // Patterns: axum::Router, axum::Json, axum::Extension, axum::extract::, axum::routing::
            for needle in &["axum::Router", "axum::Json", "axum::Extension", "axum::extract::", "axum::routing::"] {
                if stripped.contains(needle) {
                    violations.push(format!("{}: contains '{}'", f.display(), needle));
                }
            }
        }
        assert!(violations.is_empty(),
            "Architectural regression: axum is restricted to ServerV18 admin sidecar in server.rs. Violations:\n{}",
            violations.join("\n"));
    }
