    ```rust
    //! Shared bench-bootstrap helper module.
    //!
    //! Builds an ApplyShard wired to a synthetic in-memory WAL, registry, and
    //! AppState. Used by Phase 12.6+ apply-path benches to avoid duplicating
    //! the bootstrap across cells.
    //!
    //! The shape mirrors `crates/beava-server/tests/phase18_02_inline_wal_test.rs`
    //! lines 150-174 (`dispatch_push_periodic_returns_committed_lsn`) — the
    //! canonical WalGlue + WalBufferRing + WalLsn setup — extended to wire a
    //! Registry constructed from the bench's register_payload + an AppState
    //! with empty apply tables.

    use beava_server::apply_shard::ApplyShard;
    use beava_server::runtime_core_glue::WalGlue;
    use beava_server::AppState;
    use beava_persistence::wal_buffer::WalBufferRing;
    use beava_persistence::wal_lsn::WalLsn;
    use beava_core::registry::Registry;
    use std::sync::Arc;

    pub struct BenchHarness {
        pub shard: ApplyShard,
        pub state: Arc<AppState>,
        pub registry: Arc<Registry>,
    }

    pub fn build_apply_shard_with_pipeline(register_payload: serde_json::Value) -> BenchHarness {
        // 1. Construct WAL primitives (copy-inlined from phase18_02_inline_wal_test.rs:154-157)
        let lsn = Arc::new(WalLsn::new());
        let ring = Arc::new(WalBufferRing::new(3, 16 * 1024, Arc::clone(&lsn)));
        let _wal_glue = WalGlue::new(Arc::clone(&ring), Arc::clone(&lsn));

        // 2. Build a Registry from the register_payload by invoking
        //    register_validate / OpChain::compile / Registry::insert as in
        //    apply_shard.rs::dispatch_register_sync. Cite the path:
        //    `crates/beava-server/src/apply_shard.rs::dispatch_register_sync`
        //    has the canonical "JSON payload → typed Registry" boundary.
        let registry = Arc::new(Registry::from_register_payload(register_payload)
            .expect("register_payload should compile"));

        // 3. Construct AppState with default apply tables (empty entity-keyed HashMaps).
        let state = Arc::new(AppState::new_for_bench(Arc::clone(&registry)));

        // 4. Construct ApplyShard wired to the WAL ring + AppState.
        let shard = ApplyShard::new_for_bench(Arc::clone(&state), Arc::clone(&ring), Arc::clone(&lsn));

        BenchHarness { shard, state, registry }
    }
    ```

    ```rust
    //!
    //! Three cells covering a cross-section of operator work after the legacy axum

    mod common;

    use beava_core::row::{Row, Value};
    use beava_runtime_core::wire_request::WireRequest;
    use bytes::Bytes;
    use criterion::{black_box, criterion_group, criterion_main, BatchSize, Criterion};
    use serde_json::json;

    /// Cell 1: simple_counter — Tx.count() per user_id, 1k events.
    fn bench_simple_counter(c: &mut Criterion) {
        let payload = json!({
            "nodes": [
                {"kind": "event", "name": "Tx",
                 "schema": {"fields": {"user_id": "str", "amount": "f64"}, "optional_fields": []}},
                {"kind": "derivation", "name": "TxCount", "output_kind": "event",
                 "upstreams": ["Tx"],
                 "ops": [{"op": "group_by", "keys": ["user_id"]}],
                 "aggs": [{"name": "cnt", "kind": "count"}],
                 "schema": {"fields": {"user_id": "str", "cnt": "i64"}, "optional_fields": []}}
            ]
        });
        let harness = common::build_apply_shard_with_pipeline(payload);
        c.bench_function("phase12_6/simple_counter/1k_events", |b| {
            b.iter_batched(
                || (0u64..1_000).map(|i| {
                    let mut row = Row::new();
                    row.insert("user_id".into(), Value::Str(format!("u{}", i % 100).into()));
                    row.insert("amount".into(), Value::F64(42.0));
                    row
                }).collect::<Vec<Row>>(),
                |rows| {
                    for row in rows {
                        let req = WireRequest::TcpPush {
                            event_name: "Tx".to_string(),
                            body: Bytes::new(),
                        };
                        let _resp = harness.shard.dispatch_push_sync_with_row(req, row);
                        black_box(());
                    }
                },
                BatchSize::SmallInput,
            );
        });
    }
    ```
