    "decay": "08",
    "velocity": "09",

GEO_OPS = {"geo_velocity", "geo_distance", "geo_spread", "distance_from_home"}

def family_or_geo(family, op):
    if family == "buffer-geo" and op in GEO_OPS:
        return "11"
    if family == "buffer-geo":
        return "10"
    return POLISH_PLAN[family]


def main():
    repo_root = pathlib.Path(__file__).resolve().parent.parent
    for family, op, renamed_from, brief in OPS:
        target_dir = repo_root / "docs" / "operators" / family
        target_dir.mkdir(parents=True, exist_ok=True)
        target = target_dir / f"{op}.md"
        rename_note = ""
        if renamed_from:
            rename_note = (
                f"\n> Previously called `bv.{renamed_from}`. "
                f"Renamed to `{op}` per [ADR-002](../../../.planning/decisions/ADR-002-polars-op-rename.md) "

            op=op,
            brief=brief,
            rename_note=rename_note,
            polish_plan=family_or_geo(family, op),

            "## Aliases\n\n- `bv.ema` — same op; alias preserved as a convention shortcut.\n\n## See also",

    # Master catalog index
    index_path = repo_root / "docs" / "operators" / "index.md"
    lines = ["# Operator Catalog", "", "> 54 operator pages (53 unique AggKind variants + ema alias inline in ewma.md), across 7 families.", ""]

    for family in ["core", "sketch", "point-ordinal", "recency", "decay", "velocity", "buffer-geo"]:
        ops = by_family[family]
        lines.append(f"## {family}/ ({len(ops)})")
        lines.append("")
        lines.append("| Op | Description |")
        lines.append("|----|-------------|")
        for op, brief, renamed in ops:
            tag = f" *(renamed from `{renamed}` per ADR-002)*" if renamed else ""
            lines.append(f"| [`bv.{op}`](./{family}/{op}.md){tag} | {brief} |")
        lines.append("")
    lines.append("## Aliases")
    lines.append("")
    lines.append("- `bv.ema` is an alias of [`bv.ewma`](./decay/ewma.md) — documented inline on the ewma page (same AggKind variant; 53 unique kinds + 1 alias = 54 page paths).")
    lines.append("")
    lines.append("## Cost-class metadata")
    lines.append("")
    lines.append("- See [cost-class.md](./cost-class.md) for per-op CPU tier (Tier 1/2/3).")
    lines.append("")
    index_path.write_text("\n".join(lines))

    print(f"OK — scaffolded {len(OPS)} op pages + index.md")
    return 0


if __name__ == "__main__":
    sys.exit(main())

```python
#!/usr/bin/env python3
"""
Phase 13.0 Plan 05 — operator-catalog coverage tripwire.

Asserts 1:1 between docs/operators/<family>/<op>.md page files and the
known-good 54-path catalogue (defined in scripts/scaffold_op_pages.py;
54 paths = 53 unique AggKind variants + ema alias inline in ewma.md).
ema is special-cased — must be documented INSIDE ewma.md as an alias.

Used as CI tripwire post-Plan 13.0-15. If a new op gets added to the engine
without a corresponding docs page, this fails and forces the addition.

Usage: python3 scripts/check_op_page_coverage.py
Exits 0 on coverage match; exits 1 with diagnostics on mismatch.
"""
import pathlib
import re
import sys

# Import OPS list from sibling scaffold script
sys.path.insert(0, str(pathlib.Path(__file__).parent))
from scaffold_op_pages import OPS  # noqa: E402


def main() -> int:
    repo_root = pathlib.Path(__file__).resolve().parent.parent
    docs_dir = repo_root / "docs" / "operators"

    expected_paths = {
        docs_dir / family / f"{op}.md"
        for family, op, _, _ in OPS
    }

    actual_paths = {
        p for p in docs_dir.rglob("*.md")
        if p.name not in {"index.md", "cost-class.md"}
        and p.name != ".keep"
    }

    missing = expected_paths - actual_paths
    extra = actual_paths - expected_paths

    if missing or extra:
        print("FAIL — op-page coverage mismatch:", file=sys.stderr)
        for p in sorted(missing):
            print(f"  MISSING: {p.relative_to(repo_root)}", file=sys.stderr)
        for p in sorted(extra):
            print(f"  UNEXPECTED: {p.relative_to(repo_root)}", file=sys.stderr)
        return 1

    # Verify ewma has Aliases section with bv.ema
    ewma_path = docs_dir / "decay" / "ewma.md"
    ewma_text = ewma_path.read_text()
    if "## Aliases" not in ewma_text or "bv.ema" not in ewma_text:
        print(
            f"FAIL — {ewma_path.relative_to(repo_root)} missing '## Aliases' "
            "section or 'bv.ema' reference",
            file=sys.stderr,
        )
        return 1

    print(f"OK — {len(expected_paths)} op pages match catalogue (ema alias inside ewma.md)")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
