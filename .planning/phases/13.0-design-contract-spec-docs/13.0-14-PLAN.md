"""
Phase 13.0 mock backend for runnable demos.

In-memory, fixture-replay App shim. Drop-in for bv.App() during Phase 13.0
verification. Replaced by real bv.App() once Phase 13.5 lands (single-line
edit per demo).

Per Q2 locked answer: this is the canonical Phase 13.0 mock backend.
"""
import json
import pathlib
from typing import Any


class MockApp:
    """Phase 13.0 stub. Replaced by real bv.App() in Phase 13.5."""

    def __init__(self, fixtures_dir: pathlib.Path | str | None = None):
        self.fixtures = pathlib.Path(fixtures_dir or "examples/wire")
        self._registered: set[str] = set()
        # state[table_name][key] = row dict
        self._state: dict[str, dict[str, dict[str, Any]]] = {}
        self._registry_version = 0

    def register(self, *descs, force: bool = False, dry_run: bool = False):
        if dry_run:
            return {"status": "ok", "registry_version": self._registry_version, "added": [], "removed": [], "changed": [], "diff": {}}
        for d in descs:
            name = getattr(d, "_name", None) or getattr(d, "name", None) or str(d)
            self._registered.add(name)
        self._registry_version = len(self._registered)
        return {"status": "ok", "registry_version": self._registry_version, "added": list(self._registered)}

    def push(self, event_name: str, fields: dict) -> dict:
        # In a real demo we would advance state via registered derivations.
        # For Phase 13.0 mock we simply ack and return a stub LSN.
        # Demos may seed self._state directly to make assertions deterministic.
        return {"ack_lsn": 1, "registry_version": self._registry_version}

    def get(self, table: str, key) -> dict[str, Any]:
        key_str = key if isinstance(key, str) else str(key)
        return self._state.get(table, {}).get(key_str, {})

    def batch_get(self, requests: list) -> list[dict[str, Any]]:
        out = []
        for r in requests:
            if isinstance(r, dict):
                out.append(self.get(r["table"], r["key"]))
            else:  # tuple-like
                out.append(self.get(r[0], r[1]))
        return out

    def reset(self) -> None:
        self._state.clear()

    def ping(self) -> dict:
        return {"server_version": "0.0.0-mock", "registry_version": self._registry_version}

    def close(self) -> None:
        pass

    def __enter__(self):
        return self

    def __exit__(self, *_):
        self.close()

    # --- demo helpers (NOT part of the real bv.App API) ---
    def _seed(self, table: str, key: str, row: dict) -> None:
        """Demo-only: pre-populate state so app.get() returns the expected row.
        Real bv.App computes via registered derivations from app.push() events.
        """
        self._state.setdefault(table, {})[key] = row


def App(*args, **kwargs):
    """Phase 13.0 stub. Replaced by `from beava import App` in Phase 13.5."""
    return MockApp()

# Phase 13.0 examples smoke test.
# Runs all 9 vertical demo files (3 langs × 3 verticals) against

run() {

        echo "PASS  $label"
    else
        echo "FAIL  $label"
        failed=$((failed + 1))
    fi

    done

    echo "SKIP  typescript demos (npx not on PATH)"

# Go (requires go)

    for f in adtech fraud ecommerce; do
        run "go/${f}.go" go run "$REPO_ROOT/examples/go/${f}.go" >/dev/null || true
    done
else
    echo "SKIP  go demos (go not on PATH)"
fi

if [ $failed -eq 0 ]; then
    echo ""
