```rust
/// For a salted stream's read, returns the set of derived keys to probe.
/// Unsalted: returns `vec![original_key]`.
/// Salted(N): returns `vec!["key:0", "key:1", ..., "key:N-1"]`.
pub fn expand_salt_variants(
    original_key: &str,
    salt_cardinality: Option<u8>,
) -> Vec<String>;

/// Combine N salt-variant results into one using the operator's combine fn.
/// For commutative operators (sum, count), the combine is addition.
/// For last-value, pick the row with the freshest event_time.
/// For sketch types (hll/cms/uddsketch), `.merge()` is delegated.
pub fn combine_salt_variants<T>(
    variants: Vec<Option<T>>,
    combine: impl Fn(T, T) -> T,
) -> Option<T>;
```
