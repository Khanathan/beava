  ```rust
  pub fn shard_hint_for_event_salted(
      event: &serde_json::Value,
      key_field: Option<&str>,
      salt_cardinality: Option<u8>,
      primary_event_id: Option<&str>,
  ) -> u32 {
      let n = match salt_cardinality {
          None => return shard_hint_for_event(event, key_field),
          Some(n) => n as u64,
      };

      let orig_key = key_field
          .and_then(|kf| event.get(kf).and_then(|v| v.as_str()))
          .unwrap_or("");

      let pid = primary_event_id
          .map(|s| s.to_string())
          .unwrap_or_else(|| {
              let mut h = ahash::AHasher::default();
              // Hash the canonical JSON form as a stable per-event id.
              serde_json::to_string(event).unwrap_or_default().hash(&mut h);
              format!("{:016x}", h.finish())
          });

      let mut ph = ahash::AHasher::default();
      pid.hash(&mut ph);
      let salt_idx = (ph.finish() % n) as u8;

      let derived = format!("{}:{}", orig_key, salt_idx);
      let mut dh = ahash::AHasher::default();
      derived.hash(&mut dh);
      dh.finish() as u32
  }

  pub fn derive_storage_key(
      event: &serde_json::Value,
      key_field: Option<&str>,
      salt_cardinality: Option<u8>,
      primary_event_id: Option<&str>,
  ) -> Option<String> {
      let kf = key_field?;
      let orig = event.get(kf).and_then(|v| v.as_str())?.to_string();
      let Some(n) = salt_cardinality else { return Some(orig); };

      let pid = /* same derivation as above */ ;
      let mut ph = ahash::AHasher::default();
      pid.hash(&mut ph);
      let salt_idx = (ph.finish() % (n as u64)) as u8;
      Some(format!("{}:{}", orig, salt_idx))
  }
  ```
