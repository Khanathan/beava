Plan 12-10 adds 4 new variants between `HttpPushBatch` and `HttpGet`:
```rust
HttpPushAndGet { event_name: String, body: Bytes, body_format: u8 },     // POST /push-and-get/:event_name
HttpPushSyncAndGet { event_name: String, body: Bytes, body_format: u8 }, // POST /push-sync-and-get/:event_name
TcpPushAndGet { body: Bytes, body_format: u8 },                          // OP_PUSH_AND_GET (0x0015) — event_name in payload
TcpPushSyncAndGet { body: Bytes, body_format: u8 },                      // OP_PUSH_SYNC_AND_GET (0x0016) — event_name in payload
```

**TCP variant rationale:** TCP has no URL path component, so the event name lives in the JSON/msgpack payload alongside `row` and `query`. Body shape on TCP: `{"event": "Transaction", "row": {...}, "query": {"entity_key": {...}, "features": [...]}}` — mirrors the existing `OP_PUSH` envelope pattern (`{"event": "...", "body": {...}}`). On HTTP the body shape is `{"row": {...}, "query": {...}}` and the event_name comes from the URL path, matching the existing axum-side `push_and_get.rs` semantics.

    def send_push_and_get(
        self,
        event_name: str,
        row_dict: dict,
        query_dict: dict,
        *,
        wire_format: str = "json",
        sync: bool = False,
    ) -> dict:

        envelope_obj = {"event": event_name, "row": row_dict, "query": query_dict}
        if wire_format == "json":
            envelope = json.dumps(envelope_obj).encode("utf-8")
            ct = CT_JSON
        elif wire_format == "msgpack":
            import msgpack
            envelope = msgpack.packb(envelope_obj, use_bin_type=True)
            ct = CT_MSGPACK
        else:
            raise ValueError(f"wire_format must be 'json' or 'msgpack', got {wire_format!r}")
        sock = self._ensure_connected()
        sock.sendall(encode_frame(op, ct, envelope))
        frame = read_frame(sock, self.max_frame_bytes)
        if frame.op == OP_ERROR_RESPONSE:
            error_body = json.loads(frame.payload.decode("utf-8"))
            error = error_body.get("error", {})
            raise BeavaError(code=error.get("code", "unknown"),
                             message=error.get("message", ""),
                             registry_version=error_body.get("registry_version"))
        if frame.op != OP_PUSH_AND_GET_RESPONSE:
            raise BeavaError(code="unexpected_frame",
                             message=f"expected OP_PUSH_AND_GET_RESPONSE (0x0024), got op={frame.op:#06x}")
        if ct == CT_MSGPACK:
            import msgpack
            return msgpack.unpackb(frame.payload, raw=False)
        return json.loads(frame.payload.decode("utf-8"))
    ```

        self,

            row: Event row fields as a plain dict.
            entity_key: Map of group-key name → string value (e.g. ``{"user_id": "alice"}``).
            features: List of feature names to query.
            sync: If True, route to /push-sync-and-get (acks=all, awaits fsync).
                  If False, route to /push-and-get (acks=1, default).
            wire_format: "json" (default) or "msgpack" — TCP only; HTTP always JSON.

        Returns:
            Tuple of (PushAck namedtuple, features_dict).
            features_dict maps feature name → value (or None for unknown features
            and unknown entities). Unknown features emit a UserWarning.

        Raises:
            BeavaError: Server returned event_not_found / invalid_event / wal_sync_timeout.
        """

        transport = self._require_transport()
        event_name = event_type._name  # mirror existing app.upsert pattern
        query_dict = {"entity_key": entity_key, "features": features}

        if isinstance(transport, HttpTransport):
            response = transport.send_push_and_get(event_name, row, query_dict, sync=sync)
        elif isinstance(transport, TcpTransport):
            response = transport.send_push_and_get(event_name, row, query_dict,
                                                   wire_format=wire_format, sync=sync)
        elif isinstance(transport, EmbedTransport):

        features_dict = response.get("features", {})
        return ack, features_dict
    ```
