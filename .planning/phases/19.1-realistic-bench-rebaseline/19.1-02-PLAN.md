```rust
// AggKind enum (around line 154+) lists every supported op:
//   Count, Sum, Avg, Min, Max, Variance, StdDev, Ratio,
//   CountDistinct, Percentile, TopK, Entropy, BloomMember,
//   First, Last, FirstN, LastN, Lag, FirstSeen, LastSeen, Age, HasSeen,
//   TimeSince, TimeSinceLastN, Streak, MaxStreak, NegativeStreak, FirstSeenInWindow,
//   Ewma, EwVar, EwZScore, DecayedSum, DecayedCount, Twa,
//   RateOfChange, InterArrivalStats, BurstCount, DeltaFromPrev, Trend, TrendResidual,
//   OutlierCount, ValueChangeCount, ZScore,
//   Histogram, HourOfDayHistogram, DowHourHistogram, SeasonalDeviation,
//   EventTypeMix, MostRecentN, ReservoirSample,
//   GeoVelocity, GeoDistance, GeoSpread, UniqueCells, GeoEntropy, DistanceFromHome

pub struct AggOpDescriptor {
    pub kind: AggKind,
    pub field: Option<String>,        // numeric/string field for ops that read one
    pub where_expr: Option<String>,
    pub window_ms: Option<u64>,       // None = lifetime; Some = windowed wrap
    pub bucket_ms: Option<u64>,
    pub n: Option<usize>,             // for first_n/last_n/lag/most_recent_n/etc.
    pub k: Option<usize>,             // for top_k
    pub buckets: Option<Vec<f64>>,    // for histogram (NOT `bins`)
    pub categories: Option<Vec<String>>, // for event_type_mix
    pub q: Option<f64>,               // for percentile (UDDSketch quantile)
    pub half_life_ms: Option<u64>,    // for decay ops
    pub sigma: Option<f64>,           // for outlier_count
    pub sub_window_ms: Option<u64>,   // for rate_of_change / burst_count
    pub baseline_window_ms: Option<u64>, // for z_score
    pub current_field: Option<String>,   // for z_score
    pub lat_field: Option<String>,    // for GEO ops
    pub lon_field: Option<String>,    // for GEO ops
    pub precision: Option<u8>,        // for unique_cells / geo_entropy
    pub samples: Option<usize>,       // for distance_from_home
    pub fpr: Option<f64>,             // for bloom_member
    pub capacity: Option<usize>,      // for bloom_member
    pub exact_threshold: Option<usize>, // for hybrid sketches
    pub hybrid_precision: Option<u8>,
    pub hybrid_alpha: Option<f64>,
    pub hybrid_width: Option<usize>,
    pub hybrid_depth: Option<usize>,
    // ... see file for the rest
}
```

**Critical note for executor:**
- Geo ops use `lat_field` / `lon_field` (NOT `lat` / `lon`) — fraud-team.json has the wrong names.
- Histogram uses `buckets: Vec<f64>` (NOT `bins: usize`) — but per D-15 we rewrite this op to `bv.percentile` instead (since v0 has no windowed_histogram).
- `reservoir_sample` uses `n` (NOT `size`) per descriptor.
- `burst_count` uses `sub_window_ms` (NOT `threshold_secs`) and `window_ms` (NOT `window`).
- `lag` / `first_n` / `last_n` use `n: usize`.

The full canonical list is in `crates/beava-core/src/agg_op.rs` `AggOpDescriptor` struct + the `from_json` parser. Executor reads that file once, then walks fraud-team.json line-by-line and corrects each op-spec.
