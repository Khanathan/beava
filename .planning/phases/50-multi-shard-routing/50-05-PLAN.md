fn bind_reuseport_socket(addr: std::net::SocketAddr) -> std::io::Result<std::net::TcpListener> {
    use std::os::unix::io::FromRawFd;
    let socket = socket2::Socket::new(
        socket2::Domain::IPV4,
        socket2::Type::STREAM,
        None,
    )?;
    socket.set_reuse_port(true)?;
    socket.set_reuse_address(true)?;
    socket.set_nonblocking(true)?;
    socket.bind(&addr.into())?;
    socket.listen(1024)?;
    Ok(unsafe { std::net::TcpListener::from_raw_fd(socket.into_raw_fd()) })
}

```rust
pub struct AxumServerSet {
    // Holds join handles for all per-shard axum servers.
    // On shutdown, all servers are gracefully stopped.
    servers: Vec<tokio::task::JoinHandle<()>>,
}

impl AxumServerSet {
    pub async fn new(
        shard_count: usize,
        addr: std::net::SocketAddr,
        state: SharedState,
    ) -> std::io::Result<Self> {
        let mut servers = Vec::with_capacity(shard_count);

        #[cfg(target_os = "linux")]
        {
            for shard_index in 0..shard_count {
                let listener = bind_reuseport_socket(addr)?;
                let router = build_shard_router(state.clone(), shard_index);
                let handle = tokio::spawn(async move {
                    axum::Server::from_tcp(listener)
                        .expect("from_tcp")
                        .serve(router.into_make_service())
                        .await
                        .expect("axum server error");
                });
                servers.push(handle);
            }
        }

        #[cfg(not(target_os = "linux"))]
        {
            // D-04: single listener on macOS — inline dispatch handles shard routing.
            let router = build_shard_router(state, 0);
            let handle = tokio::spawn(async move {
                axum::Server::bind(&addr)
                    .serve(router.into_make_service())
                    .await
                    .expect("axum server error");
            });
            servers.push(handle);
        }

        Ok(Self { servers })
    }
}
```
