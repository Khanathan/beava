fn shard_event_loop(shard_index: usize, rx: crossbeam_channel::Receiver<ShardEvent>) {
    let rt = tokio::runtime::Builder::new_current_thread()
        .enable_all()
        .build()
        .expect("per-shard tokio runtime");

    rt.block_on(async move {
        let mut event_count: u64 = 0;
        let mut last_gauge_update = std::time::Instant::now();

        while let Ok(event) = rx.recv() {
            // TODO(Wave 3+): route to Shard state machine
            event_count += 1;

            crate::shard::metrics::record_shard_event(
                shard_index,
                crate::shard::metrics::Outcome::Accepted,
            );

            // Emit gauges every 1000 events OR every 100ms — avoid per-event gauge overhead.
            if event_count % 1000 == 0 || last_gauge_update.elapsed().as_millis() >= 100 {
                let inbox_depth = rx.len(); // crossbeam Receiver::len() = pending messages
                crate::shard::metrics::update_shard_gauges(
                    shard_index,
                    0.0,   // reactor_utilization: placeholder until Shard state machine tracks it
                    inbox_depth,
                    0,     // keys_owned: placeholder until Shard state machine is wired
                    0.0,   // watermark_lag_seconds: placeholder
                );
                last_gauge_update = std::time::Instant::now();
            }

            if let Some(tx) = event.response_tx {
                let _ = tx.send(crate::shard::thread::ShardResult::Ok);
            }
        }
    });
}

In `src/server/shard_probe.rs`, extend ShardProbe with per-shard counters:

```rust
use std::sync::atomic::{AtomicU64, Ordering};

pub struct ShardProbe {
    // Per-shard event counters — indexed by shard_index.
    shard_event_counts: Vec<Arc<AtomicU64>>,
    total_events: Arc<AtomicU64>,
}

impl ShardProbe {
    pub fn new(shard_count: usize) -> Self {
        Self {
            shard_event_counts: (0..shard_count).map(|_| Arc::new(AtomicU64::new(0))).collect(),
            total_events: Arc::new(AtomicU64::new(0)),
        }
    }

    #[inline]
    pub fn record_routed_event(&self, shard_index: usize) {
        if let Some(counter) = self.shard_event_counts.get(shard_index) {
            counter.fetch_add(1, Ordering::Relaxed);
        }
        self.total_events.fetch_add(1, Ordering::Relaxed);
    }

    /// Fraction of events that did NOT land on shard 0 (cross-shard indicator).
    /// Returns 0.0 at N=1. Gate: must be < 0.40 on the release workload.
    pub fn cross_shard_fraction(&self) -> f64 {
        let total = self.total_events.load(Ordering::Relaxed);
        if total == 0 { return 0.0; }
        let shard0_count = self.shard_event_counts.get(0)
            .map(|c| c.load(Ordering::Relaxed))
            .unwrap_or(0);
        let cross = total.saturating_sub(shard0_count);
        cross as f64 / total as f64
    }
}
```
