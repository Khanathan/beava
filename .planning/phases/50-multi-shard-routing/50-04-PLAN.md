```rust
/// Complete event envelope: zero-copy payload + response channel + routing info.
pub struct ShardEvent {
    /// Raw event bytes — bytes::Bytes is O(1) clone (Arc-backed). Zero copy.
    pub payload: bytes::Bytes,
    /// Stream name for routing to correct Shard state machine.
    pub stream_name: Arc<str>,
    /// Precomputed shard_hint from ingest parser (Phase 48).
    pub shard_hint: u32,
    /// Response channel — shard sends result back to listener.
    /// None for fire-and-forget paths (TCP push without ACK needed).
    pub response_tx: Option<oneshot::Sender<ShardResult>>,
}

#[derive(Debug)]
pub enum ShardResult {
    Ok,
    Err(ShardDispatchError),
}

#[derive(Debug)]
pub enum ShardDispatchError {
    Down,
    ProcessingError(String),
}
```

```rust
fn shard_event_loop(shard_index: usize, rx: crossbeam_channel::Receiver<ShardEvent>) {
    // Each shard runs a tokio current_thread runtime on its pinned OS thread.
    // This lets async code run inside the shard without cross-thread task migration.
    let rt = tokio::runtime::Builder::new_current_thread()
        .enable_all()
        .build()
        .expect("failed to build per-shard tokio runtime");

    rt.block_on(async move {
        while let Ok(event) = rx.recv() {
            // TODO(50-05): route to per-shard Shard state machine
            crate::shard::metrics::record_shard_event(shard_index, crate::shard::metrics::Outcome::Accepted);
            if let Some(tx) = event.response_tx {
                let _ = tx.send(ShardResult::Ok);
            }
        }
    });
}
```

Add property test for backpressure contract:
```rust
#[cfg(test)]
mod backpressure_tests {
    use super::*;
    use std::sync::atomic::{AtomicU64, Ordering};

    // Simulates the backpressure contract: inbox capacity=1, push N events,
    // exactly (N-1) should be dropped (try_send returns Err on full).
    #[test]
    fn inbox_full_drops_excess_events() {
        let (tx, _rx) = crossbeam_channel::bounded::<ShardEvent>(1);
        let drop_count = Arc::new(AtomicU64::new(0));

        let first = ShardEvent {
            payload: bytes::Bytes::from_static(b"event0"),
            stream_name: Arc::from("s"),
            shard_hint: 0,
            response_tx: None,
        };
        // First send succeeds.
        assert!(tx.try_send(first).is_ok());

        // Subsequent sends on a full inbox must fail.
        for _ in 1..10u64 {
            let ev = ShardEvent {
                payload: bytes::Bytes::from_static(b"eventN"),
                stream_name: Arc::from("s"),
                shard_hint: 0,
                response_tx: None,
            };
            if tx.try_send(ev).is_err() {
                drop_count.fetch_add(1, Ordering::Relaxed);
            }
        }
        assert_eq!(drop_count.load(Ordering::Relaxed), 9);
    }
}
```
