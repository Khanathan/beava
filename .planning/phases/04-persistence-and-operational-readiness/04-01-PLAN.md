1. Expand `src/state/snapshot.rs` (after Task 1's OperatorState enum) with:

```rust
const SNAPSHOT_FORMAT_VERSION: u8 = 1;

/// Serializable pipeline definition for snapshot persistence.
/// Mirrors StreamDefinition but with serializable FeatureDef.
/// FeatureDef already derives Clone but Expr inside Derive does not derive Serialize.
/// Solution: store the raw RegisterRequest JSON for each stream, and re-parse on load.
/// This avoids serializing the Expr AST.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SerializablePipeline {
    pub name: String,
    pub key_field: String,
    pub raw_register_json: serde_json::Value,
}

/// Serializable entity state for snapshot persistence.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SerializableEntityState {
    pub live_operators: Vec<(String, OperatorState)>,
    pub static_features: Vec<(String, StaticFeature)>,
    pub last_event_at: Option<SystemTime>,
}

/// Top-level serializable snapshot state.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SnapshotState {
    pub entities: Vec<(String, SerializableEntityState)>,
    pub pipelines: Vec<SerializablePipeline>,
}

pub fn save_snapshot(data: &SnapshotState) -> Vec<u8> {
    let mut buf = vec![SNAPSHOT_FORMAT_VERSION];
    buf.extend_from_slice(&postcard::to_stdvec(data).expect("snapshot serialization failed"));
    buf
}

pub fn load_snapshot(bytes: &[u8]) -> Option<SnapshotState> {
    if bytes.is_empty() {
        return None;
    }
    let version = bytes[0];
    if version != SNAPSHOT_FORMAT_VERSION {
        eprintln!("Snapshot version mismatch: found {}, expected {}. Starting fresh.", version, SNAPSHOT_FORMAT_VERSION);
        return None;
    }
    postcard::from_bytes(&bytes[1..]).ok()
}
```

Note about pipelines in snapshots: The Expr type in FeatureDef::Derive is an AST from the expression parser. Check if Expr derives Serialize/Deserialize. If it does, we can serialize FeatureDef directly. If not, store the raw JSON that was used to register (the RegisterRequest JSON) and re-parse on load via convert_register_request. /// Clone full state for snapshot serialization.
```rust
/// AHashMap is not directly serializable -- convert to Vec<(K, V)>.
pub fn clone_for_snapshot(&self) -> Vec<(String, SerializableEntityState)> {
    self.entities.iter().map(|(key, entity)| {
        (key.clone(), SerializableEntityState {
            live_operators: entity.live_operators.clone(),
            static_features: entity.static_features.iter()

}
```

```rust
/// Remove entities whose last_event_at is older than `ttl` from `now`.
/// Entities with `last_event_at = None` are not evicted (never received an event).
/// Returns the count of evicted entities.
pub fn remove_expired_entities(&mut self, now: SystemTime, ttl: std::time::Duration) -> usize {
    let before = self.entities.len();
    self.entities.retain(|_key, entity| {
        match entity.last_event_at {
            None => true, // Never pushed -- don't evict
            Some(last) => {
                now.duration_since(last).unwrap_or(std::time::Duration::ZERO) < ttl
            }
        }
    });
    before - self.entities.len()
}
```

```rust
/// Return the maximum window duration across all registered streams.
/// Returns Duration::ZERO if no streams are registered.
pub fn max_window_duration(&self) -> std::time::Duration {
    self.streams.values()
        .flat_map(|s| s.features.iter())
        .filter_map(|(_, def)| match def {
            FeatureDef::Count { window, .. } => Some(*window),
            FeatureDef::Sum { window, .. } => Some(*window),
            FeatureDef::Avg { window, .. } => Some(*window),
            FeatureDef::Derive { .. } => None,
        })
        .max()
        .unwrap_or(std::time::Duration::ZERO)
}
```

5. Create `src/state/eviction.rs`:

```rust
//! TTL-based key eviction.
//!
//! Keys with no events for 2x the largest window are evicted from memory.
//! Evicted keys re-initialize fresh on next event (CLAUDE.md spec).

use std::time::{Duration, SystemTime};
use crate::state::store::StateStore;
use crate::engine::pipeline::PipelineEngine;

/// Evict entity keys whose last_event_at is older than ttl_multiplier * max_window.
/// Returns the number of evicted keys.
pub fn evict_expired_keys(
    store: &mut StateStore,
    engine: &PipelineEngine,
    now: SystemTime,
    ttl_multiplier: u32,
) -> usize {
    let max_window = engine.max_window_duration();
    if max_window.is_zero() {
        return 0; // No streams registered -- nothing to evict
    }
    let ttl = max_window * ttl_multiplier;
    store.remove_expired_entities(now, ttl)
}
```
