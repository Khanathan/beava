    In `crates/beava-core/src/agg_apply.rs`, add the new wrapper:
    ```rust
    pub fn apply_with_modifiability(
        source_desc: &EventDescriptor,
        row: &Row,
        event_time_ms: i64,
        event_id: u64,
        registry: &Registry,
        state_tables: &mut BTreeMap<String, AggStateTable>,
        metrics: &mut ModifiabilityMetrics, // new: eviction counter, promotion counter
    ) {
        if !source_desc.modifiable {
            // Fast path — identical to Phase 14.
            return apply_event_to_aggregations(
                &source_desc.name, row, event_time_ms, event_id,
                registry, state_tables);
        }

        // Modifiable source: determine per-feature fast vs slow path.
        for desc in registry.compiled_aggregations_for_source(&source_desc.name) {
            let entity_key = match EntityKey::from_row(&desc.group_keys, row) {
                Some(k) => k, None => continue,
            };
            let table = state_tables.entry(desc.node_name.clone()).or_default();
            let entity_row = table.get_or_init_row(&entity_key, &desc);

            let forward = event_time_ms >= entity_row.max_event_time_ms;

            for (i, feat) in desc.features.iter().enumerate() {
                if feat.kind.is_tier_3_replay_candidate() {
                    let depth = feat.modification_log_depth
                        .unwrap_or(source_desc.modification_log_depth);
                    let log = entity_row.ensure_mod_log(i, depth);
                    let entry = ModLogEntry {
                        event_time_ms, event_id, row: row.clone(),
                    };
                    let outcome = if forward { log.push(entry) } else { log.insert(entry) };
                    match outcome {
                        InsertOutcome::Pushed => {
                            // Fast path: normal update.
                            entity_row.features[i].update_with_row(
                                row, event_time_ms, feat.descriptor.field.as_deref(),
                                feat.descriptor.where_expr.as_ref());
                        }
                        InsertOutcome::Inserted { .. } | InsertOutcome::EvictedOldest { .. } => {
                            // Slow path: rebuild state from log.
                            let log_slice = entity_row.mod_logs[i].as_ref().unwrap().as_slice();
                            entity_row.features[i] = entity_row.features[i]
                                .replay_from_log(log_slice, &feat.descriptor);
                        }
                        InsertOutcome::RejectedOlderThanOldest => {
                            metrics.events_rejected_log_underflow.inc();
                            // Do not alter state; rate-limited WARN log (reuse Phase 14 rate-limit helper).
                        }
                    }
                } else {
                    // Tier C path: plain update, no log.
                    entity_row.features[i].update_with_row(
                        row, event_time_ms, feat.descriptor.field.as_deref(),
                        feat.descriptor.where_expr.as_ref());
                }
            }

            entity_row.observe_event_time(event_time_ms);
        }
    }
    ```
