    Create `crates/beava-core/src/mod_log.rs`:
    ```rust
    use crate::row::Row;
    use serde::{Deserialize, Serialize};

    #[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    pub struct ModLogEntry {
        pub event_time_ms: i64,
        pub event_id: u64,
        pub row: Row,
    }

    #[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
    pub struct ModificationLog {
        capacity: u16,
        /// Sorted by (event_time_ms, event_id) ascending. Len ≤ capacity.
        entries: Vec<ModLogEntry>,
    }

    #[derive(Debug, PartialEq)]
    pub enum InsertOutcome {
        Pushed,                                // appended to end (newest)
        Inserted { position: usize },          // inserted mid-log
        EvictedOldest { evicted: ModLogEntry, position: usize },
        RejectedOlderThanOldest,
    }

    impl ModificationLog {
        pub fn new(capacity: u16) -> Self { ... }
        pub fn push(&mut self, e: ModLogEntry) -> InsertOutcome { ... }
        pub fn insert(&mut self, e: ModLogEntry) -> InsertOutcome { ... }
        pub fn as_slice(&self) -> &[ModLogEntry] { &self.entries }
        pub fn len(&self) -> usize { self.entries.len() }
        pub fn capacity(&self) -> u16 { self.capacity }
    }
    ```

    Insertion algorithm (both `push` and `insert` delegate to this):
    - Binary-search by `(event_time_ms, event_id)` → insertion index `i`.
    - If `self.entries.len() < capacity` → `entries.insert(i, e)`; returns
      `Inserted{position:i}` (or `Pushed` if `i == entries.len()`).
    - If full and `i == 0` → `RejectedOlderThanOldest` (do not evict the
      currently-oldest entry to make room for an even-older one).
    - If full and `i > 0` → evict `entries[0]`, then `entries.insert(i-1, e)`
      at the shifted position; returns `EvictedOldest{...}`.

    Export from `crates/beava-core/src/lib.rs`:
    ```rust
    pub mod mod_log;
    ```

    Unit tests + proptest in the same file.

    ```rust
    #[derive(Debug, Clone, Copy, PartialEq, Eq)]
    pub enum Tier {
        /// Subtractive-exact: commutative, no replay needed.
        C,
        /// Approximate, bounded drift — log allocated, replay best-effort.
        A,
        /// Deterministic-reject: analyzer errors at register time.
        B,
    }

    impl AggKind {
        pub fn tier(self) -> Tier {
            use AggKind::*;
            match self {
                // Tier C — commutative
                Count | Sum | Avg | Variance | StdDev | Ratio
                    | Histogram | HourOfDayHistogram | DowHourHistogram
                    | EventTypeMix | DecayedSum | DecayedCount | BurstCount
                    | BloomMember | Twa => Tier::C,

                // Tier A — approximate or order-sensitive-with-drift
                Ewma | EwVar | EwZScore
                    | CountDistinct | Percentile | TopK | Entropy
                    | SeasonalDeviation => Tier::A,

                // Tier B — deterministic-reject
                Min | Max | First | Last | FirstN | LastN | Lag
                    | FirstSeen | LastSeen | Age | HasSeen
                    | TimeSince | TimeSinceLastN
                    | Streak | MaxStreak | NegativeStreak
                    | FirstSeenInWindow
                    | RateOfChange | InterArrivalStats
                    | DeltaFromPrev | Trend | TrendResidual
                    | OutlierCount | ValueChangeCount | ZScore
                    | MostRecentN | ReservoirSample
                    | GeoVelocity | GeoDistance | GeoSpread
                    | UniqueCells | GeoEntropy | DistanceFromHome => Tier::B,
            }
        }

        /// Does this op allocate a per-entity ModificationLog at runtime when
        /// its source is `modifiable=True`?
        pub fn is_tier_3_replay_candidate(self) -> bool {
            matches!(self.tier(), Tier::A | Tier::B)
        }
    }
    ```

    (Note: `GeoDistance`, `GeoSpread`, `UniqueCells`, `GeoEntropy` are
    classified as Tier B here — they're position/sequence-shaped over the
    recent event window. Double-check this against the analyzer catalogue in
    Plan 03; if any are re-classified to Tier C, update both in lock-step.)
