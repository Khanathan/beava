**Add new save/load functions (keep existing save_snapshot/load_snapshot for backward compat in tests):**

```rust
/// Serialize a BaseSnapshotState to bytes: [version 6][type_tag 0x00][postcard data]
pub fn save_base_snapshot(data: &BaseSnapshotState) -> Result<Vec<u8>, postcard::Error> {
    let mut buf = vec![SNAPSHOT_FORMAT_VERSION, 0x00]; // 0x00 = base
    buf.extend_from_slice(&postcard::to_stdvec(data)?);
    Ok(buf)
}

/// Serialize a DeltaSnapshotState to bytes: [version 6][type_tag 0x01][postcard data]
pub fn save_delta_snapshot(data: &DeltaSnapshotState) -> Result<Vec<u8>, postcard::Error> {
    let mut buf = vec![SNAPSHOT_FORMAT_VERSION, 0x01]; // 0x01 = delta
    buf.extend_from_slice(&postcard::to_stdvec(data)?);
    Ok(buf)
}

/// Load a v6 snapshot file (base or delta). Returns None on version mismatch or corruption.
pub fn load_snapshot_file(bytes: &[u8]) -> Option<SnapshotFile> {
    if bytes.len() < 2 { return None; }
    if bytes[0] != SNAPSHOT_FORMAT_VERSION { return None; }
    match bytes[1] {
        0x00 => postcard::from_bytes::<BaseSnapshotState>(&bytes[2..]).ok().map(SnapshotFile::Base),
        0x01 => postcard::from_bytes::<DeltaSnapshotState>(&bytes[2..]).ok().map(SnapshotFile::Delta),
        _ => None,
    }
}

/// Load a legacy v5 snapshot. Returns None if not v5 or corrupt.
pub fn load_legacy_v5(bytes: &[u8]) -> Option<SnapshotState> {
    if bytes.is_empty() { return None; }
    if bytes[0] != LEGACY_V5_FORMAT { return None; }
    postcard::from_bytes(&bytes[1..]).ok()
}
```

pub fn load_snapshot(bytes: &[u8]) -> Option<SnapshotState> {
    if bytes.is_empty() { return None; }
    let version = bytes[0];
    if version == LEGACY_V5_FORMAT {
        return postcard::from_bytes(&bytes[1..]).ok();
    }
    if version != SNAPSHOT_FORMAT_VERSION {
        eprintln!("Snapshot version mismatch: found {}, expected {}. Starting fresh.", version, SNAPSHOT_FORMAT_VERSION);
        return None;
    }
    // v6: only handle base type via this legacy function
    if bytes.len() < 2 { return None; }
    if bytes[1] != 0x00 { return None; } // Only base snapshots via legacy API
    let base: BaseSnapshotState = postcard::from_bytes(&bytes[2..]).ok()?;
    Some(SnapshotState {
        entities: base.entities,
        pipelines: base.pipelines,
        backfill_complete: base.backfill_complete,
    })
}

```rust
/// Apply a delta snapshot on top of existing state.
/// For each deleted key: remove from entities.
/// For each changed entity: replace the entire EntityState (same conversion as restore_from_snapshot).
pub fn apply_delta(&mut self, changed_entities: Vec<(String, SerializableEntityState)>, deleted_keys: Vec<String>) {
    for key in deleted_keys {
        self.entities.remove(&key);
    }
    for (key, state) in changed_entities {
        let mut streams = AHashMap::new();
        for (stream_name, stream_state) in state.streams {
            streams.insert(stream_name, StreamEntityState {
                operators: stream_state.operators,
                last_event_at: stream_state.last_event_at,
            });
        }
        let entity = EntityState {
            streams,
            static_features: state.static_features.into_iter().collect(),
        };
        self.entities.insert(key, entity);
    }
}
```
