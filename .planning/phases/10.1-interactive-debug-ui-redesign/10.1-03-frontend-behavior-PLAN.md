'use strict';

// Minimal DOM helper — textContent-only, no innerHTML.
function el(tag, attrs, children) {
  const node = document.createElement(tag);
  if (attrs) {
    for (const k of Object.keys(attrs)) {
      if (k === 'text') {
        node.textContent = attrs[k];
      } else if (k === 'class') {
        node.className = attrs[k];
      } else if (k.startsWith('on')) {
        // Event listener: attach via addEventListener, never as inline attribute string
        node.addEventListener(k.slice(2).toLowerCase(), attrs[k]);
      } else {
        node.setAttribute(k, attrs[k]);
      }
    }
  }
  if (Array.isArray(children)) {
    for (const c of children) { if (c) node.appendChild(c); }
  } else if (children) {
    node.appendChild(children);
  }
  return node;
}

function clear(node) {
  while (node.firstChild) node.removeChild(node.firstChild);
}

// SVG elements require the SVG namespace — document.createElement won't work.
function svgEl(tag, attrs, children) {
  const node = document.createElementNS('http://www.w3.org/2000/svg', tag);
  if (attrs) {
    for (const k of Object.keys(attrs)) {
      if (k === 'text') {
        node.textContent = attrs[k];
      } else if (k === 'class') {
        node.setAttribute('class', attrs[k]);
      } else {
        node.setAttribute(k, attrs[k]);
      }
    }
  }
  if (Array.isArray(children)) {
    for (const c of children) { if (c) node.appendChild(c); }
  } else if (children) {
    node.appendChild(children);
  }
  return node;
}

// Format events/sec as compact label: '—/s', '0/s', '12.3/s', '450/s',
// '1.2k/s', '12k/s', '1.2M/s'. RESEARCH Pattern 3.
function formatFps(n) {
  if (n == null || Number.isNaN(n)) return '—/s';
  if (n === 0) return '0/s';
  const abs = Math.abs(n);
  if (abs < 10)   return n.toFixed(1) + '/s';
  if (abs < 1000) return Math.round(n) + '/s';
  if (abs < 1e6) {
    const k = n / 1000;
    return (k < 10 ? k.toFixed(1) : Math.round(k)) + 'k/s';
  }
  const m = n / 1e6;
  return (m < 10 ? m.toFixed(1) : Math.round(m)) + 'M/s';
}

function formatBytes(n) {
  if (n == null || Number.isNaN(n)) return '—';
  if (n === 0) return '0 B';
  const units = ['B', 'KB', 'MB', 'GB'];
  let i = 0;
  let v = n;
  while (v >= 1024 && i < units.length - 1) { v /= 1024; i++; }
  return (v < 10 ? v.toFixed(1) : Math.round(v)) + ' ' + units[i];
}

function formatFeatureValue(v) {
  if (v == null) return 'null';
  if (typeof v === 'boolean') return v ? 'true' : 'false';
  if (typeof v === 'number') {
    if (Number.isInteger(v)) return String(v);
    return v.toFixed(2);
  }
  if (typeof v === 'string') return v;
  // Objects/arrays — JSON.stringify is safe because the result goes through textContent.
  try { return JSON.stringify(v); } catch (_) { return String(v); }
}

function fmtTime(d) {
  if (!d) return '—';
  const t = (d instanceof Date) ? d : new Date(d);
  const hh = String(t.getHours()).padStart(2, '0');
  const mm = String(t.getMinutes()).padStart(2, '0');
  const ss = String(t.getSeconds()).padStart(2, '0');
  return hh + ':' + mm + ':' + ss;
}

// Build the tally-mark icon as SVG using createElementNS (XSS-safe).
function makeTallyMarkIcon() {
  const svg = svgEl('svg', { width: '48', height: '48', viewBox: '0 0 24 24',
    fill: 'none', stroke: 'currentColor', 'stroke-width': '2',
    'stroke-linecap': 'round' });
  for (const [x1, y1, x2, y2] of [
    [4, 4, 4, 20], [9, 4, 9, 20], [14, 4, 14, 20], [19, 4, 19, 20],
    [2, 18, 21, 6]
  ]) {
    svg.appendChild(svgEl('line', { x1: x1, y1: y1, x2: x2, y2: y2 }));
  }
  return svg;
}

  return state.topology.nodes.find(function(n) { return n.name === name; }) || null;

function renderDrillInPanel(name) {
  const panel = document.getElementById('drill-in-panel');
  if (!panel) return;
  clear(panel);

  if (name == null) {
    panel.setAttribute('data-empty', 'true');
    panel.appendChild(el('p', { class: 'drill-in-placeholder',
      text: 'Select a stream to see details' }));
    return;
  }
  panel.removeAttribute('data-empty');

  const node = findNode(name);
  if (!node) return;

  renderStreamHeader(panel, node);
  if (node.kind === 'view') {
    // Views: omit Memory and Throughput; State → Features
    renderFeaturesSection(panel, node);
    renderEntityLookupSection(panel, node);
  } else {
    renderMemorySection(panel, node, state.memory);
    renderStateSection(panel, node);
    renderThroughputSection(panel, node, state.throughput);
    renderEntityLookupSection(panel, node);
  }
}

function renderStreamHeader(panel, node) {

  // Remove any previous empty-state overlay (in case topology was re-fetched

      paddingX: 16, paddingY: 12,
    });

    (svgH - 24) / Math.max(bbox.height, 1),

  const ty = (svgH - bbox.height * scale) / 2 - bbox.y * scale;

async function lookupEntityForSelectedStream(rawKey) {
  const result = document.getElementById('entity-result');
  if (!result) return;
  clear(result);

  const key = (rawKey || '').trim();
  const selectedStream = state.selectedStream;
  if (!key) {
    result.appendChild(el('p', { class: 'entity-idle',
      text: 'Enter a key above to inspect its features.' }));
    return;
  }
  if (!selectedStream) return;

  const url = '/debug/key/' + encodeURIComponent(key);
  let res;
  try { res = await fetch(url); }
  catch (e) {
    result.appendChild(el('div', { class: 'error-state' }, [
      el('h3', { text: 'Lookup failed' }),
      el('p', { text: 'Network error: ' + String(e.message || e) }),
    ]));
    return;
  }

  if (res.status === 404) {
    result.appendChild(el('div', { class: 'empty-state-result' }, [
      el('h3', { text: 'No features for "' + key + '"' }),
      el('p', { text: 'This key has not received any events recently, or has been evicted by TTL.' }),
    ]));
    return;
  }
  if (!res.ok) {
    result.appendChild(el('div', { class: 'error-state' }, [
      el('h3', { text: 'Lookup failed' }),
      el('p', { text: 'Server returned HTTP ' + res.status }),
    ]));
    return;
  }

  let data;
  try { data = await res.json(); }
  catch (_) {
    result.appendChild(el('div', { class: 'error-state' }, [
      el('h3', { text: 'Lookup failed' }),
      el('p', { text: 'Invalid JSON response' }),
    ]));
    return;
  }

  // Prefer computed_features, fall back to features (Phase 10 compat).
  const computed = data.computed_features || data.features || {};
  const filtered = {};
  for (const fullName of Object.keys(computed)) {
    const dot = fullName.indexOf('.');
    if (dot <= 0) continue;                      // orphan: skip
    const source = fullName.substring(0, dot);
    if (source !== selectedStream) continue;     // other stream: skip
    const label = fullName.substring(dot + 1);
    filtered[label] = computed[fullName];
  }

  if (Object.keys(filtered).length === 0) {
    result.appendChild(el('div', { class: 'empty-state-result' }, [
      el('h3', { text: 'No features for "' + key + '"' }),
      el('p', { text: 'Stream "' + selectedStream +
        '" has not computed features for this key.' }),
    ]));
    return;
  }

  const grid = el('div', { class: 'entity-feature-grid' });
  for (const name of Object.keys(filtered)) {
    const value = filtered[name];
    grid.appendChild(el('div', { class: 'entity-feature-cell' }, [
      el('div', { class: 'label', text: name }),
      el('div', { class: 'value', text: formatFeatureValue(value) }),
    ]));
  }
  result.appendChild(grid);
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| `Browser → app.js → /debug/topology, /debug/throughput, /debug/memory, /debug/key/{key}` | User-typed entity keys flow through `/debug/key/{key}`; all response strings flow into the DOM via textContent |
| `Backend raw_register_jsons → /debug/topology → renderStateSection` | User-supplied where-clauses, expressions, feature names reach the DOM as operator summaries |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-10.1-10 | Tampering (XSS) | Any user-supplied string reaching the drill-in panel (feature names, where-clauses, expressions, entity keys, feature values, stream names) | mitigate | Every DOM write uses `.textContent` via the `el({text: ...})` helper or direct assignment. d3 writes use `.text()` on selections. SVG elements use `createElementNS`. Regression test `app_js_has_no_innerhtml_or_eval_sinks` at tests/test_debug_ui.rs:762-787 asserts the forbidden-substring list at test time and fails loudly on any violation. |
| T-10.1-11 | Tampering (URL injection) | Entity key user-typed into the lookup input | mitigate | `encodeURIComponent(key)` is applied before concatenation into `/debug/key/{key}` path. Backend `src/server/ui.rs:57` additionally rejects `..`, leading slashes, and null bytes for the static asset route (not the debug key route, but the pattern establishes the defense-in-depth discipline). |
| T-10.1-12 | Tampering (SVG injection) | The tally-mark icon in the empty state | mitigate | Icon is built via `document.createElementNS('http://www.w3.org/2000/svg', ...)` with static line coordinates — no user input reaches SVG attribute values. |
| T-10.1-13 | Denial of service (render loop) | Per-tick full DAG re-render causing 5-30ms stalls on every 1 Hz tick | mitigate | `render(inner, g)` is called exactly ONCE in `loadTopology`. The tick callback uses `d3.select(...).text(...)` to update `tspan` elements only (RESEARCH Pattern 6). Verify via browser perf tab — each tick should be sub-millisecond. |
| T-10.1-14 | Denial of service (network retry storm) | Failed `/debug/throughput` fetch triggering exponential retry | mitigate | The tick callback swallows errors silently and the next tick retries on the 1000ms cadence. No retry loop, no backoff. Worst-case network cost is one extra request per second, matching the happy-path cost. |
| T-10.1-15 | Tampering (race during pause toggle) | Rapid Pause/Resume clicks causing overlapping fetches | accept | The `state.paused` check runs at the top of each tick(). If a fetch is in flight when pause fires, it completes and updates state — the next tick sees paused and short-circuits. No data corruption, worst case is one stale tick after the pause click. |
| T-10.1-16 | Information disclosure (where-clause text) | User-supplied where-clauses rendered in the State section | accept | Same accept-list as Plan 01 — the debug UI is localhost-only. |
