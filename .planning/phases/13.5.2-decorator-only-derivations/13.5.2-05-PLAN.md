# BEFORE (broken under new contract):
@bv.event
class Click:
    user_id: str
    page: str

Tagged = Click.with_columns(source=bv.lit("web")).named("Tagged")  # ← raw chain

@bv.table(key="user_id")
def UserClicks(tagged: Click):  # ← annotated with raw event class
    return tagged.group_by("user_id").agg(...)

app.register(Click, Tagged, UserClicks)  # ← Tagged is EventDerivation → REJECT

# AFTER (compliant with D-01 + D-02):
@bv.event
class Click:
    user_id: str
    page: str

@bv.event
def Tagged(click: Click):  # ← @bv.event def wraps the chain
    return click.with_columns(source=bv.lit("web"))

@bv.table(key="user_id")
def UserClicks(tagged: Tagged):  # ← annotation is the @bv.event def output (carries _is_bv_event_function marker)
    return tagged.group_by("user_id").agg(...)

app.register(Click, Tagged, UserClicks)  # ← Tagged is @bv.event def output → ACCEPT
