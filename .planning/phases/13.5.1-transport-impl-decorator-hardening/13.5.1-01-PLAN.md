def _resolve_upstream_proxies(fn: Callable[..., Any]) -> list[Any]:
    sig = inspect.signature(fn)
    params = list(sig.parameters.values())
    if not params:
        raise TypeError(
            f"@bv.table function {fn.__name__!r} must take at least one parameter"
        )
    try:
        resolved = get_type_hints(fn)
    except Exception:
        resolved = {}
    proxies: list[Any] = []
    for p in params:
        ann = resolved.get(p.name, p.annotation)   # <— silently yields
        if isinstance(ann, str):                    #    inspect.Parameter.empty
            ann = fn.__globals__.get(ann, ann)      #    when annotation missing
        proxies.append(ann)                         #    → AttributeError downstream
    return proxies
