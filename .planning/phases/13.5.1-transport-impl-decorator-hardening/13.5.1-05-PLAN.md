    self,
    table: str,
    key: str | list[str | int | bool] | None = None,

def batch_get(
    self,
    requests: list[
        tuple[str, str | list[str | int | bool]]
        | tuple[str, str | list[str | int | bool], list[str] | None]
    ],
    # D-03 ADDS: per-entry optional `features` via 3-tuple form. 2-tuple stays valid.
) -> list[dict[str, Any]]:
    t = self._require_transport()
    coerced: list[tuple[str, str | list[Any]] | tuple[str, str | list[Any], list[str] | None]] = []
    for entry in requests:
        if len(entry) == 2:
            tbl, k = entry
            coerced.append((tbl, k))  # type: ignore[arg-type]
        elif len(entry) == 3:
            tbl, k, feats = entry
            coerced.append((tbl, k, feats))  # type: ignore[arg-type]
        else:
            raise TypeError(
                f"batch_get request entry must be a 2- or 3-tuple "
                f"(table, key) or (table, key, features); got {len(entry)}-tuple"
            )
    result: list[dict[str, Any]] = t.send_batch_get(requests=coerced)
    return result
