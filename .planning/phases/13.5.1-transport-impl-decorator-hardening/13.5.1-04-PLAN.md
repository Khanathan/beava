def _resolve_upstream_proxies(fn: Callable[..., Any]) -> list[Any]:
    sig = inspect.signature(fn)
    params = list(sig.parameters.values())
    if not params:
        raise TypeError(
            f"@bv.table function {fn.__name__!r} must take at least one parameter"
        )
    try:
        resolved = get_type_hints(fn)
    except Exception:
        resolved = {}
    proxies: list[Any] = []
    for p in params:
        ann = resolved.get(p.name, p.annotation)
        if ann is inspect.Parameter.empty:
            raise TypeError(
                f"@bv.table function {fn.__name__!r} parameter {p.name!r} must be "
                f"annotated with the upstream event class — "
                f"e.g. def {fn.__name__}({p.name}: Click): ..."
            )
