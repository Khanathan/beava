       ```rust
       use pyo3::prelude::*;
       use pyo3::exceptions::{PyStopIteration, PyRuntimeError};
       use crossbeam_channel::{Receiver, Sender, unbounded};
       use dashmap::DashMap;
       use std::sync::{Arc, Mutex};
       use std::sync::atomic::{AtomicU64, Ordering};

       pub struct WatchItem { pub seq: u64, pub event_json: String, pub value_json: Option<String> }

       pub struct WatcherEntry { pub filter: (String, String), pub tx: Sender<WatchItem> }

       pub struct WatcherRegistry { pub map: DashMap<u64, WatcherEntry>, pub next_id: AtomicU64 }

       impl WatcherRegistry {
           pub fn new() -> Arc<Self> { ... }
           pub fn register(&self, stream: String, key: String) -> (u64, Receiver<WatchItem>) { ... }
           pub fn unregister(&self, id: u64) { self.map.remove(&id); }
           pub fn dispatch(&self, stream: &str, key: &str, seq: u64, event_json: &str, value_json: Option<&str>) {
               for entry in self.map.iter() {
                   let (f_stream, f_key) = &entry.filter;
                   if f_stream == stream && f_key == key {
                       let _ = entry.tx.try_send(WatchItem { seq, event_json: event_json.into(),
                                                             value_json: value_json.map(|s| s.into()) });
                       // Full/closed → drop (v0 policy); caller already logs in the pipeline bg thread on drop count.
                   }
               }
           }
       }

       #[pyclass]
       pub struct Watcher {
           rx: Receiver<WatchItem>,
           status: Arc<Mutex<StreamingStatus>>,
           registry: std::sync::Weak<WatcherRegistry>,
           id: u64,
       }

       #[pymethods]
       impl Watcher {
           fn __iter__(slf: PyRef<Self>) -> PyRef<Self> { slf }

           fn __next__(&mut self, py: Python) -> PyResult<PyObject> {
               // 1. Quick status check.
               if let StreamingStatus::Stopped(reason) = self.status.lock().unwrap().clone() {
                   return raise_for_reason(py, &reason);  // helper: maps StopReason → Python exception
               }
               // 2. Blocking recv with GIL released.
               let recv_result = py.allow_threads(|| self.rx.recv());
               match recv_result {
                   Ok(item) => build_py_dict(py, item),
                   Err(_) => {
                       // Channel closed → check status.
                       let s = self.status.lock().unwrap().clone();
                       if let StreamingStatus::Stopped(r) = s { raise_for_reason(py, &r) }
                       else { Err(PyStopIteration::new_err(())) }
                   }
               }
           }
       }

       impl Drop for Watcher {
           fn drop(&mut self) {
               if let Some(reg) = self.registry.upgrade() { reg.unregister(self.id); }
           }
       }

       fn raise_for_reason(py: Python, r: &StopReason) -> PyResult<PyObject> {
           match r {
               StopReason::UserRequested => Err(PyStopIteration::new_err(())),
               StopReason::Dropped { at_seq } => {
                   let errors = py.import("tally.errors")?;
                   let exc_cls = errors.getattr("SubscriberDroppedError")?;
                   let exc = exc_cls.call1((*at_seq,))?;
                   Err(PyErr::from_value(exc))
               }
               StopReason::Io(msg) => {
                   let errors = py.import("tally.errors")?;
                   let exc_cls = errors.getattr("TallyError")?;
                   let exc = exc_cls.call1((msg.clone(),))?;
                   Err(PyErr::from_value(exc))
               }
           }
       }

       fn build_py_dict(py: Python, item: WatchItem) -> PyResult<PyObject> {
           let json = py.import("json")?;
           let event = json.call_method1("loads", (item.event_json,))?;
           let value = match item.value_json {
               Some(s) => json.call_method1("loads", (s,))?.into(),
               None => py.None(),
           };
           let d = pyo3::types::PyDict::new(py);
           d.set_item("seq", item.seq)?;
           d.set_item("event", event)?;
           d.set_item("value", value)?;
           Ok(d.into())
       }
       ```
