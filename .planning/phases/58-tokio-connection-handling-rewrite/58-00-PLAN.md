    fn tokio_share_on_push_path_under_15_pct() {
        // Invoke the probe script.
        let out = Command::new("bash")
            .arg("scripts/samply-probe-tokio-share.sh")
            .output()
            .expect("samply-probe-tokio-share.sh must be executable");

        assert!(
            out.status.success(),
            "probe script failed: stdout={} stderr={}",

            .lines()
            .rev()

            .expect("probe must emit TOKIO_SHARE_PCT=<num|unknown>");
