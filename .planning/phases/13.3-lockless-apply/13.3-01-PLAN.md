Target accept-loop shape (from axum examples, adapted to Rc state):
```rust
use std::rc::Rc;
use std::cell::RefCell;
use tokio::net::TcpListener;
use hyper::server::conn::http1;
use hyper_util::rt::TokioIo;
use tower::Service;

pub async fn serve_local<S: 'static>(
    listener: TcpListener,
    router: axum::Router<Rc<RefCell<S>>>,
    state: Rc<RefCell<S>>,
    shutdown: impl std::future::Future<Output = ()>,
) -> std::io::Result<()> {
    let app = router.with_state(state);
    // Drive the accept loop on the current LocalSet — each connection is a spawn_local.
    tokio::pin!(shutdown);
    loop {
        tokio::select! {
            _ = &mut shutdown => return Ok(()),
            accept = listener.accept() => {
                let (stream, _addr) = accept?;
                let io = TokioIo::new(stream);
                let tower_svc = app.clone();
                // axum::Router IS a tower::Service; wrap via ServiceExt to produce
                // a hyper-compatible service. The whole future is !Send because
                // tower_svc borrows Rc<RefCell<S>>.
                tokio::task::spawn_local(async move {
                    let hyper_svc = hyper::service::service_fn(move |req| {
                        let mut svc = tower_svc.clone();
                        async move { svc.call(req).await }
                    });
                    let _ = http1::Builder::new()
                        .serve_connection(io, hyper_svc)
                        .await;
                });
            }
        }
    }
}
```
