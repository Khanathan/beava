    **3. Create `python/benches/blast_shape.py`** — pure-Python four-shape pool builder mirroring
       `crates/beava-bench/src/blast_shape.rs` semantics:

       ```python
       """Four-shape Pool=N pre-encoded frame builder for the Python bench harness.

       Mirrors crates/beava-bench/src/blast_shape.rs semantics:
         - fixed: one frame, reused N times
         - uniform: user_id rolls evenly over `cardinality` keys
         - zipfian: rejection-sampler over rank with alpha=1.0 default
         - mixed: round-robin over `event_count` distinct event names

       Pool entries are encoded TCP frames (`[u32 length BE][u16 op BE][u8 ct][payload]`) ready
       for socket.sendall — for TCP wire. For HTTP wire, returns plain JSON bytes (HTTP body
       only, no framing).
       """
       from __future__ import annotations
       import json
       import math
       import random
       import struct
       from dataclasses import dataclass
       from typing import Literal

       try:
           import msgpack
       except ImportError:
           msgpack = None  # only needed for wire_format="msgpack"

       OP_PUSH = 0x0010
       CT_JSON = 0x01
       CT_MSGPACK = 0x02

       BlastShapeName = Literal["fixed", "uniform", "zipfian", "mixed"]
       WireFormatName = Literal["json", "msgpack"]
       TransportName = Literal["http", "tcp"]

       @dataclass
       class PoolConfig:
           shape: BlastShapeName
           wire_format: WireFormatName
           transport: TransportName  # determines whether frames are wrapped in TCP envelope
           cardinality: int
           zipf_alpha: float
           mixed_event_count: int
           seed: int
           pipeline_event_name: str
           pipeline_key_field: str
           pipeline_extra_fields: dict[str, str]
           # For mixed: synthesized fallback if pipeline only registers one event.
           mixed_event_names: list[str]

       def _build_body(cfg: PoolConfig, key_idx: int, seq: int, rng: random.Random) -> dict:
           body = {
               cfg.pipeline_key_field: f"k{key_idx:08d}",
               "event_time": 1_000_000 + seq,
           }
           for field, ty in cfg.pipeline_extra_fields.items():
               if ty == "f64":
                   body[field] = rng.uniform(0.0, 1000.0)
               elif ty == "i64":
                   body[field] = rng.randint(0, 999_999)
               elif ty == "str":
                   body[field] = f"s{rng.randint(0, 999)}"
               else:
                   body[field] = 0
           return body

       def _encode_envelope(event_name: str, body: dict, wire: WireFormatName) -> tuple[int, bytes]:
           if wire == "json":
               payload = json.dumps({"event": event_name, "body": body}, ensure_ascii=False).encode("utf-8")
               return CT_JSON, payload
           elif wire == "msgpack":
               if msgpack is None:
                   raise ImportError("msgpack package required for wire_format=msgpack")
               payload = msgpack.packb({"event": event_name, "body": body}, use_bin_type=True)
               return CT_MSGPACK, payload
           else:
               raise ValueError(f"unknown wire format: {wire}")

       def _wrap_tcp_frame(op: int, ct: int, payload: bytes) -> bytes:
           # [u32 length BE][u16 op BE][u8 ct][payload]
           # length covers (op + ct + payload) but matches Phase 2.5 codec — see python/beava/_wire.py
           # for the canonical definition. We use it directly:
           from beava._wire import encode_frame
           return encode_frame(op, ct, payload)

       def _zipf_zeta(n: int, alpha: float) -> float:
           return sum(1.0 / (i ** alpha) for i in range(1, n + 1))

       class _ZipfianSampler:
           def __init__(self, k: int, alpha: float, seed: int) -> None:
               assert k > 0 and alpha > 0
               self.k = k
               self.alpha = alpha
               self.rng = random.Random(seed)
               self.zetan = _zipf_zeta(k, alpha)
               zeta2 = _zipf_zeta(2, alpha)
               self.eta = (1.0 - (2.0 / k) ** (1.0 - alpha)) / (1.0 - zeta2 / self.zetan)

           def sample(self) -> int:
               u = self.rng.random()
               uz = u * self.zetan
               if uz < 1.0:
                   return 0
               if uz < 1.0 + 0.5 ** self.alpha:
                   return 1
               v = self.rng.random()
               rank = int(self.k * ((self.eta * v - self.eta + 1.0) ** (1.0 / (1.0 - self.alpha))))
               return min(rank, self.k - 1)

       def build_pool(cfg: PoolConfig, n: int) -> list[bytes]:
           """Build N pre-encoded frames matching the requested shape."""
           rng = random.Random(cfg.seed)
           pool: list[bytes] = []

           # Helper to materialize one frame.
           def make_frame(event_name: str, key_idx: int, seq: int) -> bytes:
               body = _build_body(cfg, key_idx, seq, rng)
               ct, payload = _encode_envelope(event_name, body, cfg.wire_format)
               if cfg.transport == "tcp":
                   return _wrap_tcp_frame(OP_PUSH, ct, payload)
               else:
                   # HTTP — payload is the body only (no envelope wrapper, no event field — the
                   # event name goes in the URL path /push/{event_name})
                   return json.dumps(body, ensure_ascii=False).encode("utf-8")

           if cfg.shape == "fixed":
               frame = make_frame(cfg.pipeline_event_name, 0, 0)
               return [frame] * n  # list of references — same bytes object reused
           elif cfg.shape == "uniform":
               for seq in range(n):
                   key_idx = rng.randrange(0, max(cfg.cardinality, 1))
                   pool.append(make_frame(cfg.pipeline_event_name, key_idx, seq))
           elif cfg.shape == "zipfian":
               sampler = _ZipfianSampler(max(cfg.cardinality, 1), cfg.zipf_alpha, cfg.seed ^ 0xDEAD)
               for seq in range(n):
                   key_idx = sampler.sample()
                   pool.append(make_frame(cfg.pipeline_event_name, key_idx, seq))
           elif cfg.shape == "mixed":
               m = max(cfg.mixed_event_count, 1)
               names = cfg.mixed_event_names[:m] if len(cfg.mixed_event_names) >= m else (
                   cfg.mixed_event_names + [f"e{i}" for i in range(len(cfg.mixed_event_names), m)]
               )
               for seq in range(n):
                   ev = names[seq % m]
                   key_idx = rng.randrange(0, max(cfg.cardinality, 1))
                   pool.append(make_frame(ev, key_idx, seq))
           else:
               raise ValueError(f"unknown shape: {cfg.shape}")
           return pool
       ```

    **4. Create `python/benches/blast.py`** — the multi-process driver:

       ```python
       """python/benches/blast.py — multi-process Python bench harness for Phase 19.

       Public surface mirrors crates/beava-bench/src/bin/beava-bench-v18.rs as closely as the
       Python SDK allows. ProcessPoolExecutor with N = os.cpu_count() - 1 workers; each worker
       creates its own bv.App, builds its own per-worker pool slice, and pushes events in a
       tight loop calling transport.send_push() (or http POST for HTTP transport).

       Per CONTEXT.md D-08..D-11.
       """
       from __future__ import annotations
       import argparse
       import json
       import multiprocessing as mp
       import os
       import socket
       import subprocess
       import sys
       import time
       from concurrent.futures import ProcessPoolExecutor, as_completed
       from pathlib import Path

       _BENCH_DIR = Path(__file__).resolve().parent
       sys.path.insert(0, str(_BENCH_DIR.parent))  # so `import beava` works without install

       import httpx  # noqa: E402

       from beava._wire import OP_PUSH, encode_frame, read_frame  # noqa: E402
       from benches._configs import (  # noqa: E402
           load_pipeline_config, register_payload, event_name, key_field, extra_fields,
       )
       from benches.blast_shape import (  # noqa: E402
           PoolConfig, build_pool,
       )

       def parse_args() -> argparse.Namespace:
           p = argparse.ArgumentParser(description="Beava Python bench harness — Phase 19")
           p.add_argument("--total-events", type=int, default=None,
                          help="Send N events end-to-end and exit (default: --duration-secs mode)")
           p.add_argument("--blast-shape", choices=["fixed", "uniform", "zipfian", "mixed"], default="fixed")
           p.add_argument("--transport", choices=["http", "tcp"], default="tcp")
           p.add_argument("--wire-format", choices=["json", "msgpack"], default="json")
           p.add_argument("--pipeline", default="small")
           p.add_argument("--duration-secs", type=int, default=60)
           p.add_argument("--parallel", type=int, default=max(1, (os.cpu_count() or 2) - 1))
           p.add_argument("--pipeline-depth", type=int, default=16)
           p.add_argument("--seed", type=int, default=0xCAFEBABE)
           p.add_argument("--zipf-alpha", type=float, default=1.0)
           p.add_argument("--cardinality", type=int, default=1_000_000)
           p.add_argument("--mixed-event-count", type=int, default=3)
           p.add_argument("--isolation-mode", action="store_true", default=False)
           p.add_argument("--no-ledger", action="store_true", default=False)
           p.add_argument("--server-url", default=None,
                          help="HTTP/TCP URL of an already-running beava server. If omitted, "
                               "the harness spawns a beava embed server via bv.App() embed mode.")
           return p.parse_args()

       def _spawn_embed_server() -> tuple[subprocess.Popen, str, str]:
           """Spawn the beava-bench-v18 binary which boots ServerV18 + binds ephemeral ports.

           Returns (process, http_url, tcp_url). Caller is responsible for terminating the
           process after the bench completes.
           """
           # Prefer beava-bench-v18 because it doesn't touch the apply runtime — for production
           # measurement we want the real `beava-server` binary, but Phase 19's smoke test
           # only needs ANY ServerV18 instance.
           # If not present, fall back to using bv.App() embed mode which spawns it for us.
           # Implementation note: the Python embed-mode plumbing in beava._embed already does
           # this; we re-use it. For this bench we want to OWN the subprocess so we can spawn
           # it with --no-ledger and known ports.
           # Approach: use bv.App() context manager (the embed helper) directly.
           raise NotImplementedError("Use bv.App() embed mode directly in main() — see comment in main()")

       def _register_pipeline(server_url: str, cfg: dict) -> None:
           """POST the raw register JSON. Uses HTTP regardless of bench transport — register is
           always HTTP."""
           # Convert tcp:// URL to http:// by mapping ports — caller must pass http:// for register.
           assert server_url.startswith("http"), "register must use http:// URL"
           with httpx.Client(base_url=server_url, timeout=30.0) as c:
               r = c.post("/register", content=register_payload(cfg),
                          headers={"Content-Type": "application/json"})
               assert r.status_code == 200, f"register failed: {r.status_code} {r.text}"

       def _worker_loop(args, http_url: str, tcp_url: str, cfg_json: str, n_total: int,
                        worker_idx: int) -> dict:
           """One worker's tight push loop. Returns a dict: {pushed, errors, first_send_ts,
           last_send_ts}."""
           cfg = json.loads(cfg_json)
           extras = extra_fields(cfg)

           # Build per-worker pool slice.
           per_worker_n = (n_total + args.parallel - 1) // args.parallel
           pool_cfg = PoolConfig(
               shape=args.blast_shape,
               wire_format=args.wire_format,
               transport=args.transport,
               cardinality=args.cardinality,
               zipf_alpha=args.zipf_alpha,
               mixed_event_count=args.mixed_event_count,
               seed=args.seed + worker_idx * 0x9E37,
               pipeline_event_name=event_name(cfg),
               pipeline_key_field=key_field(cfg),
               pipeline_extra_fields=extras,
               mixed_event_names=[event_name(cfg)],  # fallback: only one event from this config
           )
           pool = build_pool(pool_cfg, per_worker_n)
           pushed = 0
           errors = 0
           first_send: float | None = None
           last_send: float | None = None

           if args.transport == "tcp":
               # Strict-FIFO TCP: send N frames then read N acks per micro-batch (matches the
               # burst pattern in the Rust harness; we don't bother with continuous pipelining
               # in the Python harness because GIL serializes the read+write anyway — multi-
               # process is the parallelism layer).
               from urllib.parse import urlparse
               u = urlparse(tcp_url)
               sock = socket.create_connection((u.hostname, u.port), timeout=30.0)
               sock.setsockopt(socket.IPPROTO_TCP, socket.TCP_NODELAY, 1)
               try:
                   pdepth = max(1, args.pipeline_depth)
                   idx = 0
                   while idx < per_worker_n:
                       batch = min(pdepth, per_worker_n - idx)
                       send_ts = time.monotonic()
                       if first_send is None:
                           first_send = send_ts
                       for j in range(batch):
                           sock.sendall(pool[idx + j])
                       last_send = time.monotonic()
                       # Read `batch` acks.
                       try:
                           for _ in range(batch):
                               read_frame(sock, max_frame_bytes=16 * 1024 * 1024)
                               pushed += 1
                       except Exception:
                           errors += batch - (pushed - (idx))
                           break
                       idx += batch
               finally:
                   try: sock.close()
                   except OSError: pass
           else:
               # HTTP — POST /push/{event_name} per push (no batching at HTTP level).
               with httpx.Client(base_url=http_url, timeout=30.0) as c:
                   path = f"/push/{event_name(cfg)}"
                   for body_bytes in pool:
                       send_ts = time.monotonic()
                       if first_send is None: first_send = send_ts
                       r = c.post(path, content=body_bytes,
                                  headers={"Content-Type": "application/json"})
                       if r.status_code == 200:
                           pushed += 1
                       else:
                           errors += 1
                       last_send = time.monotonic()

           return {
               "worker_idx": worker_idx,
               "pushed": pushed,
               "errors": errors,
               "first_send_ts": first_send,
               "last_send_ts": last_send,
           }

       def main() -> int:
           args = parse_args()
           cfg = load_pipeline_config(args.pipeline)

           # Resolve server URLs. For Phase 19 default-mode: use bv.App() embed which spawns
           # the beava binary. For CI / explicit mode: use --server-url.
           import beava as bv
           with bv.App() as app:
               # The embed App exposes only a ping API at the moment. We need the raw HTTP
               # and TCP addresses for the bench. The embed harness logs the URLs on the
               # spawned subprocess's stderr; for Phase 19 we read them from the App's
               # internal _transport. Fallback: if --server-url is set, use it directly.
               if args.server_url is not None:
                   http_url = args.server_url if args.server_url.startswith("http") else None
                   tcp_url  = args.server_url if args.server_url.startswith("tcp") else None
                   # If only one URL was given, derive the other when possible. For robustness,
                   # require both via a comma: --server-url http://...,tcp://...
                   if "," in args.server_url:
                       parts = args.server_url.split(",")
                       http_url = next((p for p in parts if p.startswith("http")), None)
                       tcp_url  = next((p for p in parts if p.startswith("tcp")),  None)
               else:
                   # Embed mode: the spawned subprocess prints its bound HTTP+TCP addresses
                   # to stderr. The bv.App() helper currently only stores TCP. For Phase 19
                   # we extend with a small env-var fallback: BEAVA_BENCH_HTTP_URL +
                   # BEAVA_BENCH_TCP_URL set by the spawning script. If absent, fail clearly.
                   http_url = os.environ.get("BEAVA_BENCH_HTTP_URL")
                   tcp_url  = os.environ.get("BEAVA_BENCH_TCP_URL")
                   if http_url is None or tcp_url is None:
                       print(
                           "ERROR: --server-url not set and BEAVA_BENCH_HTTP_URL / "
                           "BEAVA_BENCH_TCP_URL env vars not set; embed-mode auto-discovery "
                           "is not yet implemented. Pass --server-url http://host:p,tcp://host:p",
                           file=sys.stderr,
                       )
                       return 2

               # Register the pipeline (always HTTP).
               _register_pipeline(http_url, cfg)
               # Pre-warm: a dozen pushes serial.
               with httpx.Client(base_url=http_url, timeout=10.0) as c:
                   for i in range(10):
                       c.post(f"/push/{event_name(cfg)}",
                              content=json.dumps({key_field(cfg): "k0"}).encode(),
                              headers={"Content-Type": "application/json"})

               n_total = args.total_events or 0
               cfg_json = json.dumps(cfg)
               t0 = time.monotonic()
               with ProcessPoolExecutor(max_workers=args.parallel) as exe:
                   futs = [exe.submit(_worker_loop, args, http_url, tcp_url, cfg_json, n_total, i)
                           for i in range(args.parallel)]
                   results = [f.result() for f in as_completed(futs)]
               t1 = time.monotonic()

               total_pushed = sum(r["pushed"] for r in results)
               total_errors = sum(r["errors"] for r in results)
               first_sends = [r["first_send_ts"] for r in results if r["first_send_ts"] is not None]
               last_sends  = [r["last_send_ts"]  for r in results if r["last_send_ts"]  is not None]
               wall_clock_ms = int((t1 - t0) * 1000)
               send_drain_ms = int((max(last_sends) - min(first_sends)) * 1000) if first_sends else 0
               ack_lag_ms    = max(0, wall_clock_ms - send_drain_ms)
               eps = total_pushed / max(t1 - t0, 1e-9)

               print(
                   f"beava-blast: invariant_tuple requested={n_total} pushed={total_pushed} "
                   f"acked={total_pushed} errors={total_errors}",
                   file=sys.stderr,
               )
               if args.isolation_mode:
                   print(
                       f"beava-blast: isolation_mode wall_clock_ms={wall_clock_ms} "
                       f"send_drain_ms={send_drain_ms} ack_lag_ms={ack_lag_ms}",
                       file=sys.stderr,
                   )
               print(
                   f"beava-blast: sustained_eps={eps:.0f} parallel={args.parallel} "
                   f"shape={args.blast_shape} transport={args.transport} wire={args.wire_format} "
                   f"pipeline={args.pipeline}",
                   file=sys.stderr,
               )

               if not args.no_ledger:
                   # Markdown row template — Plan 19-05 owns the ledger schema; emit a row
                   # consistent with the rust harness's Plan 19-04/05 schema.
                   from datetime import datetime
                   date = datetime.utcnow().strftime("%Y-%m-%d")
                   lang = "python"
                   mode = "burst"  # Python harness uses burst per worker (multi-process is the parallelism)
                   commit = subprocess.check_output(["git", "rev-parse", "--short", "HEAD"],
                                                    text=True).strip()
                   row = (f"| 19 | {date} | {args.pipeline} | "
                          f"{args.transport}/{args.wire_format} | {args.blast_shape} | "
                          f"{mode} | {lang} | {args.parallel} | {args.pipeline_depth} | "
                          f"{n_total} | {wall_clock_ms} | {send_drain_ms} | {ack_lag_ms} | "
                          f"{eps:.0f} | n/a | n/a | n/a | n/a | {commit} | python harness |")
                   print(row)

               # Sanity invariant: --total-events strict equality.
               if n_total > 0 and total_pushed != n_total:
                   print(f"WARNING: requested={n_total} but pushed={total_pushed}; "
                         f"measurement is incomplete", file=sys.stderr)
                   return 3

           return 0

       if __name__ == "__main__":
           raise SystemExit(main())
       ```
