    **2. Zipfian sampler (deterministic, no rand_distr dep):**

    Use the rejection-sampling recipe from Gray et al. ("Quickly Generating Billion-Record Synthetic Databases"). For alpha=1.0 we apply the well-known hybrid algorithm:

    ```rust
    pub struct ZipfianSampler {
        rng: StdRng,
        alpha: f64,
        k: u64,
        zetan: f64,    // ζ(N, α) — normalising constant
        eta: f64,      // (1 - (2/N)^(1-α)) / (1 - ζ(2,α)/ζ(N,α))
    }

    impl ZipfianSampler {
        pub fn new(alpha: f64, k: u64, seed: u64) -> Self {
            assert!(alpha > 0.0, "alpha must be > 0");
            assert!(k > 0, "k must be > 0");
            let zetan = zeta(k, alpha);
            let zeta2 = zeta(2, alpha);
            let eta = (1.0 - (2.0_f64 / k as f64).powf(1.0 - alpha))
                    / (1.0 - zeta2 / zetan);
            Self {
                rng: StdRng::seed_from_u64(seed),
                alpha, k, zetan, eta,
            }
        }

        /// Returns a rank in `[0, k)`. Rank 0 is the most frequent.
        pub fn sample(&mut self) -> u64 {
            let u: f64 = self.rng.gen();
            let uz = u * self.zetan;
            if uz < 1.0 { return 0; }
            if uz < 1.0 + (0.5_f64).powf(self.alpha) { return 1; }
            let v: f64 = self.rng.gen();
            let rank_f = (self.k as f64) * (self.eta * v - self.eta + 1.0).powf(1.0 / (1.0 - self.alpha));
            (rank_f as u64).min(self.k - 1)
        }
    }

    fn zeta(n: u64, alpha: f64) -> f64 {
        let mut s = 0.0;
        for i in 1..=n {
            s += 1.0 / (i as f64).powf(alpha);
        }
        s
    }
    ```

    Note: this deterministic recipe makes Test 8 (`zipfian_sampler_deterministic`) pass without `rand_distr`. For very large k (≥1M), `zeta()` is O(k) at constructor time but is one-shot setup — fine for our use case where the pool is built once.

    ```rust
    pub fn build_pool(
        shape: BlastShape,
        cfg: &BlastShapeConfig,
        n: u64,
    ) -> Result<Vec<Bytes>, BlastShapeError> {
        match shape {
            BlastShape::Mixed { event_count } if cfg.event_names_for_mixed.len() < event_count => {
                return Err(BlastShapeError::MixedRequiresMultipleEvents);
            }
            BlastShape::Zipfian { alpha, .. } if alpha <= 0.0 => {
                return Err(BlastShapeError::InvalidAlpha);
            }
            BlastShape::Zipfian { cardinality, .. } | BlastShape::Uniform { cardinality }
                if cardinality == 0 => {
                return Err(BlastShapeError::InvalidCardinality);
            }
            _ => {}
        }

        let mut rng = StdRng::seed_from_u64(cfg.seed);
        let mut zipf = if let BlastShape::Zipfian { alpha, cardinality } = shape {
            Some(ZipfianSampler::new(alpha, cardinality, cfg.seed.wrapping_add(0xDEAD)))
        } else { None };

        let mut pool: Vec<Bytes> = Vec::with_capacity(n as usize);

        // Single encode buffer reused per iteration to amortise allocation.
        let mut buf = BytesMut::with_capacity(4 * 1024);

        // Determine the per-iteration key + event-name strategy.
        for seq in 0..n {
            let key_idx: u64 = match shape {
                BlastShape::Fixed => 0,
                BlastShape::Uniform { cardinality } => rng.gen_range(0..cardinality),
                BlastShape::Zipfian { .. } => zipf.as_mut().unwrap().sample(),
                BlastShape::Mixed { .. } => rng.gen_range(0..1_000_000_u64), // mixed varies event, not necessarily key
            };
            let event_name: &str = match shape {
                BlastShape::Mixed { event_count } => {
                    let idx = rng.gen_range(0..event_count);
                    cfg.event_names_for_mixed[idx]
                }
                _ => &cfg.pipeline.event_name,
            };

            // Build the event body. For BlastShape::Fixed, seq=0 every iteration to keep
            // the body bytes byte-identical across the whole pool — proves Test 2.
            let body_seq = if matches!(shape, BlastShape::Fixed) { 0 } else { seq };
            let body = build_event_body(cfg.pipeline, key_idx, body_seq, &mut rng);

            let payload_bytes = match cfg.wire_format {
                WireFormat::Json => {
                    let envelope = serde_json::json!({ "event": event_name, "body": body });
                    serde_json::to_vec(&envelope).expect("json encode")
                }
                WireFormat::Msgpack => {
                    #[derive(Serialize)]
                    struct Envelope<'a> { event: &'a str, body: &'a Value }
                    let env = Envelope { event: event_name, body: &body };
                    rmp_serde::to_vec_named(&env).expect("msgpack encode")
                }
            };
            let ct = match cfg.wire_format { WireFormat::Json => CT_JSON, WireFormat::Msgpack => CT_MSGPACK };

            let frame = Frame {
                op: OP_PUSH,
                content_type: ct,
                payload: Bytes::from(payload_bytes),
            };
            buf.clear();
            encode_frame(&frame, &mut buf);
            pool.push(buf.to_vec().into());
        }

        Ok(pool)
    }

    pub fn build_pool_timed(
        shape: BlastShape,
        cfg: &BlastShapeConfig,
        n: u64,
    ) -> Result<(Vec<Bytes>, Duration), BlastShapeError> {
        let t0 = Instant::now();
        let pool = build_pool(shape, cfg, n)?;
        Ok((pool, t0.elapsed()))
    }

    fn build_event_body(
        pipeline: &PipelineConfig,
        key_idx: u64,
        seq: u64,
        rng: &mut StdRng,
    ) -> Value {
        let mut obj = serde_json::Map::new();
        obj.insert(pipeline.key_field.clone(), Value::String(format!("k{key_idx:08}")));
        obj.insert("event_time".into(), Value::Number((1_000_000 + seq as i64).into()));
        for (field, ty) in &pipeline.extra_fields {
            let v = match ty.as_str().unwrap_or("f64") {
                "f64" => serde_json::json!(rng.gen_range(0.0..1000.0)),
                "i64" => serde_json::json!(rng.gen_range(0_i64..1_000_000)),
                "str" => serde_json::json!(format!("s{}", rng.gen_range(0..1000))),
                _ => serde_json::json!(0),
            };
            obj.insert(field.clone(), v);
        }
        Value::Object(obj)
    }
    ```
