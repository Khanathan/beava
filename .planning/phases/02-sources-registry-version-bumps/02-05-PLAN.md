```rust
use axum::{
    extract::{DefaultBodyLimit, State},
    http::{header, HeaderMap, StatusCode},
    routing::post,
    Json, Router,
};
use beava_core::{
    register_validate::{validate_payload, ValidationError},
    registry::{EventDescriptor, TableDescriptor, DerivationDescriptor, Registry},
    registry_diff::{compute_diff, PayloadNode, RegistryDiff, ConflictDetail},
};
use serde::{Deserialize, Serialize};
use std::sync::Arc;
use tracing::{info, warn};

/// The wire shape of `POST /register` body. Using `PayloadNode` directly via serde's
/// internally-tagged enum means the `kind` field discriminates at the type level.
#[derive(Debug, Deserialize)]
pub struct RegisterPayload {
    pub nodes: Vec<PayloadNode>,
}

/// Shared state given to the axum handler.
#[derive(Clone)]
pub struct RegisterAppState {
    pub registry: Arc<Registry>,
}

#[derive(Debug, Serialize)]
pub struct RegisterSuccess {
    pub status: &'static str,                        // always "ok"
    pub registry_version: u64,
    pub registered_descriptors: Vec<String>,         // input order of payload nodes
    pub added: Vec<String>,
    pub already_present: Vec<String>,
}

#[derive(Debug, Serialize)]
pub struct RegisterErrorBody {
    pub error: RegisterError,
    pub registry_version: u64,
}

#[derive(Debug, Serialize)]
#[serde(untagged)]
pub enum RegisterError {
    Validation {
        code: &'static str,                          // "invalid_registration"
        path: String,
        reason: String,
    },
    Conflict {
        code: &'static str,                          // "registration_conflict"
        message: &'static str,
        diff: ResponseDiff,                          // {added, removed: [], changed}
    },
}

#[derive(Debug, Serialize)]
pub struct ResponseDiff {
    pub added: Vec<String>,
    pub removed: Vec<String>,                        // always [] in v0
    pub changed: Vec<ConflictDetail>,                // from registry_diff
}

pub fn register_router(state: RegisterAppState) -> Router {
    Router::new()
        .route("/register", post(post_register))
        .layer(DefaultBodyLimit::max(1 * 1024 * 1024))   // 1 MiB cap
        .with_state(state)
}

async fn post_register(
    headers: HeaderMap,
    State(state): State<RegisterAppState>,
    body: axum::body::Bytes,                          // raw bytes so we control JSON error paths
) -> (StatusCode, Json<serde_json::Value>);
```

### Request handling pipeline (pseudocode)

```rust
async fn post_register(headers, State(state), body) -> (StatusCode, Json<Value>) {
    // 1. Content-Type check (SRV-API-11)
    let ct = headers.get(header::CONTENT_TYPE);
    if !is_json_content_type(ct) {
        return (StatusCode::UNSUPPORTED_MEDIA_TYPE, json_error(415, "unsupported_media_type", "<header>.content_type", "expected application/json"));
    }

    // 2. JSON parse → 400 with json-pointer path on failure
    let payload: RegisterPayload = match serde_json::from_slice(&body) {
        Ok(p) => p,
        Err(e) => {
            let (path, reason) = format_serde_error(&e);
            warn!(kind = "register.parse_error", path = %path, reason = %reason, "malformed register payload");
            return (StatusCode::BAD_REQUEST, json_error_body(&state, "invalid_registration", &path, &reason));
        }
    };

    // 3. Snapshot current for validation + diff
    let current_snapshot = state.registry.snapshot();

    // 4. Validate (Plan 04)
    let validated = match validate_payload(&current_snapshot, payload.nodes) {
        Ok(v) => v,
        Err(errs) => {
            let first = &errs[0];
            warn!(kind = "register.validation_failed", path = %first.path, code = ?first.code, error_count = errs.len(), "register validation failed");
            return (StatusCode::BAD_REQUEST, json_error_body_with_version(&state, "invalid_registration", &first.path, &first.reason));
        }
    };

    let nodes = validated.into_inner();
    let descriptor_names: Vec<String> = nodes.iter().map(|n| n.name().to_string()).collect();

    // 5. Diff (Plan 03)
    let diff: RegistryDiff = compute_diff(&current_snapshot, &nodes);

    // 6. Conflict → 409 (no mutation)
    if !diff.changed.is_empty() {
        warn!(kind = "register.conflict", version = current_snapshot.version, changed = ?diff.changed.iter().map(|c| &c.name).collect::<Vec<_>>(), "register conflict");
        let body = RegisterErrorBody {
            error: RegisterError::Conflict {
                code: "registration_conflict",
                message: "Registration would change or remove existing descriptors",
                diff: ResponseDiff {
                    added: diff.added,
                    removed: Vec::new(),
                    changed: diff.changed,
                },
            },
            registry_version: current_snapshot.version,
        };
        return (StatusCode::CONFLICT, Json(serde_json::to_value(body).unwrap()));
    }

    // 7. No-op? (only already_present, no added) → 200 with SAME version
    if diff.added.is_empty() {
        let resp = RegisterSuccess {
            status: "ok",
            registry_version: current_snapshot.version,
            registered_descriptors: descriptor_names,
            added: Vec::new(),
            already_present: diff.already_present,
        };
        info!(kind = "register.noop", version = current_snapshot.version, nodes = resp.registered_descriptors.len(), "register no-op");
        return (StatusCode::OK, Json(serde_json::to_value(resp).unwrap()));
    }

    // 8. Additive install (Plan 01 Registry helper) — atomic under write lock
    let new_version = state.registry.apply_registration(nodes)?;
    info!(kind = "register.success", version = new_version, added = ?diff.added, already_present_count = diff.already_present.len(), "register succeeded");
    let resp = RegisterSuccess {
        status: "ok",
        registry_version: new_version,
        registered_descriptors: descriptor_names,
        added: diff.added,
        already_present: diff.already_present,
    };
    (StatusCode::OK, Json(serde_json::to_value(resp).unwrap()))
}
```

### Registry::apply_registration — new public method on Registry (this plan adds it)

```rust
// In crates/beava-core/src/registry.rs — add as a public method:
impl Registry {
    /// Atomically install a batch of already-validated, non-conflicting PayloadNodes.
    /// Bumps version by 1 and stamps each new descriptor with registered_at_version = new_version.
    /// Existing (already_present) descriptors are left as-is.
    ///
    /// Precondition: `nodes` has passed `validate_payload` and `compute_diff` yielded
    /// `changed = []` + `added != []`. Caller (Plan 05 endpoint) enforces this.
    pub fn apply_registration(&self, nodes: Vec<PayloadNode>) -> u64 {
        let mut w = self.inner.write();
        let new_version = w.version + 1;
        for n in nodes {
            match n {
                PayloadNode::Event(mut e) => {
                    if !w.events.contains_key(&e.name) {
                        e.registered_at_version = new_version;
                        w.events.insert(e.name.clone(), e);
                    }
                }
                PayloadNode::Table(mut t) => { /* similar */ }
                PayloadNode::Derivation(mut d) => { /* similar */ }
            }
        }
        w.version = new_version;
        new_version
    }
}
```

### Content-Type matching rule

`is_json_content_type(ct)` returns true if the header value's media type (before `;`) equals `application/json` (case-insensitive, trimmed). `application/json; charset=utf-8` passes. `text/plain`, `application/xml`, empty header → false.

### JSON parse error → path extraction

`serde_json::Error` exposes `line()` and `column()` but NOT a JSON pointer. We format:
- Path: `$.<whatever-serde-said>` using `err.to_string()` which often includes the field name
- For simplicity, the path for parse errors is a best-effort: if the error contains `missing field \`X\``, path becomes `nodes` or `nodes[i].X` depending on whether we can parse a partial JSON (we can't easily). v0 returns `path = "<body>"` or the line:column marker; reason string carries the full serde_json::Error.

**Decision (v0):** Parse errors return `path = "<body>"` and `reason = err.to_string()`. Future work (Phase 3+) can parse into a tolerant intermediate type to extract better paths. SDK tests that assert on parse-error paths are out of scope for Phase 2 — the SDK is responsible for sending well-formed JSON.

| HTTP client → axum handler | Hostile client can send malformed JSON, oversized bodies, wrong Content-Type, malicious node names |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-05-01 | DoS | Oversized request body | mitigate | `DefaultBodyLimit::max(1 MiB)` layered on the router. Axum responds 413 before handler runs. Unit test confirms. |
| T-02-05-02 | DoS | Lock contention from register storm | accept | RwLock; registrations are admin-path. Phase 5 benchmarks this — if a hot-spot appears, swap to ArcSwap<RegistryInner>. Phase 2 keeps RwLock for simplicity. |
| T-02-05-03 | Tampering | Content-Type bypass | mitigate | `is_json_content_type` strictly checks `application/json` media type. Unit tests cover text/plain rejection + charset acceptance. |
| T-02-05-04 | Information disclosure | Error reason reflection | mitigate | Reasons from ValidationError are server-constructed (not client-reflected). Only descriptor NAMES from the payload appear in details — and names have been validated against `[A-Za-z_][A-Za-z0-9_]*` before reaching the response serializer. |
| T-02-05-05 | Repudiation | Register audit trail | mitigate | Every register attempt (success, validation failure, conflict) emits a structured tracing log with kind/version/error-code. Phase 6 WAL will add persistence. |
| T-02-05-06 | Elevation of privilege | Auth bypass | accept | v0 has NO authentication — single-tenant single-operator design (PROJECT.md). Commercial tier adds auth later. |
| T-02-05-07 | Tampering | Partial install on panic | mitigate | apply_registration takes the write lock and performs inserts under it; a panic mid-loop leaves the RegistryInner in an inconsistent state. For Phase 2 this is acceptable (single-tenant, in-memory, restart wipes state). Phase 7 (snapshot) formalizes transactional boundaries. NOTE: we install into `w.events`, `w.tables`, `w.derivations` then bump `w.version` LAST so a panic mid-install leaves version at old value — partial inserts visible to readers but version-tagged-as-old, mitigating most of the hazard. |
