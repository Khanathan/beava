## Setup

### 1. Build release binary

```bash
cd /data/home/tally
export PATH="/data/home/.cargo/bin:$PATH"
cargo build --release
```

### 2. Kill any stale servers on 6400/6401

```bash
lsof -ti:6400,6401 | xargs -r kill -9 2>/dev/null
```

### 3. Start the server

```bash
./target/release/tally &>/tmp/tally-server.log &
```

Confirm it's listening:

```bash
sleep 1 && curl -s http://localhost:6401/health
# Expected: {"status":"ok"}
```

### 4. Register a composable pipeline + push test traffic

```bash
cd /data/home/tally/python && python3 -c "
import tally as st
import time

app = st.App('localhost:6400')

@st.stream(key='user_id')
class Transactions:
    tx_count_1h = st.count(window='1h')
    tx_sum_1h = st.sum('amount', window='1h')
    avg_amount_1h = st.avg('amount', window='1h')

@st.stream(key='user_id')
class Logins:
    login_count_1h = st.count(window='1h')

@st.view(key='user_id')
class UserRisk:
    ratio = st.derive('Transactions.tx_count_1h / Logins.login_count_1h')

app.register(Transactions, Logins, UserRisk)

# Push 100 txns across 10 users
for i in range(100):
    app.push(Transactions, {
        'user_id': f'user_{i % 10}',
        'amount': float(i),
    })

# Push 50 logins across 10 users
for i in range(50):
    app.push(Logins, {'user_id': f'user_{i % 10}'})

# GET queries
for i in range(20):
    app.get(f'user_{i % 10}')

# SET batch
for i in range(10):
    app.set(f'user_{i}', {'segment': 'A'})

print('Pushed 100 txns, 50 logins, 20 GETs, 10 SETs')
"
```

### 5. Open browser

Navigate to: **http://localhost:6401/**
