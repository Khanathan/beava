```rust
// Slow query: extract key preview (first 32 chars of entity key)

let key_preview = {
    let kf = app.engine.get_stream(&stream_name).and_then(|s| s.key_field.clone());
    if let Some(ref kf) = kf {
        if let Some(serde_json::Value::String(k)) = payload.get(kf.as_str()) {

            kp
        } else { String::new() }
    } else { String::new() }

    crate::server::latency::CommandKind::Push,
```

Note: The key_preview extraction happens AFTER the push logic, inside the same lock scope. **6. Instrument MSET path** (in handle_connection, around lines 169-172):
MSET is async (handle_mset uses cooperative yielding). Capture timing BEFORE the match dispatch. After handle_mset returns, acquire a FRESH lock to record latency. CRITICAL: no .await across the recording lock.

Restructure the dispatch block in handle_connection:
```rust
let cmd_start = std::time::Instant::now();
let is_mset = matches!(&cmd, Command::Mset { .. });
let response = match cmd {
    Command::Mset { entries } => handle_mset(entries, &state).await,
    other => handle_sync_command(other, &state),
};
// Record MSET latency AFTER async completion, in a separate lock
if is_mset {
    let mset_us = cmd_start.elapsed().as_secs_f64() * 1_000_000.0;

```

```rust
#[tokio::test]
async fn debug_latency_returns_valid_json_with_all_sections() {
    let (port, _state) = start_debug_ui_server().await;

    // Hit the endpoint
    let response = http_get(port, "/debug/latency").await;
    assert!(response.contains("200 OK"), "Expected 200 OK");

    // Parse JSON body
    let body = extract_json_body(&response);
    let json: serde_json::Value = serde_json::from_str(&body).expect("valid JSON");

    // Verify top-level structure
    let per_command = json["per_command"].as_array().expect("per_command array");
    assert_eq!(per_command.len(), 4, "should have 4 command entries (PUSH/GET/SET/MSET)");

    // Each command entry has required fields
    for entry in per_command {
        assert!(entry["command"].is_string());
        assert!(entry["count"].is_number());
        assert!(entry["p50_us"].is_number());
        assert!(entry["p95_us"].is_number());
        assert!(entry["p99_us"].is_number());
        assert!(entry["histogram"]["bin_edges_us"].is_array());
        assert!(entry["histogram"]["counts"].is_array());
    }

    // per_stream is an array (may be empty with no pushes)
    assert!(json["per_stream"].is_array());

    // slow_queries is an array (may be empty)
    assert!(json["slow_queries"].is_array());
}
```

Use the existing `http_get` and `extract_json_body` test helpers from test_debug_ui.rs. If `extract_json_body` doesn't exist, parse the HTTP response manually by splitting on the blank line separating headers from body (same pattern used in existing tests).
