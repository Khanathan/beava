       ```python
       from dataclasses import dataclass, field
       from typing import Any
       import re

       _VALID_OPS = frozenset({"count", "sum", "avg", "min", "max", "variance", "stddev", "ratio"})
       _WINDOW_PATTERN = re.compile(r"^(?:\d+(?:ms|s|m|h|d)|forever)$")

       def _validate_window(window: str | None, op: str, requires_window: bool) -> None:
           if window is None:
               if requires_window:
                   raise ValueError(f"window is required for {op!r} operator")
               return
           if not _WINDOW_PATTERN.match(window):
               raise ValueError(
                   f"window={window!r} must match regex \\d+(ms|s|m|h|d) or 'forever'"
               )

       def _serialize_where(where) -> str | None:
           if where is None:
               return None
           # Duck-typed: anything with .to_expr_string() is accepted (Phase 4 _ExprAST).
           if not hasattr(where, "to_expr_string"):
               raise TypeError(f"where= must be a bv.col expression; got {type(where).__name__}")
           return where.to_expr_string()

       @dataclass(frozen=True)
       class AggDescriptor:
           op: str
           field: str | None = None
           window: str | None = None
           where: str | None = None

           def to_agg_spec(self) -> dict:
               params: dict[str, Any] = {}
               if self.field is not None:
                   params["field"] = self.field
               if self.window is not None:
                   params["window"] = self.window
               if self.where is not None:
                   params["where"] = self.where
               return {"op": self.op, "params": params}

       def count(*, window: str | None = None, where=None) -> AggDescriptor:
           _validate_window(window, "count", requires_window=False)
           return AggDescriptor(op="count", field=None, window=window, where=_serialize_where(where))

       def sum(field: str, *, window: str, where=None) -> AggDescriptor:
           _validate_window(window, "sum", requires_window=True)
           return AggDescriptor(op="sum", field=field, window=window, where=_serialize_where(where))

       def avg(field: str, *, window: str, where=None) -> AggDescriptor:
           _validate_window(window, "avg", requires_window=True)
           return AggDescriptor(op="avg", field=field, window=window, where=_serialize_where(where))

       def min(field: str, *, window: str, where=None) -> AggDescriptor:
           _validate_window(window, "min", requires_window=True)
           return AggDescriptor(op="min", field=field, window=window, where=_serialize_where(where))

       def max(field: str, *, window: str, where=None) -> AggDescriptor:
           _validate_window(window, "max", requires_window=True)
           return AggDescriptor(op="max", field=field, window=window, where=_serialize_where(where))

       def variance(field: str, *, window: str, where=None) -> AggDescriptor:
           _validate_window(window, "variance", requires_window=True)
           return AggDescriptor(op="variance", field=field, window=window, where=_serialize_where(where))

       def stddev(field: str, *, window: str, where=None) -> AggDescriptor:
           _validate_window(window, "stddev", requires_window=True)
           return AggDescriptor(op="stddev", field=field, window=window, where=_serialize_where(where))

       def ratio(*, window: str | None = None, where=None) -> AggDescriptor:
           _validate_window(window, "ratio", requires_window=False)
           return AggDescriptor(op="ratio", field=None, window=window, where=_serialize_where(where))

       class GroupBy:
           def __init__(self, upstream, keys: list[str]) -> None:
               self._upstream = upstream
               self._keys = keys

           def agg(self, **named_features: AggDescriptor) -> "TableDerivation":
               # Validate: every value is an AggDescriptor; feature names are strings
               for name, desc in named_features.items():
                   if not isinstance(desc, AggDescriptor):
                       raise TypeError(f"agg(...) kwarg {name!r} must be an AggDescriptor (from bv.count/sum/...), got {type(desc).__name__}")

               # Build REGISTER JSON GroupBy opnode
               agg_map = {name: desc.to_agg_spec() for name, desc in named_features.items()}
               op_node = {"op": "group_by", "keys": list(self._keys), "agg": agg_map}

               # Construct a TableDerivation with the groupby op appended
               from ._tables import TableDerivation  # avoid circular import
               upstream_name = getattr(self._upstream, "_name", None)
               if upstream_name is None:
                   raise TypeError("group_by() upstream must have a _name attribute (EventSource/EventDerivation)")
               existing_ops = list(getattr(self._upstream, "_ops", []))
               return TableDerivation(
                   name=f"{upstream_name}_grouped_{''.join(self._keys)}",  # temp name; usually overridden via @bv.table
                   schema=self._upstream._schema,  # placeholder; server-propagated schema is authoritative
                   upstreams=[upstream_name],
                   ops=[*existing_ops, op_node],
                   output_kind="table",
                   table_primary_key=list(self._keys),
                   upstream=self._upstream,
               )
       ```
