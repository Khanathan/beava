3. `retract(key, target_lsn, retract_lsn, wall_ms)`:

   - If `target_lsn` is not in the chain's keys → `Err(TargetNotFound)`.
   - Walk `chain.range(target_lsn..).map(|(_, v)| v)` to detect existing `Retracted{undo_of: target_lsn, ..}` → `Err(AlreadyRetracted)`.

4. `lookup_at_lsn(key, as_of_lsn)`:

5. `sweep_retention(now_wall_ms, retention_ms)`: for each chain, drain entries where `now_wall_ms.saturating_sub(version.wall_ms()) > retention_ms`. Return total dropped count. Helper `MvccVersion::wall_ms(&self) -> u64` reads the embedded wall_ms field.

//! Phase 11.5 — temporal tables + retraction smoke test.
use beava_server::testing::TestServer;
use serde_json::json;

#[tokio::test]
async fn temporal_table_upsert_retract_returns_prior_value() {
    let srv = TestServer::start_with_defaults().await;

    // 1. Register a temporal table.
    let reg = json!({
        "events": [],
        "tables": [{
            "name": "merch",
            "schema": {"fields": {"mid": "Str", "score": "I64"}, "optional_fields": []},
            "key": ["mid"],
            "mode": "upsert",
            "temporal": true,
            "retention_ms": 3_600_000
        }],
        "derivations": []
    });
    let r = srv.client.post(srv.url("/register")).json(&reg).send().await.unwrap();
    assert_eq!(r.status(), 200, "register: {}", r.text().await.unwrap());

    // 2. Upsert at t=0 -> ack_lsn captured.
    let r0 = srv.client.post(srv.url("/push-table/merch")).json(&json!({"mid": "m1", "score": 100})).send().await.unwrap();
    assert_eq!(r0.status(), 200);
    let ack0: serde_json::Value = r0.json().await.unwrap();
    let lsn0 = ack0["ack_lsn"].as_u64().unwrap();

    // 3. Upsert at t=1 -> distinct ack_lsn.
    let r1 = srv.client.post(srv.url("/push-table/merch")).json(&json!({"mid": "m1", "score": 200})).send().await.unwrap();
    assert_eq!(r1.status(), 200);
    let ack1: serde_json::Value = r1.json().await.unwrap();
    let lsn1 = ack1["ack_lsn"].as_u64().unwrap();
    assert!(lsn1 > lsn0);

    // 4. GET current row -> score=200.
    let g_now = srv.client.get(srv.url(&format!("/table/merch?key=m1"))).send().await.unwrap();
    assert_eq!(g_now.status(), 200);
    let v_now: serde_json::Value = g_now.json().await.unwrap();
    assert_eq!(v_now["row"]["score"], 200);

    // 5. GET as_of=lsn0 -> score=100.
    let g_past = srv.client.get(srv.url(&format!("/table/merch?key=m1&as_of={}", lsn0))).send().await.unwrap();
    assert_eq!(g_past.status(), 200);
    let v_past: serde_json::Value = g_past.json().await.unwrap();
    assert_eq!(v_past["row"]["score"], 100);

    // 6. POST /retract event_id=lsn1 -> 200.
    let r_retract = srv.client.post(srv.url("/retract")).json(&json!({"event_id": lsn1})).send().await.unwrap();
    assert_eq!(r_retract.status(), 200, "retract: {}", r_retract.text().await.unwrap());

    // 7. GET current row -> score=100 (lsn0 restored).
    let g_after = srv.client.get(srv.url(&format!("/table/merch?key=m1"))).send().await.unwrap();
    assert_eq!(g_after.status(), 200);
    let v_after: serde_json::Value = g_after.json().await.unwrap();
    assert_eq!(v_after["row"]["score"], 100);

    // 8. GET as_of=lsn0 -> still score=100 (history before retraction preserved).
    let g_past2 = srv.client.get(srv.url(&format!("/table/merch?key=m1&as_of={}", lsn0))).send().await.unwrap();
    assert_eq!(g_past2.status(), 200);
    let v_past2: serde_json::Value = g_past2.json().await.unwrap();
    assert_eq!(v_past2["row"]["score"], 100);
}
