    ```rust
    //! CLI driver for `tally query` / `tally inspect`. Pure Rust — no Python.
    use crate::client::{Session, ClientConfig, Scope, Mode};

    pub struct QueryParams { /* remote, streams, keys, key_prefix, token, lookup_stream, lookup_key, since */ }
    pub struct InspectParams { /* remote, streams, keys, key_prefix, token, since */ }

    pub fn run_query(p: QueryParams) -> Result<(), CliError> {
        let token = p.token.or_else(|| std::env::var("TALLY_TOKEN").ok());
        let scope = Scope::new(p.streams, p.keys, p.key_prefix);
        let cfg = ClientConfig { remote: p.remote, token, scope, mode: Mode::Historical, since: p.since };
        let mut session = Session::new(cfg)?;
        session.run()?;
        match session.state_store().get(&p.lookup_stream, &p.lookup_key)? {
            Some(value) => {
                println!("{}", serde_json::to_string(&value)?);
                Ok(())
            }
            None => {
                println!("null");
                Ok(())
            }
        }
    }

    pub fn run_inspect(p: InspectParams) -> Result<(), CliError> {
        let token = p.token.or_else(|| std::env::var("TALLY_TOKEN").ok());
        let scope = Scope::new(p.streams, p.keys, p.key_prefix);
        let cfg = ClientConfig { remote: p.remote, token, scope, mode: Mode::Historical, since: p.since };
        let mut session = Session::new(cfg)?;
        session.run()?;
        let counts = session.state_store().inspect();
        println!("{}", serde_json::to_string(&counts)?);
        Ok(())
    }
    ```

    ```python
    import os, socket, subprocess, time
    from pathlib import Path
    import pytest

    REPO_ROOT = Path(__file__).resolve().parents[3]
    SERVER_BIN = REPO_ROOT / "target" / "release" / "tally"  # or wherever the v0 server binary builds
    CLI_BIN    = REPO_ROOT / "target" / "release" / "tally_cli"

    def _free_port() -> int:
        with socket.socket() as s:
            s.bind(("127.0.0.1", 0))
            return s.getsockname()[1]

    def _wait_for_port(host, port, timeout=10.0):
        deadline = time.time() + timeout
        while time.time() < deadline:
            try:
                with socket.create_connection((host, port), timeout=0.5):
                    return
            except OSError:
                time.sleep(0.1)
        raise TimeoutError(f"server did not listen on {host}:{port}")

    @pytest.fixture
    def live_tally_server(tmp_path):
        port = _free_port()
        # Flags/env mirror existing test_push_table_e2e.py conventions
        env = {**os.environ, "TALLY_DATA_DIR": str(tmp_path)}
        proc = subprocess.Popen(
            [str(SERVER_BIN), "--listen", f"127.0.0.1:{port}"],
            env=env, stdout=subprocess.PIPE, stderr=subprocess.PIPE,
        )
        try:
            _wait_for_port("127.0.0.1", port, timeout=15.0)
            yield type("Srv", (), {"remote": f"127.0.0.1:{port}", "proc": proc, "data_dir": tmp_path})()
        finally:
            proc.terminate()
            try: proc.wait(timeout=5)
            except subprocess.TimeoutExpired: proc.kill()
    ```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-30-06 | Information disclosure | `tally query` printing a token in error messages | mitigate | `CliError` Display impl must scrub token fields from `ClientError::Handshake`. Inline test asserts error strings don't contain the string "Bearer" or token-shaped substrings. |
| T-30-07 | Tampering | Out-of-scope query silently succeeding via CLI | mitigate | CLI maps `StateError::OutOfScope` to a non-zero exit + "OutOfScope" in stderr. Asserted in E2E test `test_query_out_of_scope_exits_nonzero`. |
| T-30-08 | Denial of service | pytest hangs forever if server startup wedges | mitigate | `_wait_for_port` fixture has 15s timeout; per-test `--timeout=60` via pytest-timeout. |
| T-30-09 | Spoofing | E2E test running against a stale binary | mitigate | CI step `cargo build --release --features client --bins` runs before pytest; fixture resolves `CLI_BIN` path from the repo root, not `$PATH`. |
| T-30-10 | Repudiation | Test flake from timing, blamed as real bug | accept | E2E uses a deterministic `client.flush()` barrier before `Pipeline.run`; if a real flake surfaces, the E2E test owner quarantines via pytest mark, not sleeps. |
