```rust
#[derive(Serialize)]
pub struct ShardInfo {
    pub id: usize,
    pub inbox_depth: u64,
    pub reactor_utilization: f64,   // 0.0..=1.0 EWMA from per-shard metrics
    pub keys_owned: u64,
    pub watermark_lag_seconds: f64,
    pub events_total: u64,
    pub inbox_full_total: u64,
    pub down: bool,
}

#[derive(Serialize)]
pub struct HotShardEntry {
    pub shard: usize,
    pub keys_owned: u64,
    pub fleet_mean: f64,
    pub ratio: f64,
}

#[derive(Serialize)]
pub struct ShardDiagnosticsReport {
    pub shard_count: usize,
    pub shards: Vec<ShardInfo>,
    pub hot_shards: Vec<HotShardEntry>,
    pub ready: bool,
}

pub struct HotShardConfig {
    pub threshold: f64,  // default 1.5, clamped 1.1..=10.0
}

impl HotShardConfig {
    pub fn from_env() -> Self;  // reads BEAVA_HOT_SHARD_THRESHOLD
}
```
